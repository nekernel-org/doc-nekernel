138 bool grant_or_fail(const String permission_name)
139 {
140 if (std::find(kPermsList.cbegin(), kPermsList.cend(), permission_name) != kPermsList.end())
141 return true;
142
143 if (this->prompt("Photon - Security", "The page is asking for: " + permission_name + "\nAccept?") == "NO")
144 return false;
145
146 kPermsList.push_back(permission_name);
