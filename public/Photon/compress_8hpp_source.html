52 // The windowBits parameter is the base two logarithm of the window size (the size of the history buffer).
53 // It should be in the range 8..15 for this version of the library.
54 // Larger values of this parameter result in better compression at the expense of memory usage.
55 // This range of values also changes the decoding type:
56 // -8 to -15 for raw deflate
57 // 8 to 15 for zlib
58 // (8 to 15) + 16 for gzip
59 // (8 to 15) + 32 to automatically detect gzip/zlib header (decompression/inflate only)
60 constexpr int window_bits = 15 + 16; // gzip with windowbits of 15
61
62 constexpr int mem_level = 8;
63 // The memory requirements for deflate are (in bytes):
64 // (1 << (window_bits+2)) + (1 << (mem_level+9))
65 // with a default value of 8 for mem_level and our window_bits of 15
66 // this is 128Kb
67
68#pragma GCC diagnostic push
69#pragma GCC diagnostic ignored "-Wold-style-cast"
70 if (deflateInit2(&deflate_s, level_, Z_DEFLATED, window_bits, mem_level, Z_DEFAULT_STRATEGY) != Z_OK)
71 {
72 throw std::runtime_error("deflate init failed");
73 }
74#pragma GCC diagnostic pop
75
76 deflate_s.next_in = reinterpret_cast<z_const Bytef*>(data);
77 deflate_s.avail_in = static_cast<unsigned int>(size);
78
79 std::size_t size_compressed = 0;
80 do
81 {
82 size_t increase = size / 2 + 1024;
83 if (output.size() < (size_compressed + increase))
84 {
85 output.resize(size_compressed + increase);
86 }
87 // There is no way we see that "increase" would not fit in an unsigned int,
88 // hence we use static cast here to avoid -Wshorten-64-to-32 error
89 deflate_s.avail_out = static_cast<unsigned int>(increase);
90 deflate_s.next_out = reinterpret_cast<Bytef*>((&output[0] + size_compressed));
91 // From http://www.zlib.net/zlib_how.html
92 // "deflate() has a return value that can indicate errors, yet we do not check it here.
93 // Why not? Well, it turns out that deflate() can do no wrong here."
94 // Basically only possible error is from deflateInit not working properly
95 deflate(&deflate_s, Z_FINISH);
96 size_compressed += (increase - deflate_s.avail_out);
97 } while (deflate_s.avail_out == 0);
98
99 deflateEnd(&deflate_s);
