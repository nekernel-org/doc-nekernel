606 .find(kLdDefineSymbol) != CompilerKit::STLString::npos &&
607 CompilerKit::STLString(command_headers[sub_command_header_index].Name)
608 .find(kLdDynamicSym) == CompilerKit::STLString::npos) {
609 if (kVerbose) {
610 kConsoleOut << "Ignoring :UndefinedSymbol: headers...\n";
611 }
612
613 // ignore :UndefinedSymbol: headers, they do not contain code.
614 continue;
615 }
616
617 auto& command_hdr = command_headers[sub_command_header_index];
618
619 if (command_hdr.Name == CompilerKit::STLString(command_headers[commandHeaderIndex].Name)) {
620 if (std::find(dupl_symbols.cbegin(), dupl_symbols.cend(), command_hdr.Name) ==
621 dupl_symbols.cend()) {
622 dupl_symbols.emplace_back(command_hdr.Name);
623 }
624
625 if (kVerbose) kConsoleOut << "Found duplicate symbols of: " << command_hdr.Name << "\n";
626
627 kDuplicateSymbols = true;
628 }
629 }
630 }
631
632 if (!dupl_symbols.empty()) {
633 for (auto& symbol : dupl_symbols) {
634 kConsoleOut << "Multiple symbols of: " << symbol << " detected, cannot continue.\n";
635 }
636
637 return NECTI_EXEC_ERROR;
638 }
639
640 // step 2.5: write program bytes.
641
642 for (auto& struct_of_blob : kObjectBytes) {
643 output_fc.write(struct_of_blob.mBlob.data(), struct_of_blob.mBlob.size());
644 }
645
646 if (kVerbose) {
647 kConsoleOut << "Wrote contents of: " << kOutput << "\n";
648 }
649
650 // step 3: check if we have those symbols
651
652 std::vector<CompilerKit::STLString> unreferenced_symbols;
653
654 for (auto& command_hdr : command_headers) {
655 if (auto it =
656 std::find(not_found.begin(), not_found.end(), CompilerKit::STLString(command_hdr.Name));
657 it != not_found.end()) {
658 unreferenced_symbols.emplace_back(command_hdr.Name);
659 }
660 }
661
662 if (!unreferenced_symbols.empty()) {
663 for (auto& unreferenced_symbol : unreferenced_symbols) {
664 kConsoleOut << "Undefined symbol " << unreferenced_symbol << "\n";
665 }
666
667 return NECTI_EXEC_ERROR;
668 }
669
670 if ((!kStartFound || kDuplicateSymbols) &&
671 (std::filesystem::exists(kOutput) || !unreferenced_symbols.empty())) {
672 if (kVerbose) {
673 kConsoleOut << "File: " << kOutput << " is corrupt now...\n";
674 }
675
676 return NECTI_EXEC_ERROR;
677 }
678
679 return NECTI_SUCCESS;
680}
