51// Avoids relative_path which could discard parts of the original.
52std::filesystem::path expand_home(const std::filesystem::path& input) {
53 const std::string& raw = input.string();
54
55 if (!raw.empty() && raw[0] == '~') {
56 const char* home = std::getenv("HOME");
57 if (!home) home = std::getenv("USERPROFILE");
58
59 if (!home) throw std::runtime_error("Home directory not found in environment variables");
60
61 return std::filesystem::path(home) / raw.substr(1);
62 }
63
64 return input;
65}

173CompilerKit::SyntaxLeafList::SyntaxLeaf CompilerFrontendCPlusPlusAMD64::Compile(
174 CompilerKit::STLString text, CompilerKit::STLString file) {
175 CompilerKit::SyntaxLeafList::SyntaxLeaf syntax_tree;
176
177 if (text.length() < 1) return syntax_tree;
178
179 std::size_t index = 0UL;
180 std::vector<std::pair<CompilerKit::CompilerKeyword, std::size_t>> keywords_list;
181
182 for (auto& keyword : kKeywords) {
183 if (text.find(keyword.keyword_name) != std::string::npos) {
184 switch (keyword.keyword_kind) {
185 case CompilerKit::kKeywordKindCommentInline: {
186 break;
187 }
188 default:
189 break;
190 }
191
192 std::size_t pos = text.find(keyword.keyword_name);
193 if (pos == std::string::npos) continue;
194
195 // can't go before start of string
196 if (pos > 0 && text[pos - 1] == '+' &&
197 keyword.keyword_kind == CompilerKit::kKeywordKindVariableAssign)
198 continue;
199
200 if (pos > 0 && text[pos - 1] == '-' &&
201 keyword.keyword_kind == CompilerKit::kKeywordKindVariableAssign)
202 continue;
203
204 // don't go out of range
205 if ((pos + keyword.keyword_name.size()) < text.size() &&
206 text[pos + keyword.keyword_name.size()] == '=' &&
207 keyword.keyword_kind == CompilerKit::kKeywordKindVariableAssign)
208 continue;
209
210 keywords_list.emplace_back(std::make_pair(keyword, index));
211 ++index;
212 }
213 }
214
215 for (auto& keyword : keywords_list) {
216 if (text.find(keyword.first.keyword_name) == CompilerKit::STLString::npos) continue;
217
218 switch (keyword.first.keyword_kind) {
219 case CompilerKit::KeywordKind::kKeywordKindClass: {
220 ++kOnClassScope;
221 break;
222 }
223 case CompilerKit::KeywordKind::kKeywordKindIf: {
224 std::size_t keywordPos = text.find(keyword.first.keyword_name);
225 std::size_t openParen = text.find("(", keywordPos);
226 std::size_t closeParen = text.find(")", openParen);
227
228 if (keywordPos == CompilerKit::STLString::npos ||
229 openParen == CompilerKit::STLString::npos ||
230 closeParen == CompilerKit::STLString::npos || closeParen <= openParen) {
231 Detail::print_error("Malformed if expression: " + text, file);
232 break;
233 }
234
235 auto expr = text.substr(openParen + 1, closeParen - openParen - 1);
236
237 if (expr.find(">=") != CompilerKit::STLString::npos) {
238 auto left = text.substr(
239 text.find(keyword.first.keyword_name) + keyword.first.keyword_name.size() + 2,
240 expr.find("<=") + strlen("<="));
241 auto right = text.substr(expr.find(">=") + strlen(">="), text.find(")") - 1);
242
243 // Trim whitespace
244 while (!right.empty() && (right.back() == ' ' || right.back() == '\t')) {
245 right.pop_back();
246 }
247 while (!right.empty() && (right.front() == ' ' || right.front() == '\t')) {
248 right.erase(0, 1);
249 }
250
251 while (!left.empty() && (left.back() == ' ' || left.back() == '\t')) {
252 left.pop_back();
253 }
254 while (!left.empty() && (left.front() == ' ' || left.front() == '\t')) {
255 left.erase(0, 1);
256 }
257
258 if ((!left.empty() && !isdigit(left[0])) || (!right.empty() && !isdigit(right[0]))) {
259 auto indexRight = 0UL;
260
261 auto& valueOfVar = (!left.empty() && !isdigit(left[0])) ? left : right;
262
263 if (!valueOfVar.empty()) {
264 for (auto pairRight : kRegisterMap) {
265 ++indexRight;
266
267 CompilerKit::STLString instr = "mov ";
268
269 if (pairRight != valueOfVar) {
270 auto& valueOfVarOpposite = (!left.empty() && isdigit(left[0])) ? left : right;
271
272 syntax_tree.fUserValue +=
273 instr + kRegisterList[indexRight + 1] + ", " + valueOfVarOpposite + "\n";
274 syntax_tree.fUserValue += "cmp " + kRegisterList[kRegisterMap.size() - 1] + "," +
275 kRegisterList[indexRight + 1] + "\n";
276
277 goto lc_done_iterarting_on_if;
278 }
279
280 auto& valueOfVarOpposite = (!left.empty() && isdigit(left[0])) ? left : right;
281
282 syntax_tree.fUserValue +=
283 instr + kRegisterList[indexRight + 1] + ", " + valueOfVarOpposite + "\n";
284 syntax_tree.fUserValue += "cmp " + kRegisterList[kRegisterMap.size() - 1] + ", " +
285 kRegisterList[indexRight + 1] + "\n";
286
287 break;
288 }
289 }
290 }
291
292 lc_done_iterarting_on_if:
293
294 CompilerKit::STLString symbol_name_fn = text;
295
296 symbol_name_fn.erase(symbol_name_fn.find(keyword.first.keyword_name));
297
298 for (auto& ch : symbol_name_fn) {
299 if (ch == ' ') ch = '_';
300 }
301
302 syntax_tree.fUserValue +=
303 "jge __OFFSET_ON_TRUE_LC\nsegment .code64 __OFFSET_ON_TRUE_LC:\n";
304 }
305
306 break;
307 }
308 case CompilerKit::KeywordKind::kKeywordKindFunctionStart: {
309 for (auto& ch : text) {
310 if (isdigit(ch)) {
311 goto dont_accept;
312 }
313 }
314
315 goto accept;
316
317 dont_accept:
318 break;
319
320 accept:
321 CompilerKit::STLString symbol_name_fn = text;
322 size_t indexFnName = 0;
323
324 // this one is for the type.
325 for (auto& ch : text) {
326 ++indexFnName;
327
328 if (ch == '\t') break;
329 if (ch == ' ') break;
330 }
331
332 symbol_name_fn = text.substr(indexFnName);
333
334 if (text.find("return ") != CompilerKit::STLString::npos) {
335 text.erase(0, text.find("return "));
336 break;
337 }
338
339 if (text.ends_with(";") && text.find("return") == CompilerKit::STLString::npos)
340 goto lc_write_assembly;
341 else if (text.size() <= indexFnName)
342 Detail::print_error("Invalid function name: " + symbol_name_fn, file);
343
344 indexFnName = 0;
345
346 for (auto& ch : symbol_name_fn) {
347 if (ch == ' ' || ch == '\t') {
348 if (symbol_name_fn[indexFnName - 1] != ')')
349 Detail::print_error("Invalid function name: " + symbol_name_fn, file);
350 }
351
352 ++indexFnName;
353 }
354
355 if (symbol_name_fn.find("(") != CompilerKit::STLString::npos) {
356 symbol_name_fn.erase(symbol_name_fn.find("("));
357 }
358
359 syntax_tree.fUserValue = "public_segment .code64 __NECTI_" + symbol_name_fn + "\n";
360 ++kFunctionEmbedLevel;
361
362 kOriginMap.push_back({"__NECTI_" + symbol_name_fn, kOrigin});
363
364 break;
365
366 lc_write_assembly:
367 auto it = std::find_if(
368 kOriginMap.begin(), kOriginMap.end(),
369 [&symbol_name_fn](std::pair<CompilerKit::STLString, std::uintptr_t> pair) -> bool {
370 return symbol_name_fn == pair.first;
371 });
372
373 if (it != kOriginMap.end()) {
374 std::stringstream ss;
375 ss << std::hex << it->second;
376
377 syntax_tree.fUserValue = "jmp " + ss.str() + "\n";
378 kOrigin += 1UL;
379 }
380 }
381 case CompilerKit::KeywordKind::kKeywordKindFunctionEnd: {
382 if (kOnClassScope) --kOnClassScope;
383
384 if (text.ends_with(";")) break;
385
386 --kFunctionEmbedLevel;
387
388 if (kRegisterMap.size() > kRegisterList.size()) {
389 --kFunctionEmbedLevel;
390 }
391
392 if (kFunctionEmbedLevel < 1) kRegisterMap.clear();
393
394 break;
395 }
396 case CompilerKit::KeywordKind::kKeywordKindEndInstr:
397 case CompilerKit::KeywordKind::kKeywordKindVariableInc:
398 case CompilerKit::KeywordKind::kKeywordKindVariableDec:
399 case CompilerKit::KeywordKind::kKeywordKindVariableAssign: {
400 CompilerKit::STLString valueOfVar = "";
401
402 if (keyword.first.keyword_kind == CompilerKit::KeywordKind::kKeywordKindVariableInc) {
403 valueOfVar = text.substr(text.find("+=") + 2);
404 } else if (keyword.first.keyword_kind ==
405 CompilerKit::KeywordKind::kKeywordKindVariableDec) {
406 valueOfVar = text.substr(text.find("-=") + 2);
407 } else if (keyword.first.keyword_kind ==
408 CompilerKit::KeywordKind::kKeywordKindVariableAssign) {
409 valueOfVar = text.substr(text.find("=") + 1);
410 } else if (keyword.first.keyword_kind == CompilerKit::KeywordKind::kKeywordKindEndInstr) {
411 break;
412 }
413
414 while (valueOfVar.find(";") != CompilerKit::STLString::npos &&
415 keyword.first.keyword_kind != CompilerKit::KeywordKind::kKeywordKindEndInstr) {
416 valueOfVar.erase(valueOfVar.find(";"));
417 }
418
419 CompilerKit::STLString varName = text;
420
421 if (keyword.first.keyword_kind == CompilerKit::KeywordKind::kKeywordKindVariableInc) {
422 varName.erase(varName.find("+="));
423 } else if (keyword.first.keyword_kind ==
424 CompilerKit::KeywordKind::kKeywordKindVariableDec) {
425 varName.erase(varName.find("-="));
426 } else if (keyword.first.keyword_kind ==
427 CompilerKit::KeywordKind::kKeywordKindVariableAssign) {
428 varName.erase(varName.find("="));
429 } else if (keyword.first.keyword_kind == CompilerKit::KeywordKind::kKeywordKindEndInstr) {
430 varName.erase(varName.find(";"));
431 }
432
433 static Boolean typeFound = false;
434
435 for (auto& keyword : kKeywords) {
436 if (keyword.keyword_kind == CompilerKit::kKeywordKindType) {
437 if (text.find(keyword.keyword_name) != CompilerKit::STLString::npos) {
438 if (text[text.find(keyword.keyword_name)] == ' ') {
439 typeFound = false;
440 continue;
441 }
442
443 typeFound = true;
444 }
445 }
446 }
447
448 CompilerKit::STLString instr = "mov ";
449
450 std::vector<CompilerKit::STLString> newVars;
451
452 if (typeFound &&
453 keyword.first.keyword_kind != CompilerKit::KeywordKind::kKeywordKindVariableInc &&
454 keyword.first.keyword_kind != CompilerKit::KeywordKind::kKeywordKindVariableDec) {
455 if (kRegisterMap.size() > kRegisterList.size()) {
456 ++kFunctionEmbedLevel;
457 }
458
459 while (varName.find(" ") != CompilerKit::STLString::npos) {
460 varName.erase(varName.find(" "), 1);
461 }
462
463 while (varName.find("\t") != CompilerKit::STLString::npos) {
464 varName.erase(varName.find("\t"), 1);
465 }
466
467 // Remove whitespace only (keep operators and quotes)
468 while (!valueOfVar.empty() && (valueOfVar[0] == ' ' || valueOfVar[0] == '\t')) {
469 valueOfVar.erase(0, 1);
470 }
471
472 constexpr auto kTrueVal = "true";
473 constexpr auto kFalseVal = "false";
474
475 if (valueOfVar == kTrueVal) {
476 valueOfVar = "1";
477 } else if (valueOfVar == kFalseVal) {
478 valueOfVar = "0";
479 }
480
481 std::size_t indexRight = 0UL;
482
483 for (auto pairRight : kRegisterMap) {
484 ++indexRight;
485
486 if (pairRight != valueOfVar) {
487 if (valueOfVar[0] == '\"') {
488 syntax_tree.fUserValue = "segment .data64 __NECTI_LOCAL_VAR_" + varName + ": db " +
489 valueOfVar + ", 0\n\n";
490 syntax_tree.fUserValue += instr + kRegisterList[kRegisterMap.size() - 1] + ", " +
491 "__NECTI_LOCAL_VAR_" + varName + "\n";
492 kOrigin += 1UL;
493 } else {
494 syntax_tree.fUserValue =
495 instr + kRegisterList[kRegisterMap.size() - 1] + ", " + valueOfVar + "\n";
496 kOrigin += 1UL;
497 }
498
499 goto done;
500 }
501 }
502
503 if (((int) indexRight - 1) < 0) {
504 if (valueOfVar[0] == '\"') {
505 syntax_tree.fUserValue =
506 "segment .data64 __NECTI_LOCAL_VAR_" + varName + ": db " + valueOfVar + ", 0\n";
507 syntax_tree.fUserValue += instr + kRegisterList[kRegisterMap.size()] + ", " +
508 "__NECTI_LOCAL_VAR_" + varName + "\n";
509 kOrigin += 1UL;
510 } else {
511 syntax_tree.fUserValue =
512 instr + kRegisterList[kRegisterMap.size()] + ", " + valueOfVar + "\n";
513 kOrigin += 1UL;
514 }
515
516 goto done;
517 }
518
519 if (!valueOfVar.empty() && valueOfVar[0] != '\"' && valueOfVar[0] != '\'' &&
520 !isdigit(valueOfVar[0])) {
521 for (auto pair : kRegisterMap) {
522 if (pair == valueOfVar) goto done;
523 }
524
525 Detail::print_error("Variable not declared: " + varName, file);
526 break;
527 }
528
529 done:
530 for (auto& keyword : kKeywords) {
531 if (keyword.keyword_kind == CompilerKit::kKeywordKindType &&
532 varName.find(keyword.keyword_name) != CompilerKit::STLString::npos) {
533 varName.erase(varName.find(keyword.keyword_name), keyword.keyword_name.size());
534 break;
535 }
536 }
537
538 newVars.push_back(varName);
539
540 break;
541 }
542
543 kRegisterMap.insert(kRegisterMap.end(), newVars.begin(), newVars.end());
544
545 if (keyword.second > 0 &&
546 kKeywords[keyword.second - 1].keyword_kind == CompilerKit::kKeywordKindType ||
547 kKeywords[keyword.second - 1].keyword_kind == CompilerKit::kKeywordKindTypePtr) {
548 syntax_tree.fUserValue = "\n";
549 continue;
550 }
551
552 if (keyword.first.keyword_kind == CompilerKit::KeywordKind::kKeywordKindEndInstr) {
553 syntax_tree.fUserValue = "\n";
554 continue;
555 }
556
557 if (keyword.first.keyword_kind == CompilerKit::KeywordKind::kKeywordKindVariableInc) {
558 instr = "add ";
559 } else if (keyword.first.keyword_kind ==
560 CompilerKit::KeywordKind::kKeywordKindVariableDec) {
561 instr = "sub ";
562 }
563
564 CompilerKit::STLString varErrCpy = varName;
565
566 while (varName.find(" ") != CompilerKit::STLString::npos) {
567 varName.erase(varName.find(" "), 1);
568 }
569
570 while (varName.find("\t") != CompilerKit::STLString::npos) {
571 varName.erase(varName.find("\t"), 1);
572 }
573
574 std::size_t indxReg = 0UL;
575
576 while (!valueOfVar.empty() && (valueOfVar[0] == ' ' || valueOfVar[0] == '\t')) {
577 valueOfVar.erase(0, 1);
578 }
579
580 while (valueOfVar.find(" ") != CompilerKit::STLString::npos) {
581 valueOfVar.erase(valueOfVar.find(" "), 1);
582 }
583
584 while (valueOfVar.find("\t") != CompilerKit::STLString::npos) {
585 valueOfVar.erase(valueOfVar.find("\t"), 1);
586 }
587
588 constexpr auto kTrueVal = "true";
589 constexpr auto kFalseVal = "false";
590
592
593 if (valueOfVar == kTrueVal) {
594 valueOfVar = "1";
595 } else if (valueOfVar == kFalseVal) {
596 valueOfVar = "0";
597 }
598
599 for (auto pair : kRegisterMap) {
600 ++indxReg;
601
602 if (pair != varName) continue;
603
604 std::size_t indexRight = 0ul;
605
606 for (auto pairRight : kRegisterMap) {
607 ++indexRight;
608
609 if (pairRight != varName) {
610 syntax_tree.fUserValue =
611 instr + kRegisterList[kRegisterMap.size()] + ", " + valueOfVar + "\n";
612 kOrigin += 1UL;
613 continue;
614 }
615
616 syntax_tree.fUserValue =
617 instr + kRegisterList[indexRight - 1] + ", " + valueOfVar + "\n";
618 kOrigin += 1UL;
619 break;
620 }
621
622 newVars.push_back(varName);
623 break;
624 }
625
626 if (syntax_tree.fUserValue.empty()) {
627 Detail::print_error("Variable not declared: " + varName, file);
628 }
629
630 kRegisterMap.insert(kRegisterMap.end(), newVars.begin(), newVars.end());
631
632 break;
633 }
634 case CompilerKit::KeywordKind::kKeywordKindReturn: {
635 try {
636 auto pos = text.find("return") + strlen("return") + 1;
637 CompilerKit::STLString subText = text.substr(pos);
638 subText = subText.erase(subText.find(";"));
639 size_t indxReg = 0UL;
640
641 if (!subText.empty() && subText[0] != '\"' && subText[0] != '\'') {
642 if (!isdigit(subText[0])) {
643 for (auto pair : kRegisterMap) {
644 ++indxReg;
645
646 if (pair != subText) continue;
647
648 syntax_tree.fUserValue = "mov rax, " + kRegisterList[indxReg - 1] + "\nret\n";
649 kOrigin += 1UL;
650
651 break;
652 }
653 } else {
654 syntax_tree.fUserValue = "mov rax, " + subText + "\nret\n";
655 kOrigin += 1UL;
656
657 break;
658 }
659 } else {
660 syntax_tree.fUserValue = "__NECTI_LOCAL_RETURN_STRING: db " + subText +
661 ", 0\nmov rcx, __NECTI_LOCAL_RETURN_STRING\n";
662 syntax_tree.fUserValue += "mov rax, rcx\nret\n";
663 kOrigin += 1UL;
664
665 break;
666 }
667
668 if (syntax_tree.fUserValue.empty()) {
669 if (subText.find("(") != CompilerKit::STLString::npos) {
670 subText.erase(subText.find("("));
671
672 auto it = std::find_if(
673 kOriginMap.begin(), kOriginMap.end(),
674 [&subText](std::pair<CompilerKit::STLString, std::uintptr_t> pair) -> bool {
675 return pair.first.find(subText) != CompilerKit::STLString::npos;
676 });
677
678 if (it == kOriginMap.end())
679 Detail::print_error("Invalid return value: " + subText, file);
680
681 std::stringstream ss;
682 ss << it->second;
683
684 syntax_tree.fUserValue = "jmp " + ss.str() + "\nret\n";
685 kOrigin += 1UL;
686 break;
687 }
688 }
689
690 break;
691 } catch (...) {
692 syntax_tree.fUserValue = "ret\n";
693 kOrigin += 1UL;
694 }
695 }
696 default: {
697 continue;
698 }
699 }
700 }
701
702 return syntax_tree;
703}

713class AssemblyCPlusPlusInterfaceAMD64 final CK_ASSEMBLY_INTERFACE {
714 public:
715 explicit AssemblyCPlusPlusInterfaceAMD64() = default;
716 ~AssemblyCPlusPlusInterfaceAMD64() override = default;
717
718 NECTI_COPY_DEFAULT(AssemblyCPlusPlusInterfaceAMD64);
719
720 UInt32 Arch() noexcept override { return CompilerKit::AssemblyFactory::kArchAMD64; }
721
722 Int32 CompileToFormat(CompilerKit::STLString src, Int32 arch) override {
723 if (kCompilerFrontend == nullptr) return kExitNO;
724
725 CompilerKit::STLString dest = src;
726 dest += ".pp.masm";
727
728 std::ofstream out_fp(dest);
729 std::ifstream src_fp = std::ifstream(src + ".pp");
730
731 CompilerKit::STLString line_source;
732
733 out_fp << "#bits 64\n";
734 out_fp << "#org " << kOrigin << "\n\n";
735
736 while (std::getline(src_fp, line_source)) {
737 out_fp << kCompilerFrontend->Compile(line_source, src).fUserValue;
738 }
739
740 return kExitOK;
741 }
742};
