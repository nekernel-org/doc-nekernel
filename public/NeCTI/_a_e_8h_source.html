28// Advanced Executable File Format for ld64.
29// Reloctable by offset is the default strategy.
30// You can also relocate at runtime but that's up to the operating system loader.
31
32namespace CompilerKit {
33// @brief Advanced Executable Header
34// One thing to keep in mind.
35// This object format, is reloctable.
36typedef struct AEHeader final {
37 Char fMagic[kAEMagLen] = {};
38 UInt16 fVersion{kAEVer};
39 Char fArch{};
40 Char fSubArch{};
41 SizeType fCount{};
42 Char fSize{};
43 SizeType fStartCode{};
44 SizeType fCodeSize{};
45 Char fPad[kAEPad] = {};
46} PACKED AEHeader, *AEHeaderPtr;

48// @brief Advanced Executable Record.
49// Could be data, code or bss.
50// fKind must be filled with PEF fields.
51
52typedef struct AERecordHeader final {
53 Char fName[kAESymbolLen];
54 SizeType fKind;
55 SizeType fSize;
56 SizeType fFlags;
57 UIntPtr fOffset;
58 Char fPad[kAEPad];
59} PACKED AERecordHeader, *AERecordHeaderPtr;

61enum {
