395 for (size_t not_found_idx = 0; not_found_idx < command_headers.size(); ++not_found_idx) {

410 // the reason we do is because, this may not match the symbol, and we need
411 // to look for other matching symbols.
412 for (auto& command_hdr : command_headers) {
413 if (CompilerKit::STLString(command_hdr.Name).find(symbol_imp) !=
414 CompilerKit::STLString::npos &&
415 CompilerKit::STLString(command_hdr.Name).find(kLdDefineSymbol) ==
416 CompilerKit::STLString::npos) {
417 CompilerKit::STLString undefined_symbol = command_hdr.Name;
418 auto result_of_sym = undefined_symbol.substr(undefined_symbol.find(symbol_imp));
419
420 for (int i = 0; result_of_sym[i] != 0; ++i) {
421 if (result_of_sym[i] != symbol_imp[i]) goto ld_continue_search;
