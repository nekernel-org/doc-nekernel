176CompilerKit::SyntaxLeafList::SyntaxLeaf CompilerFrontendPower64::Compile(std::string text_,
177 std::string file) {
178 std::string text = text_;
179
180 bool typeFound = false;
181 bool fnFound = false;
182
183 // setup generator.
184 std::random_device rd;
185
186 auto seed_data = std::array<int, std::mt19937::state_size>{};
187 std::generate(std::begin(seed_data), std::end(seed_data), std::ref(rd));
188 std::seed_seq seq(std::begin(seed_data), std::end(seed_data));
189 std::mt19937 generator(seq);
190
191 // start parsing
192 for (size_t text_index = 0; text_index < text.size(); ++text_index) {
193 auto syntax_leaf = CompilerKit::SyntaxLeafList::SyntaxLeaf();
194
195 auto gen = uuids::uuid_random_generator{generator};
196 uuids::uuid out = gen();
197
198 Detail::number_cast time_off = (UInt64) out.as_bytes().data();
199
200 if (!typeFound) {
201 auto substr = text.substr(text_index);
202 std::string match_type;
203
204 for (size_t y = 0; y < substr.size(); ++y) {
205 if (substr[y] == ' ') {
206 while (match_type.find(' ') != std::string::npos) {
207 match_type.erase(match_type.find(' '));
208 }
209
210 for (auto& clType : kCompilerTypes) {
211 if (clType.fName == match_type) {
212 match_type.clear();
213
214 std::string buf;
215
216 buf += clType.fValue;
217 buf += ' ';
218
219 if (substr.find('=') != std::string::npos) {
220 break;
221 }
222
223 if (text.find('(') != std::string::npos) {
224 syntax_leaf.fUserValue = buf;
225
226 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
227 }
228
229 typeFound = true;
230 break;
231 }
232 }
233
234 break;
235 }
236
237 match_type += substr[y];
238 }
239 }
240
241 if (text[text_index] == '{') {
242 if (kInStruct) {
243 continue;
244 }
245
246 kInBraces = true;
247 ++kBracesCount;
248
249 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
250 }
251
252 // return keyword handler
253 if (text[text_index] == 'r') {
254 std::string return_keyword;
255 return_keyword += "return";
256
257 std::size_t index = 0UL;
258
259 std::string value;
260
261 for (size_t return_index = text_index; return_index < text.size(); ++return_index) {
262 if (text[return_index] != return_keyword[index]) {
263 for (size_t value_index = return_index; value_index < text.size(); ++value_index) {
264 if (text[value_index] == ';') break;
265
266 value += text[value_index];
267 }
268
269 break;
270 }
271
272 ++index;
273 }
274
275 if (index == return_keyword.size()) {
276 if (!value.empty()) {
277 if (value.find('(') != std::string::npos) {
278 value.erase(value.find('('));
279 }
280
281 if (!isdigit(value[value.find('(') + 2])) {
282 std::string tmp = value;
283 bool reg_to_reg = false;
284
285 value.clear();
286
287 value += " extern_segment";
288 value += tmp;
289 }
290
291 syntax_leaf.fUserValue = "\tmr r31, ";
292
293 // make it pretty.
294 while (value.find('\t') != std::string::npos) value.erase(value.find('\t'), 1);
295
296 while (value.find(' ') != std::string::npos) value.erase(value.find(' '), 1);
297
298 while (value.find("extern_segment") != std::string::npos)
299 value.erase(value.find("extern_segment"), strlen("extern_segment"));
300
301 bool found = false;
302
303 for (auto& reg : kState.kStackFrame) {
304 if (value.find(reg.fName) != std::string::npos) {
305 found = true;
306 syntax_leaf.fUserValue += reg.fReg;
307 break;
308 }
309 }
310
311 if (!found) syntax_leaf.fUserValue += "r0";
312 }
313
314 syntax_leaf.fUserValue += "\n\tblr";
315
316 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
317
318 break;
319 }
320 }
321
322 if (text[text_index] == 'i' && text[text_index + 1] == 'f') {
323 auto expr = text.substr(text_index + 2);
324 text.erase(text_index, 2);
325
326 if (expr.find("{") != std::string::npos) {
327 expr.erase(expr.find("{"));
328 }
329
330 if (expr.find("(") != std::string::npos) expr.erase(expr.find("("));
331
332 if (expr.find(")") != std::string::npos) expr.erase(expr.find(")"));
333
334 kIfFunction = "__NECTI_IF_PROC_";
335 kIfFunction += std::to_string(time_off._Raw);
336
337 syntax_leaf.fUserValue =
338 "\tcmpw "
339 "r10, r11";
340
341 syntax_leaf.fUserValue += "\n\tbeq extern_segment " + kIfFunction +
342 " \ndword public_segment .code64 " + kIfFunction + "\n";
343
344 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
345
346 kIfFound = true;
347 }
348
349 // Parse expressions and instructions here.
350 // what does this mean?
351 // we encounter an assignment, or we reached the end of an expression.
352 if (text[text_index] == '=' || text[text_index] == ';') {
353 if (fnFound) continue;
354 if (kIfFound) continue;
355
356 if (text[text_index] == ';' && kInStruct) continue;
357
358 if (text.find("typedef ") != std::string::npos) continue;
359
360 if (text[text_index] == '=' && kInStruct) {
361 Detail::print_error("assignement of value inside a struct " + text, file);
362 continue;
363 }
364
365 if (text[text_index] == ';' && kInStruct) {
366 bool space_found_ = false;
367 std::string sym;
368
369 for (auto& ch : text) {
370 if (ch == ' ') {
371 space_found_ = true;
372 }
373
374 if (ch == ';') break;
375
376 if (space_found_) sym.push_back(ch);
377 }
378
379 kState.kStructMap[kState.kStructMap.size() - 1].fOffsets.push_back(
380 std::make_pair(kState.kStructMap[kState.kStructMap.size() - 1].fOffsetsCnt + 4, sym));
381
382 kState.kStructMap[kState.kStructMap.size() - 1].fOffsetsCnt =
383 kState.kStructMap[kState.kStructMap.size() - 1].fOffsetsCnt + 4;
384
385 continue;
386 }
387
388 if (text[text_index] == '=' && kInStruct) {
389 continue;
390 }
391
392 if (text[text_index + 1] == '=' || text[text_index - 1] == '!' ||
393 text[text_index - 1] == '<' || text[text_index - 1] == '>') {
394 continue;
395 }
396
397 std::string substr;
398
399 if (text.find('=') != std::string::npos && kInBraces && !kIfFound) {
400 if (text.find("*") != std::string::npos) {
401 if (text.find("=") > text.find("*"))
402 substr += "\tli ";
403 else
404 substr += "\tli ";
405 } else {
406 substr += "\tli ";
407 }
408 } else if (text.find('=') != std::string::npos && !kInBraces) {
409 substr += "stw public_segment .data64 ";
410 }
411
412 int first_encountered = 0;
413
414 std::string str_name;
415
416 for (size_t text_index_2 = 0; text_index_2 < text.size(); ++text_index_2) {
417 if (text[text_index_2] == '\"') {
418 ++text_index_2;
419
420 // want to add this, so that the parser recognizes that this is a
421 // string.
422 substr += '"';
423
424 for (; text_index_2 < text.size(); ++text_index_2) {
425 if (text[text_index_2] == '\"') break;
426
427 substr += text[text_index_2];
428 }
429 }
430
431 if (text[text_index_2] == '{' || text[text_index_2] == '}') continue;
432
433 if (text[text_index_2] == ';') {
434 break;
435 }
436
437 if (text[text_index_2] == ' ' || text[text_index_2] == '\t') {
438 if (first_encountered != 2) {
439 if (text[text_index] != '=' &&
440 substr.find("public_segment .data64") == std::string::npos && !kInStruct)
441 substr += "public_segment .data64 ";
442 }
443
444 ++first_encountered;
445
446 continue;
447 }
448
449 if (text[text_index_2] == '=') {
450 if (!kInBraces) {
451 substr.replace(substr.find("public_segment .data64"), strlen("public_segment .data64"),
452 "public_segment .zero64 ");
453 }
454
455 substr += ",";
456 continue;
457 }
458
459 substr += text[text_index_2];
460 }
461
462 for (auto& clType : kCompilerTypes) {
463 if (substr.find(clType.fName) != std::string::npos) {
464 if (substr.find(clType.fName) > substr.find('"')) continue;
465
466 substr.erase(substr.find(clType.fName), clType.fName.size());
467 } else if (substr.find(clType.fValue) != std::string::npos) {
468 if (substr.find(clType.fValue) > substr.find('"')) continue;
469
470 if (clType.fName == "const") continue;
471
472 substr.erase(substr.find(clType.fValue), clType.fValue.size());
473 }
474 }
475
476 if (substr.find("extern") != std::string::npos) {
477 substr.replace(substr.find("extern"), strlen("extern"), "extern_segment ");
478
479 if (substr.find("public_segment .data64") != std::string::npos)
480 substr.erase(substr.find("public_segment .data64"), strlen("public_segment .data64"));
481 }
482
483 auto var_to_find = std::find_if(
484 kCompilerVariables.cbegin(), kCompilerVariables.cend(),
485 [&](Detail::CompilerType type) { return type.fName.find(substr) != std::string::npos; });
486
487 kCompilerVariables.push_back({.fName = substr});
488
489 if (text[text_index] == ';') break;
490
491 std::string reg = kAsmRegisterPrefix;
492
493 ++kRegisterCounter;
494 reg += std::to_string(kRegisterCounter);
495
496 auto newSubstr = substr.substr(substr.find(" "));
497
498 std::string symbol;
499
500 for (size_t start = 0; start < newSubstr.size(); ++start) {
501 if (newSubstr[start] == ',') break;
502
503 if (newSubstr[start] == ' ') continue;
504
505 symbol += (newSubstr[start]);
506 }
507
508 kState.kStackFrame.push_back({.fName = symbol, .fReg = reg});
509
510 syntax_leaf.fUserValue += "\n\tli " + reg + substr.substr(substr.find(','));
511 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
512 }
513
514 // function handler.
515
516 if (text[text_index] == '(' && !fnFound && !kIfFound) {
517 std::string substr;
518 std::string args_buffer;
519 std::string args;
520
521 bool type_crossed = false;
522
523 for (size_t idx = text.find('(') + 1; idx < text.size(); ++idx) {
524 if (text[idx] == ',') continue;
525
526 if (text[idx] == ' ') continue;
527
528 if (text[idx] == ')') break;
529 }
530
531 for (char substr_first_index : text) {
532 if (substr_first_index != ',')
533 args_buffer += substr_first_index;
534 else
535 args_buffer += '$';
536
537 if (substr_first_index == ';') {
538 args_buffer = args_buffer.erase(0, args_buffer.find('('));
539 args_buffer = args_buffer.erase(args_buffer.find(';'), 1);
540 args_buffer = args_buffer.erase(args_buffer.find(')'), 1);
541 args_buffer = args_buffer.erase(args_buffer.find('('), 1);
542
543 if (!args_buffer.empty()) args += "\tldw r6, ";
544
545 std::string register_type;
546 std::size_t index = 7UL;
547
548 while (args_buffer.find("$") != std::string::npos) {
549 register_type = kRegisterPrefix;
550 register_type += std::to_string(index);
551
552 ++index;
553
554 args_buffer.replace(args_buffer.find('$'), 1, "\n\tldw " + register_type + ",");
555 }
556
557 args += args_buffer;
558 args += "\n\tli r31, ";
559 }
560 }
561
562 for (char _text_i : text) {
563 if (_text_i == '\t' || _text_i == ' ') {
564 if (!type_crossed) {
565 substr.clear();
566 type_crossed = true;
567 }
568
569 continue;
570 }
571
572 if (_text_i == '(') break;
573
574 substr += _text_i;
575 }
576
577 if (kInBraces) {
578 syntax_leaf.fUserValue = args;
579 syntax_leaf.fUserValue += substr;
580
581 syntax_leaf.fUserValue += "\n\tblr\n";
582
583 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
584
585 fnFound = true;
586 } else {
587 syntax_leaf.fUserValue.clear();
588
589 syntax_leaf.fUserValue += "public_segment .code64 ";
590
591 syntax_leaf.fUserValue += substr;
592 syntax_leaf.fUserValue += "\n";
593
594 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
595
596 fnFound = true;
597 }
598
599 kCompilerFunctions.push_back(text);
600 }
601
602 if (text[text_index] == '-' && text[text_index + 1] == '-') {
603 text = text.replace(text.find("--"), strlen("--"), "");
604
605 for (int _text_i = 0; _text_i < text.size(); ++_text_i) {
606 if (text[_text_i] == '\t' || text[_text_i] == ' ') text.erase(_text_i, 1);
607 }
608
609 syntax_leaf.fUserValue += "dec ";
610 syntax_leaf.fUserValue += text;
611
612 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
613 break;
614 }
615
616 if (text[text_index] == '}') {
617 kRegisterCounter = kStartUsable;
618
619 --kBracesCount;
620
621 if (kBracesCount < 1) {
622 kInBraces = false;
623 kBracesCount = 0;
624 }
625
626 if (kIfFound) kIfFound = false;
627
628 if (kInStruct) kInStruct = false;
629
630 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
631 }
632
633 syntax_leaf.fUserValue.clear();
634 }
635
636 auto syntax_leaf = CompilerKit::SyntaxLeafList::SyntaxLeaf();
637 syntax_leaf.fUserValue = "\n";
638 kState.fSyntaxTree->fLeafList.push_back(syntax_leaf);
639
640 return syntax_leaf;
641}

646std::string CompilerFrontendPower64::Check(const char* text, const char* file) {
647 std::string err_str;
648 std::string ln = text;
649
650 if (ln.empty()) {
651 return err_str;
652 }
653
654 bool non_ascii_found = false;
655
656 for (int i = 0; i < ln.size(); ++i) {
657 if (isalnum(ln[i])) {
658 non_ascii_found = true;
659 break;
660 }
661 }
662
663 if (kShouldHaveBraces && ln.find('{') != std::string::npos) {
664 kShouldHaveBraces = false;
665 }
666
667 if (!non_ascii_found) return err_str;
668
669 size_t string_index = 1UL;
670
671 if (ln.find('\'') != std::string::npos) {
672 string_index = ln.find('\'') + 1;
673
674 for (; string_index < ln.size(); ++string_index) {
675 if (ln[string_index] == '\'') {
676 if (ln[string_index + 1] != ';') {
677 ln.erase(string_index, 1);
678 }
679
680 return err_str;
681 }
682 }
683 } else if (ln.find('"') != std::string::npos) {
684 string_index = ln.find('"') + 1;
685
686 for (; string_index < ln.size(); ++string_index) {
687 if (ln[string_index] == '"') {
688 if (ln[string_index + 1] != ';') {
689 ln.erase(string_index, 1);
690 } else {
691 break;
692 }
693 }
694 }
695 } else if (ln.find('"') == std::string::npos && ln.find('\'') == std::string::npos) {
696 std::vector<std::string> forbidden_words;
697
698 forbidden_words.push_back("\\");
699 forbidden_words.push_back("?");
700 forbidden_words.push_back("@");
701 forbidden_words.push_back("~");
702 forbidden_words.push_back("::");
703 forbidden_words.push_back("--*");
704 forbidden_words.push_back("*/");
705
706 // add them to avoid stupid mistakes.
707 forbidden_words.push_back("namespace");
708 forbidden_words.push_back("class");
709 forbidden_words.push_back("extern \"C\"");
710
711 for (auto& forbidden : forbidden_words) {
712 if (ln.find(forbidden) != std::string::npos) {
713 err_str += "\nForbidden character detected: ";
714 err_str += forbidden;
715
716 return err_str;
717 }
718 }
719 }

721 struct CompilerVariableRange final {
722 std::string fBegin;
723 std::string fEnd;
724 };
725
726 const std::vector<CompilerVariableRange> variables_list = {
727 {.fBegin = "static ", .fEnd = "="}, {.fBegin = "=", .fEnd = ";"},
728 {.fBegin = "if(", .fEnd = "="}, {.fBegin = "if (", .fEnd = "="},
729 {.fBegin = "if(", .fEnd = "<"}, {.fBegin = "if (", .fEnd = "<"},
730 {.fBegin = "if(", .fEnd = ">"}, {.fBegin = "if (", .fEnd = ">"},
731 {.fBegin = "if(", .fEnd = ")"}, {.fBegin = "if (", .fEnd = ")"},
732
733 {.fBegin = "else(", .fEnd = "="}, {.fBegin = "else (", .fEnd = "="},
734 {.fBegin = "else(", .fEnd = "<"}, {.fBegin = "else (", .fEnd = "<"},
735 {.fBegin = "else(", .fEnd = ">"}, {.fBegin = "else (", .fEnd = ">"},
736 {.fBegin = "else(", .fEnd = ")"}, {.fBegin = "else (", .fEnd = ")"},
737 };
738
739 for (auto& variable : variables_list) {
740 if (ln.find(variable.fBegin) != std::string::npos) {
741 string_index = ln.find(variable.fBegin) + variable.fBegin.size();
742
743 while (ln[string_index] == ' ') ++string_index;
744
745 std::string keyword;
746
747 for (; string_index < ln.size(); ++string_index) {
748 if (ln[string_index] == variable.fEnd[0]) {
749 std::string varname = "";
750
751 for (size_t index_keyword = ln.find(' '); ln[index_keyword] != variable.fBegin[0];
752 ++index_keyword) {
753 if (ln[index_keyword] == ' ') {
754 continue;
755 }
756
757 if (isdigit(ln[index_keyword])) {
758 goto cc_next_loop;
759 }
760
761 varname += ln[index_keyword];
762 }
763
764 if (varname.find(' ') != std::string::npos) {
765 varname.erase(0, varname.find(' '));
766
767 if (variable.fBegin == "extern") {
768 varname.erase(0, varname.find(' '));
769 }
770 }
771
772 if (kRegisterCounter == 5 || kRegisterCounter == 6) ++kRegisterCounter;
773
774 std::string reg = kAsmRegisterPrefix;
775 reg += std::to_string(kRegisterCounter);
776
777 kCompilerVariables.push_back({.fValue = varname});
778 goto cc_check_done;
779 }
780
781 keyword.push_back(ln[string_index]);
782 }
783
784 goto cc_next_loop;
785
786 cc_check_done:
787
788 // skip digit value.
789 if (isdigit(keyword[0]) || keyword[0] == '"') {
790 goto cc_next_loop;
791 }
792
793 while (keyword.find(' ') != std::string::npos) keyword.erase(keyword.find(' '), 1);
794
795 for (auto& var : kCompilerVariables) {
796 if (var.fValue.find(keyword) != std::string::npos) {
