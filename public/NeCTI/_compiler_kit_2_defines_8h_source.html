140inline bool to_str(Char* str, Int32 limit, Int32 base) noexcept {
141 if (limit == 0) return false;
142
143 Int32 copy_limit = limit;
144 Int32 cnt = 0;
145 Int32 ret = base;
146
147 while (limit != 1) {
148 ret = ret % 10;
149 str[cnt] = ret;
150
151 ++cnt;
152 --limit;
153 --ret;
154 }
155
156 str[copy_limit] = '\0';
157 return true;
158}
