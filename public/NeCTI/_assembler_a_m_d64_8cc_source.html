83// @brief AMD64 assembler entrypoint, the program/module starts here.
84
86
87NECTI_MODULE(AssemblerMainAMD64) {
89
90 CompilerKit::install_signal(SIGSEGV, Detail::drvi_crash_handler);
91
92 std::string opcodes_jump[kJumpLimit] = {"ja", "jae", "jb", "jbe", "jc", "je", "jg", "jge",
93 "jl", "jle", "jna", "jnae", "jnb", "jnbe", "jnc", "jne",
94 "jng", "jnge", "jnl", "jnle", "jno", "jnp", "jns", "jnz",
95 "jo", "jp", "jpe", "jpo", "js", "jz"};
96
97 for (i64_hword_t i = 0; i < kJumpLimit; i++) {
98 CpuOpcodeAMD64 code{.fName = opcodes_jump[i],
99 .fOpcode = static_cast<i64_hword_t>(kAsmJumpOpcode + i)};
100 kOpcodesAMD64.push_back(code);
101 }
102
103 CpuOpcodeAMD64 code{.fName = "jcxz", .fOpcode = 0xE3};
104 kOpcodesAMD64.push_back(code);
105
106 for (i64_hword_t i = kJumpLimitStandard; i < kJumpLimitStandardLimit; i++) {
107 CpuOpcodeAMD64 code{.fName = "jmp", .fOpcode = i};
108 kOpcodesAMD64.push_back(code);
109 }
110
111 CpuOpcodeAMD64 lahf{.fName = "lahf", .fOpcode = 0x9F};
112 kOpcodesAMD64.push_back(lahf);
113
114 CpuOpcodeAMD64 lds{.fName = "lds", .fOpcode = 0xC5};
115 kOpcodesAMD64.push_back(lds);
116
117 CpuOpcodeAMD64 lea{.fName = "lea", .fOpcode = 0x8D};
118 kOpcodesAMD64.push_back(lea);
119
120 CpuOpcodeAMD64 nop{.fName = "nop", .fOpcode = 0x90};
121 kOpcodesAMD64.push_back(nop);
122
124
125 for (size_t i = 1; i < argc; ++i) {
126 if (argv[i][0] == '-') {
127 if (strcmp(argv[i], "--amd64:ver") == 0 || strcmp(argv[i], "--amd64:v") == 0) {
128 kStdOut << "AssemblerAMD64: AMD64 Assembler Driver.\nAssemblerAMD64: "
129 "v1.10\nAssemblerAMD64: Copyright "
130 "(c) Amlal El Mahrouss\n";
131 return 0;
132 } else if (strcmp(argv[i], "--amd64:h") == 0) {
133 kStdOut << "AssemblerAMD64: AMD64 Assembler Driver.\nAssemblerAMD64: Copyright (c) 2024 "
134 "Amlal El Mahrouss\n";
135 kStdOut << "--version: Print program version.\n";
136 kStdOut << "--verbose: Print verbose output.\n";
137 kStdOut << "--binary: Output as flat binary.\n";
138
139 return 0;
140 } else if (strcmp(argv[i], "--amd64:binary") == 0) {
141 kOutputAsBinary = true;
142 continue;
143 } else if (strcmp(argv[i], "--amd64:verbose") == 0) {
144 kVerbose = true;
145 continue;
146 }
147
148 kStdOut << "AssemblerAMD64: ignore " << argv[i] << "\n";
149 continue;
150 }
151
152 if (!std::filesystem::exists(argv[i])) {
153 kStdOut << "AssemblerAMD64: can't open: " << argv[i] << std::endl;
154 goto asm_fail_exit;
155 }
156
157 std::string object_output(argv[i]);
158 std::string asm_input(argv[i]);
159
160 for (auto& ext : kAsmFileExts) {
161 if (object_output.ends_with(ext)) {
162 object_output.erase(object_output.find(ext), std::strlen(ext));
163 break;
164 }
165 }
166
167 object_output += kOutputAsBinary ? kBinaryFileExt : kObjectFileExt;
168
169 std::ifstream file_ptr(argv[i]);
170 std::ofstream file_ptr_out(object_output, std::ofstream::binary);
171
172 kStdOut << "AssemblerAMD64: Assembling: " << argv[i] << "\n";
173
174 if (file_ptr_out.bad()) {
175 if (kVerbose) {
176 kStdOut << "AssemblerAMD64: error: " << strerror(errno) << "\n";
177 }
178
179 return 1;
180 }
181
182 std::string line;
183
184 CompilerKit::AEHeader hdr{0};
185
186 memset(hdr.fPad, kAENullType, kAEPad);
187
188 hdr.fMagic[0] = kAEMag0;
189 hdr.fMagic[1] = kAEMag1;
190 hdr.fSize = sizeof(CompilerKit::AEHeader);
191 hdr.fArch = kOutputArch;
192
194
195 // COMPILATION LOOP
196
198
199 CompilerKit::EncoderAMD64 asm64;
200
201 if (kVerbose) {
202 kStdOut << "Compiling: " + asm_input << "\n";
203 kStdOut << "From: " + line << "\n";
204 }
205
206 while (std::getline(file_ptr, line)) {
207 if (auto ln = asm64.CheckLine(line, argv[i]); !ln.empty()) {
208 Detail::print_error(ln, argv[i]);
209 continue;
210 }
211
212 try {
213 asm_read_attributes(line);
214 asm64.WriteLine(line, argv[i]);
215 } catch (const std::exception& e) {
216 if (kVerbose) {
217 std::string what = e.what();
218 Detail::print_warning("exit because of: " + what, "CompilerKit");
219 }
220
221 try {
222 std::filesystem::remove(object_output);
223 } catch (...) {
224 }
225
226 goto asm_fail_exit;
227 }
228 }
229
230 if (!kOutputAsBinary) {
231 if (kVerbose) {
232 kStdOut << "AssemblerAMD64: Writing object file...\n";
233 }
234
235 // this is the final step, write everything to the file.
236
237 auto pos = file_ptr_out.tellp();
238
239 hdr.fCount = kRecords.size() + kUndefinedSymbols.size();
240
241 file_ptr_out << hdr;
242
243 if (kRecords.empty()) {
244 kStdErr << "AssemblerAMD64: At least one record is needed to write an object "
245 "file.\nAssemblerAMD64: Make one using `public_segment .code64 foo_bar`.\n";
246
247 std::filesystem::remove(object_output);
248 return 1;
249 }
250
251 kRecords[kRecords.size() - 1].fSize = kAppBytes.size();
252
253 std::size_t record_count = 0UL;
254
255 for (auto& rec : kRecords) {
256 if (kVerbose) kStdOut << "AssemblerAMD64: Wrote record " << rec.fName << " to file...\n";
257
258 rec.fFlags |= CompilerKit::kKindRelocationAtRuntime;
259 rec.fOffset = record_count;
260 ++record_count;
261
262 file_ptr_out << rec;
263 }
264
265 // increment once again, so that we won't lie about the kUndefinedSymbols.
266 ++record_count;
267
268 for (auto& sym : kUndefinedSymbols) {
269 CompilerKit::AERecordHeader _record_hdr{0};
270
271 if (kVerbose) kStdOut << "AssemblerAMD64: Wrote symbol " << sym << " to file...\n";
272
273 _record_hdr.fKind = kAENullType;
274 _record_hdr.fSize = sym.size();
275 _record_hdr.fOffset = record_count;
276
277 ++record_count;
278
279 memset(_record_hdr.fPad, kAENullType, kAEPad);
280 memcpy(_record_hdr.fName, sym.c_str(), sym.size());
281
282 file_ptr_out << _record_hdr;
283
284 ++kCounter;
285 }
286
287 auto pos_end = file_ptr_out.tellp();
288
289 file_ptr_out.seekp(pos);
290
291 hdr.fStartCode = pos_end;
292 hdr.fCodeSize = kAppBytes.size();
293
294 file_ptr_out << hdr;
295
296 file_ptr_out.seekp(pos_end);
297 } else {
298 if (kVerbose) {
299 kStdOut << "AssemblerAMD64: Write raw binary...\n";
300 }
301 }
302
303 // byte from byte, we write this.
304 for (auto& byte : kAppBytes) {
305 if (byte == 0) continue;
306
307 if (byte == 0xFF) {
308 byte = 0;
309 }
310
311 file_ptr_out << reinterpret_cast<const char*>(&byte)[0];
312 }
313
314 if (kVerbose) kStdOut << "AssemblerAMD64: Wrote file with program in it.\n";
315
316 file_ptr_out.flush();
317 file_ptr_out.close();
318
319 if (kVerbose) kStdOut << "AssemblerAMD64: Exit succeeded.\n";
320
321 return 0;
322 }
323
324asm_fail_exit:
325
326 if (kVerbose) kStdOut << "AssemblerAMD64: Exit failed.\n";
327
328 return 1;
329}

333// @brief Check for attributes
334// returns true if any was found.
335
337
338static bool asm_read_attributes(std::string line) {
339 // extern_segment is the opposite of public_segment, it signals to the ld
340 // that we need this symbol.
341 if (CompilerKit::find_word(line, "extern_segment")) {
342 if (kOutputAsBinary) {
343 Detail::print_error("Invalid directive in flat binary mode.", "CompilerKit");
344 throw std::runtime_error("invalid_extern_segment_bin");
345 }
346
347 auto name = line.substr(line.find("extern_segment") + strlen("extern_segment") + 1);
348
349 if (name.size() == 0) {
350 Detail::print_error("Invalid extern_segment", "power-as");
351 throw std::runtime_error("invalid_extern_segment");
352 }
353
354 std::string result = std::to_string(name.size());
355 result += kUndefinedSymbol;
356
357 // mangle this
358 for (char& j : name) {
359 if (j == ' ' || j == ',') j = '$';
360 }
361
362 result += name;
363
364 if (name.find(kPefCode64) != std::string::npos) {
365 // data is treated as code.
366 kCurrentRecord.fKind = CompilerKit::kPefCode;
367 } else if (name.find(kPefData64) != std::string::npos) {
368 // no code will be executed from here.
369 kCurrentRecord.fKind = CompilerKit::kPefData;
370 } else if (name.find(kPefZero64) != std::string::npos) {
371 // this is a bss section.
372 kCurrentRecord.fKind = CompilerKit::kPefZero;
373 }
374
375 // this is a special case for the start stub.
376 // we want this so that ld can find it.
377
378 if (name == kPefStart) {
379 kCurrentRecord.fKind = CompilerKit::kPefCode;
380 }
381
382 // now we can tell the code size of the previous kCurrentRecord.
383
384 if (!kRecords.empty()) kRecords[kRecords.size() - 1].fSize = kAppBytes.size();
385
386 memset(kCurrentRecord.fName, 0, kAESymbolLen);
387 memcpy(kCurrentRecord.fName, result.c_str(), result.size());
388
389 ++kCounter;
390
391 memset(kCurrentRecord.fPad, kAENullType, kAEPad);
392
393 kRecords.emplace_back(kCurrentRecord);
394
395 return true;
396 }
397 // public_segment is a special keyword used by AssemblerAMD64 to tell the AE output stage to
398 // mark this section as a header. it currently supports .code64, .data64 and
399 // .zero64.
400 else if (CompilerKit::find_word(line, "public_segment")) {
401 if (kOutputAsBinary) {
402 Detail::print_error("Invalid directive in flat binary mode.", "CompilerKit");
403 throw std::runtime_error("invalid_public_segment_bin");
404 }
405
406 auto name = line.substr(line.find("public_segment") + strlen("public_segment") + 1);
407
408 std::string name_copy = name;
409
410 for (char& j : name) {
411 if (j == ' ') j = '$';
412 }
413
414 if (std::find(kDefinedSymbols.begin(), kDefinedSymbols.end(), name) != kDefinedSymbols.end()) {
415 Detail::print_error("Symbol already defined.", "CompilerKit");
416 throw std::runtime_error("invalid_public_segment_bin");
417 }
418
419 kDefinedSymbols.push_back(name);
420
421 if (name.find(".code64") != std::string::npos) {
422 // data is treated as code.
423
424 name_copy.erase(name_copy.find(".code64"), strlen(".code64"));
425 kCurrentRecord.fKind = CompilerKit::kPefCode;
426 } else if (name.find(".data64") != std::string::npos) {
427 // no code will be executed from here.
428
429 name_copy.erase(name_copy.find(".data64"), strlen(".data64"));
430 kCurrentRecord.fKind = CompilerKit::kPefData;
431 } else if (name.find(".zero64") != std::string::npos) {
432 // this is a bss section.
433
434 name_copy.erase(name_copy.find(".zero64"), strlen(".zero64"));
435 kCurrentRecord.fKind = CompilerKit::kPefZero;
436 }
437
438 // this is a special case for the start stub.
439 // we want this so that ld can find it.
440
441 if (name == kPefStart) {
442 kCurrentRecord.fKind = CompilerKit::kPefCode;
443 }
444
445 while (name_copy.find(" ") != std::string::npos) name_copy.erase(name_copy.find(" "), 1);
446
447 kOriginLabel.push_back(std::make_pair(name_copy, kOrigin));
448 ++kOrigin;
449
450 // now we can tell the code size of the previous kCurrentRecord.
451
452 if (!kRecords.empty()) kRecords[kRecords.size() - 1].fSize = kAppBytes.size();
453
454 memset(kCurrentRecord.fName, 0, kAESymbolLen);
455 memcpy(kCurrentRecord.fName, name.c_str(), name.size());
456
457 ++kCounter;
458
459 memset(kCurrentRecord.fPad, kAENullType, kAEPad);
460
461 kRecords.emplace_back(kCurrentRecord);
462
463 return true;
464 }
465
466 return false;
467}

472// \brief authorize a brief set of characters.
473static inline bool is_not_valid(char c) {
474 if ((isalpha(c) || isdigit(c)) ||
475 ((c == ' ') || (c == '\t') || (c == ',') || (c == '(') || (c == ')') || (c == '"') ||
476 (c == '*') || (c == '\'') || (c == '[') || (c == ']') || (c == '+') || (c == '_') ||
477 (c == ':') || (c == '@') || (c == '.') || (c == '#') || (c == ';')))
478 return false;
479
480 return true;
481}

483bool is_valid_amd64(std::string str) {
484 return std::find_if(str.begin(), str.end(), is_not_valid) == str.end();
485}

494std::string CompilerKit::EncoderAMD64::CheckLine(std::string line, std::string file) {
495 std::string err_str;
496
497 if (line.empty() || CompilerKit::find_word(line, "extern_segment") ||
498 CompilerKit::find_word(line, "public_segment") ||
499 CompilerKit::find_word(line, kAssemblerPragmaSymStr) || CompilerKit::find_word(line, ";") ||
500 line[0] == kAssemblerPragmaSym) {
501 if (line.find(';') != std::string::npos) {
502 line.erase(line.find(';'));
503 } else {
504 // now check the line for validity
505 if (!Detail::algorithm::is_valid_amd64(line)) {
506 err_str = "Line contains non valid characters.\nhere -> ";
507 err_str += line;
508 }
509 }
510
511 return err_str;
512 }
513
514 // check for a valid instruction format.
515
516 if (line.find(',') != std::string::npos) {
517 if (line.find(',') + 1 == line.size()) {
518 err_str += "\nInstruction lacks right register, here -> ";
519 err_str += line.substr(line.find(','));
520
521 return err_str;
522 } else {
523 bool nothing_on_right = true;
524
525 if (line.find(',') + 1 > line.size()) {
526 err_str += "\nInstruction not complete, here -> ";
527 err_str += line;
528
529 return err_str;
530 }
531
532 auto substr = line.substr(line.find(',') + 1);
533
534 for (auto& ch : substr) {
535 if (ch != ' ' && ch != '\t') {
536 nothing_on_right = false;
537 }
538 }
539
540 // this means we found nothing after that ',' .
541 if (nothing_on_right) {
542 err_str += "\nInstruction not complete, here -> ";
543 err_str += line;
544
545 return err_str;
546 }
547 }
548 }
549 for (auto& opcodeAMD64 : kOpcodesAMD64) {
550 if (CompilerKit::find_word(line, opcodeAMD64.fName)) {
551 return err_str;
552 }
553 }
554
555 err_str += "\nUnrecognized instruction -> " + line;
556
557 return err_str;
558}

560bool CompilerKit::EncoderAMD64::WriteNumber(const std::size_t& pos, std::string& jump_label) {
561 if (!isdigit(jump_label[pos])) return false;
562
563 switch (jump_label[pos + 1]) {
564 case 'x': {
565 if (auto res = strtol(jump_label.substr(pos + 2).c_str(), nullptr, 16); !res) {
566 if (errno != 0) {
567 Detail::print_error("invalid hex number: " + jump_label, "CompilerKit");
568 throw std::runtime_error("invalid_hex");
569 }
570 }
571
572 CompilerKit::NumberCast64 num =
573 CompilerKit::NumberCast64(strtol(jump_label.substr(pos + 2).c_str(), nullptr, 16));
574
575 for (char& i : num.number) {
576 if (i == 0) i = 0xFF;
577
578 kAppBytes.push_back(i);
579 }
580
581 if (kVerbose) {
582 kStdOut << "AssemblerAMD64: Found a base 16 number here: " << jump_label.substr(pos)
583 << "\n";
584 }
585
586 return true;
587 }
588 case 'b': {
589 if (auto res = strtol(jump_label.substr(pos + 2).c_str(), nullptr, 2); !res) {
590 if (errno != 0) {
591 Detail::print_error("invalid binary number: " + jump_label, "CompilerKit");
592 throw std::runtime_error("invalid_bin");
593 }
594 }
595
596 CompilerKit::NumberCast64 num =
597 CompilerKit::NumberCast64(strtol(jump_label.substr(pos + 2).c_str(), nullptr, 2));
598
599 if (kVerbose) {
600 kStdOut << "AssemblerAMD64: Found a base 2 number here: " << jump_label.substr(pos) << "\n";
601 }
602
603 for (char& i : num.number) {
604 if (i == 0) i = 0xFF;
605
606 kAppBytes.push_back(i);
607 }
608
609 return true;
610 }
611 case 'o': {
612 if (auto res = strtol(jump_label.substr(pos + 2).c_str(), nullptr, 7); !res) {
613 if (errno != 0) {
614 Detail::print_error("invalid octal number: " + jump_label, "CompilerKit");
615 throw std::runtime_error("invalid_octal");
616 }
617 }
618
619 CompilerKit::NumberCast64 num =
620 CompilerKit::NumberCast64(strtol(jump_label.substr(pos + 2).c_str(), nullptr, 7));
621
622 if (kVerbose) {
623 kStdOut << "AssemblerAMD64: Found a base 8 number here: " << jump_label.substr(pos) << "\n";
624 }
625
626 for (char& i : num.number) {
627 if (i == 0) i = 0xFF;
628
629 kAppBytes.push_back(i);
630 }
631
632 return true;
633 }
634 default: {
635 break;
636 }
637 }
638
639 /* check for errno and stuff like that */
640 if (auto res = strtol(jump_label.substr(pos).c_str(), nullptr, 10); !res) {
641 if (errno != 0) {
642 return false;
643 }
644 }
645
646 CompilerKit::NumberCast64 num =
647 CompilerKit::NumberCast64(strtol(jump_label.substr(pos).c_str(), nullptr, 10));
648
649 for (char& i : num.number) {
650 if (i == 0) i = 0xFF;
651
652 kAppBytes.push_back(i);
653 }
654
655 if (kVerbose) {
656 kStdOut << "AssemblerAMD64: Found a base 10 number here: " << jump_label.substr(pos) << "\n";
657 }
658
659 return true;
660}
