67BasicString StringBuilder::FromInt(const char* fmt, int i) {
68 if (!fmt) return BasicString(0);
69
70 Char result[sizeof(int64_t)] = {0};
71 if (!to_str(result, sizeof(int64_t), i)) return BasicString(0);
72
73 const SizeType fmt_len = string_length(fmt);
74 const SizeType res_len = string_length(result);
75
76 BasicString output(fmt_len + res_len);
77 bool inserted = false;
78
79 for (SizeType idx = 0; idx < fmt_len; ++idx) {
80 if (!inserted && fmt[idx] == '%') {
81 output += result;
82 inserted = true;
83 continue;
84 }
85 output += Char{fmt[idx]};
86 }
87
88 return output;
89}
