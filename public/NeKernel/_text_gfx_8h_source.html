148inline Kernel::Void fb_render_string_for_bitmap(const Kernel::UInt8* bitmap,
149 const Kernel::SizeT x_sz, const Kernel::SizeT y_sz,
150 Kernel::Int32& x_dst, Kernel::Int32& y_dst,
151 Kernel::Int32& color) {
152 Kernel::SizeT x, y;
153 Kernel::SizeT set;
154
155 x = 0;
156 y = 0;
157 set = 0;
158
159 for (; y < y_sz; ++y) {
160 for (x = 0; x < x_sz; ++x) {
161 set = bitmap[x] & (1 << y);
162
163 if (set) {
164 FBDrawInRegion(color, 1, 1, ((x_dst) + x), ((y_dst) + y));
165 }
166 }
167 }
