45 Char buf[kUUIDSize];

47 for (SizeT index = 0; index < 16; ++index) {
48 buf[index] = seq.Leak()->fU8[index] + kUUIDAsciiBegin;

52 buf[index] = seq.Leak()->fU16[index] + kUUIDAsciiBegin;

54

56 buf[index] = seq.Leak()->fU32[index] + kUUIDAsciiBegin;
57 }
58
59 auto view = KStringBuilder::Construct(buf);
60
61 if (view) return ErrorOr<Ref<BasicKString<>>>{view.Leak()};
62
63 return ErrorOr<Ref<BasicKString<>>>{-1};
