35 STATIC UInt64 user_fnv_generator(const Char* password, User* user) {
36 if (!password || !user) return 0;
37 if (*password == 0) return 0;

39 kout << "user_fnv_generator: Hashing user password...\r";
40
41 const UInt64 FNV_OFFSET_BASIS = 0xcbf29ce484222325ULL;
42 const UInt64 FNV_PRIME = 0x100000001b3ULL;

44 UInt64 hash = FNV_OFFSET_BASIS;

46 while (*password) {
47 hash ^= (Utf8Char) (*password++);
