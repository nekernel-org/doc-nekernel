60EXTERN_C UIntPtr mm_get_page_addr(VoidPtr virt) {
61 const UInt64 kVMAddr = (UInt64) virt;
62 const UInt64 kMask9Bits = 0x1FFULL;
63 const UInt64 kPageOffsetMask = 0xFFFULL;
64
65 UInt64 cr3 = (UInt64) hal_read_cr3() & ~kPageOffsetMask;
66
67 // Level 4
68 auto pml4 = reinterpret_cast<UInt64*>(cr3);
69 UInt64 pml4e = pml4[(kVMAddr >> 39) & kMask9Bits];
70
71 if (!(pml4e & 1)) return 0;
72
73 // Level 3
74 auto pdpt = reinterpret_cast<UInt64*>(pml4e & ~kPageOffsetMask);
75 UInt64 pdpte = pdpt[(kVMAddr >> 30) & kMask9Bits];
76
77 if (!(pdpte & 1)) return 0;
78
79 // Level 2
80 auto pd = reinterpret_cast<UInt64*>(pdpte & ~kPageOffsetMask);
81 UInt64 pde = pd[(kVMAddr >> 21) & kMask9Bits];
82
83 if (!(pde & 1)) return 0;
84
85 // 1 GiB page support
86 if (pde & (1 << 7)) {
87 return (pde & ~((1ULL << 30) - 1)) | (kVMAddr & ((1ULL << 30) - 1));
88 }
89
90 // Level 1
91 auto pt = reinterpret_cast<UInt64*>(pde & ~kPageOffsetMask);
92 Detail::PTE* pte = (Detail::PTE*) pt[(kVMAddr >> 12) & kMask9Bits];
93
94 if (!pte->Present) return 0;
95
96 mmi_page_status((Detail::PTE*) pte);
97
98 return (pte->PhysicalAddress << 12) | (kVMAddr & 0xFFF);
99}

104EXTERN_C Int32 mm_memory_fence(VoidPtr virtual_address) {
105 if (!virtual_address || !mm_get_page_addr(virtual_address)) return kErrorInvalidData;
106
107 asm volatile("clflush (%0)" : : "r"(virtual_address) : "memory");
108 asm volatile("mfence" ::: "memory");
109
110 return kErrorSuccess;
111}

120EXTERN_C Int32 mm_map_page(VoidPtr virtual_address, VoidPtr physical_address, UInt32 flags,
121 UInt32 level) {
122 if (physical_address == 0) return kErrorInvalidData;
123
124 NE_UNUSED(level);
125
126 const UInt64 kVMAddr = (UInt64) virtual_address;
127 constexpr UInt64 kMask9 = 0x1FF;
128 constexpr UInt64 kPageMask = 0xFFF;
129
130 UInt64 cr3 = (UIntPtr) hal_read_cr3() & ~kPageMask;
131
132 auto pml4 = reinterpret_cast<UInt64*>(cr3);
133 UInt64 pml4e = pml4[(kVMAddr >> 39) & kMask9];
134
135 if (!(pml4e & 1)) return kErrorInvalidData;
136
137 UInt64* pdpt = reinterpret_cast<UInt64*>(pml4e & ~kPageMask);
138 UInt64 pdpte = pdpt[(kVMAddr >> 30) & kMask9];
139
140 if (!(pdpte & 1)) return kErrorInvalidData;
141
142 UInt64* pd = reinterpret_cast<UInt64*>(pdpte & ~kPageMask);
143 UInt64 pde = pd[(kVMAddr >> 21) & kMask9];
144
145 if (!(pde & 1)) return kErrorInvalidData;
146
147 UInt64* pt = reinterpret_cast<UInt64*>(pde & ~kPageMask);
148 Detail::PTE* pte = (Detail::PTE*) pt[(kVMAddr >> 12) & kMask9];
149
150 pte->Present = !!(flags & kMMFlagsPresent);
151 pte->Wr = !!(flags & kMMFlagsWr);
152 pte->User = !!(flags & kMMFlagsUser);
153 pte->Nx = !!(flags & kMMFlagsNX);
154 pte->Pcd = !!(flags & kMMFlagsPCD);
155 pte->Pwt = !!(flags & kMMFlagsPwt);
156
157 pte->PhysicalAddress = ((UIntPtr) (physical_address)) >> 12;
158
159 hal_invl_tlb(virtual_address);
160
161 mm_memory_fence(virtual_address);
162
163 mmi_page_status(pte);
164
165 return kErrorSuccess;
166}
