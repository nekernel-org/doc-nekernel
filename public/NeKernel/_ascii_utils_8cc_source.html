14Int32 rt_string_cmp(const Char* src, const Char* cmp, Size size) {
15 for (Size i = 0; i < size; ++i) {
16 if (src[i] != cmp[i]) return static_cast<Int32>(src[i]) - static_cast<Int32>(cmp[i]);
17 }
18 return 0;
19}

33const Char* rt_alloc_string(const Char* src) {
34 SizeT slen = rt_string_len(src);

39 reinterpret_cast<voidPtr>(buffer), slen, slen + 1) < 0) {

48Int rt_copy_memory_safe(const voidPtr src, voidPtr dst, Size len, Size dst_size) {
49 if (!src || !dst || len > dst_size) {
50 if (dst && dst_size) {
51 rt_set_memory_safe(dst, 0, dst_size, dst_size);
52 }
53 return -1;
54 }
55 auto s = reinterpret_cast<const UInt8*>(src);
56 auto d = reinterpret_cast<UInt8*>(dst);
57 for (Size i = 0; i < len; ++i) d[i] = s[i];
58 return static_cast<Int>(len);
59}

61voidPtr rt_set_memory_safe(voidPtr dst, UInt32 value, Size len, Size dst_size) {
62 if (!dst || len > dst_size) return nullptr;
63 auto p = reinterpret_cast<UInt8*>(dst);
64 UInt8 v = static_cast<UInt8>(value & 0xFF);
65 for (Size i = 0; i < len; ++i) p[i] = v;
66 return dst;
67}

123Bool rt_to_string(Char* str, UInt64 value, Int32 base) {
124#ifdef __NE_AMD64__
125 Int i = 0;
126 do {
127 str[i++] = rt_to_char(value, base);
128 value /= base;
129 } while (value);
130 str[i] = '\0';
131 // in-place
132 for (Int j = 0; j < i / 2; ++j) {
133 Char tmp = str[j];
134 str[j] = str[i - j - 1];
135 str[i - j - 1] = tmp;
136 }
137#endif
138 return true;
139}

141VoidPtr rt_string_in_string(const Char* haystack, const Char* needle) {
142 SizeT needle_len = rt_string_len(needle);
143 SizeT hay_len = rt_string_len(haystack);
144
145 if (needle_len > hay_len) return nullptr;
146 for (SizeT i = 0; i <= hay_len - needle_len; ++i) {
147 if (rt_string_cmp(haystack + i, needle, needle_len) == 0) {
148 return reinterpret_cast<voidPtr>(const_cast<Char*>(haystack + i));
149 }
150 }
151 return nullptr;
152}
