75const Char* KStringBuilder::FromBool(const Char* fmt, bool i) {
76 if (!fmt) return ("?");
77
78 const Char* boolean_expr = i ? "YES" : "NO";
79 Char* ret = (Char*) RTL_ALLOCA(rt_string_len(boolean_expr) + rt_string_len(fmt));
80
81 if (!ret) return ("?");
82
83 const auto fmt_len = rt_string_len(fmt);
84 const auto res_len = rt_string_len(boolean_expr);
85
86 for (Size idx = 0; idx < fmt_len; ++idx) {
87 if (fmt[idx] == '%') {
88 SizeT result_cnt = idx;
89
90 for (auto y_idx = idx; y_idx < res_len; ++y_idx) {
91 ret[result_cnt] = boolean_expr[y_idx];
92 ++result_cnt;
93 }
94
95 break;
96 }
97
98 ret[idx] = fmt[idx];
99 }
100
101 return ret;
102}

131const Char* KStringBuilder::Format(const Char* fmt, const Char* fmt2) {
132 if (!fmt || !fmt2) return ("?");
133
134 Char* ret = (Char*) RTL_ALLOCA(sizeof(char) * (rt_string_len(fmt2) + rt_string_len(fmt)));
135
136 if (!ret) return ("?");
137
138 const auto len = rt_string_len(fmt);
139
140 for (Size idx = 0; idx < len; ++idx) {
141 if (fmt[idx] == '%' && idx < rt_string_len(fmt) && fmt[idx] == 's') {
142 Size result_cnt = idx;
143
144 for (Size y_idx = 0; y_idx < rt_string_len(fmt2); ++y_idx) {
145 ret[result_cnt] = fmt2[y_idx];
146 ++result_cnt;
147 }
148 }
149
150 ret[idx] = fmt[idx];
151 }
152
153 return ret;
154}
