109 inline TerminalDevice _write_number(const Long& x, TerminalDevice& term) {
110 UInt64 y = (x > 0 ? x : -x) / 10;
111 UInt64 h = (x > 0 ? x : -x) % 10;
112
113 if (y) _write_number(y, term);
114
115 /* fail if the number is not base-10 */
116 if (h > 10) {
117 _write_number('?', term);
118 return term;
119 }
120
121 if (y == ~0UL) y = -y;
122
123 const Char kNumbers[11] = "0123456789";
124
125 Char buf[2];
126 buf[0] = kNumbers[h];
127 buf[1] = 0;
128
129 term.operator<<(buf);
130 return term;
131 }
132
133 inline TerminalDevice _write_number_hex(const Long& x, TerminalDevice& term) {
134 UInt64 y = (x > 0 ? x : -x) / 16;
135 UInt64 h = (x > 0 ? x : -x) % 16;
136
137 if (y) _write_number_hex(y, term);
138
139 /* fail if the hex number is not base-16 */
140 if (h > 16) {
141 _write_number_hex('?', term);
142 return term;
143 }
144
145 if (y == ~0UL) y = -y;
146
147 const Char kNumbers[17] = "0123456789ABCDEF";
148
149 Char buf[2];
150 buf[0] = kNumbers[h];
151 buf[1] = 0;
152
153 term.operator<<(buf);
154 return term;
155 }
