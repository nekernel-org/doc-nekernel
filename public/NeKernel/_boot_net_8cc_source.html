82STATIC Void bootnet_read_ip_packet(BOOTNET_INTERNET_HEADER inet,
83 BOOTNET_INTERNET_HEADER** inet_out) {
84 NE_UNUSED(inet);
85
86 kEfiProtocol->Start(kEfiProtocol);
87
88 UInt32 size_inet = sizeof(inet);
89
91
93 if (kEfiProtocol->Receive(kEfiProtocol, &size_inet, (UInt32*) &inet.Length, (VoidPtr) &inet,
94 nullptr, nullptr, nullptr) == kEfiOk) {
95 BOOTNET_INTERNET_HEADER* out = nullptr;
96
97 BS->AllocatePool(EfiLoaderData, sizeof(BOOTNET_INTERNET_HEADER) + inet.Length, (VoidPtr*) &out);
98
99 if (out == nullptr) {
100 kEfiProtocol->Stop(kEfiProtocol);
101 kEfiProtocol->Shutdown(kEfiProtocol);
102 return;
103 }
104
105 SetMem(out, 0, sizeof(BOOTNET_INTERNET_HEADER));
106 SetMem(out->Data, 0, inet.Length);
107
108 size_inet = sizeof(BOOTNET_INTERNET_HEADER);
109
110 auto full_size = size_inet + inet.Length;
111
113 kEfiProtocol->Receive(kEfiProtocol, &size_inet, (UInt32*) &full_size, (VoidPtr) out, nullptr,
114 nullptr, nullptr);
115
116 *inet_out = out;
117 }
118
119 kEfiProtocol->Stop(kEfiProtocol);
120 kEfiProtocol->Shutdown(kEfiProtocol);
121}
