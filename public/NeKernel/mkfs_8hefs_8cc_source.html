53 auto opt_L = mkfs::get_option<char>(args, "-L");

62 }
63
64 auto opt_S = mkfs::get_option<char>(args, "-S");
65 unsigned long long gb = 0;

69 return EXIT_FAILURE;

78 auto opt_b = mkfs::get_option<char>(args, "-b");

167 return EXIT_FAILURE;
168 }
169
170 output_device.write(reinterpret_cast<const char*>(&boot_node), sizeof(boot_node));
171 if (!output_device.good()) {
172 mkfs::console_out() << "hefs: error: Unable to write BootNode to output device: " << output_path
173 << "\n";
174 return EXIT_FAILURE;
175 }
176
177 output_device.seekp(static_cast<std::streamoff>(boot_node.startIND));
178 if (!output_device.good()) {
179 mkfs::console_out() << "hefs: error: Failed seek to startIND.\n";
180 return EXIT_FAILURE;
181 }
182
183 output_device.flush();
184 output_device.close();
185
186 mkfs::console_out() << "hefs: info: Wrote filesystem to output device: " << output_path << "\n";
187 return EXIT_SUCCESS;
