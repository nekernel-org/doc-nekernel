63EXTERN_C int strncmp(const char* whatToCheck, const char* whatToCheckRight, size_t max_len) {
64 size_t i = 0;
65 while (i < max_len && whatToCheck[i] == whatToCheckRight[i]) {
66 if (whatToCheck[i] == '\0') return 0;
67 ++i;
68 }

70 return 0;
71 }
