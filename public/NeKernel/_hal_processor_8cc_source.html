19inline Bool hal_has_msr() noexcept {
20 static UInt32 eax, unused, edx; // eax, edx
21
22 __get_cpuid(1, &eax, &unused, &unused, &edx);
23
24 // edx returns the flag for MSR (which is 1 shifted to 5.)
25 return edx & (1 << 5);
26}

28Void hal_get_msr(UInt32 msr, UInt32* lo, UInt32* hi) noexcept {
29 if (!lo || !hi) return;
30 asm volatile("rdmsr" : "=a"(*lo), "=d"(*hi) : "c"(msr));
