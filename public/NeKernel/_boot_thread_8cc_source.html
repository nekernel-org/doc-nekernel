28BootThread::BootThread(VoidPtr blob) : fStartAddress(nullptr), fBlob(blob) {
29 // detect the format.
30 const Char* blob_bytes = reinterpret_cast<char*>(fBlob);
31
32 BootTextWriter writer;
33
34 if (!blob_bytes) {
35 // failed to provide a valid pointer.
36 return;
37 }
38
39 if (blob_bytes[0] == kMagMz0 && blob_bytes[1] == kMagMz1) {
40 LDR_EXEC_HEADER_PTR header_ptr = CF::ldr_find_exec_header(blob_bytes);
41 LDR_OPTIONAL_HEADER_PTR opt_header_ptr = CF::ldr_find_opt_exec_header(blob_bytes);
42
43 if (!header_ptr || !opt_header_ptr) return;
44
45#ifdef __NE_AMD64__
46 if (header_ptr->Machine != kPeMachineAMD64 || header_ptr->Signature != kPeSignature) {
47 writer.Write("BootZ: Not a PE32+ executable.\r");
48 return;
49 }
50#elif defined(__NE_ARM64__)
51 if (header_ptr->Machine != kPeMachineARM64 || header_ptr->Signature != kPeSignature) {
52 writer.Write("BootZ: Not a PE32+ executable.\r");
53 return;
54 }
55#endif // __NE_AMD64__ || __NE_ARM64__
56
57 writer.Write("BootZ: PE32+ executable detected (NeKernel Subsystem).\r");
58
59 auto numSecs = header_ptr->NumberOfSections;
60
61 writer.Write("BootZ: Major Linker Ver: ").Write(opt_header_ptr->MajorLinkerVersion).Write("\r");
62 writer.Write("BootZ: Minor Linker Ver: ").Write(opt_header_ptr->MinorLinkerVersion).Write("\r");
63 writer.Write("BootZ: Major Subsystem Ver: ")
64 .Write(opt_header_ptr->MajorSubsystemVersion)
65 .Write("\r");
66 writer.Write("BootZ: Minor Subsystem Ver: ")
67 .Write(opt_header_ptr->MinorSubsystemVersion)
68 .Write("\r");
69 writer.Write("BootZ: Magic: ").Write(header_ptr->Signature).Write("\r");
70
71 EfiPhysicalAddress loadStartAddress = opt_header_ptr->ImageBase;
72
73 writer.Write("BootZ: Image-Base: ").Write(loadStartAddress).Write("\r");
74
75 fStack = new UInt8[kBootThreadSz];
76
77 if (!fStack) {
78 writer.Write("BootZ: Unable to allocate stack.\r");
79 return;
80 }
81
82 LDR_SECTION_HEADER_PTR sectPtr =
83 (LDR_SECTION_HEADER_PTR) (((Char*) opt_header_ptr) + header_ptr->SizeOfOptionalHeader);
84
85 constexpr auto sectionForCode = ".text";
86 constexpr auto sectionForBootZ = ".ldr";
87
88 for (SizeT sectIndex = 0; sectIndex < numSecs; ++sectIndex) {
89 LDR_SECTION_HEADER_PTR sect = &sectPtr[sectIndex];
90
91 auto sz = sect->VirtualSize;
92
93 if (sect->SizeOfRawData > 0) sz = sect->SizeOfRawData;
94
95 SetMem((VoidPtr) (loadStartAddress + sect->VirtualAddress), 0, sz);
96
97 if (sect->SizeOfRawData > 0) {
98 CopyMem((VoidPtr) (loadStartAddress + sect->VirtualAddress),
99 (VoidPtr) ((UIntPtr) fBlob + sect->PointerToRawData), sect->SizeOfRawData);
100 }
101
102 if (StrCmp(sectionForCode, sect->Name) == 0) {
103 fStartAddress =
104 (VoidPtr) ((UIntPtr) loadStartAddress + opt_header_ptr->AddressOfEntryPoint);
105 writer.Write("BootZ: Executable entry address: ")
106 .Write((UIntPtr) fStartAddress)
107 .Write("\r");
108
110
111#ifdef __NE_ARM64__
112
113#endif
114 } else if (StrCmp(sectionForBootZ, sect->Name) == 0) {
115 struct HANDOVER_INFORMATION_STUB {
116 UInt64 HandoverMagic;
117 UInt32 HandoverType;
118 UInt32 HandoverPad;
119 UInt32 HandoverArch;
120 }* handover_struc =
121 (struct HANDOVER_INFORMATION_STUB*) ((UIntPtr) fBlob + sect->PointerToRawData);
122
123 if (handover_struc->HandoverMagic != kHandoverMagic) {
124#ifdef __NE_AMD64__
125 if (handover_struc->HandoverArch != HEL::kArchAMD64) {
126 writer.Write("BootZ: Not an handover header, bad CPU...\r");
127 }
128#elif defined(__NE_ARM64__)
129 if (handover_struc->HandoverArch != HEL::kArchARM64) {
130 writer.Write("BootZ: Not an handover header, bad CPU...\r");
131 }
132#endif
133
134 writer.Write("BootZ: Not an handover header...\r");
135 ::Boot::Stop();
136 }
137 }
138
139 writer.Write("BootZ: Raw offset: ")
140 .Write(sect->PointerToRawData)
141 .Write(" of ")
142 .Write(sect->Name)
143 .Write("\r");
144
145 CopyMem((VoidPtr) (loadStartAddress + sect->VirtualAddress),
146 (VoidPtr) ((UIntPtr) fBlob + sect->PointerToRawData), sect->SizeOfRawData);
147 }
148 } else if (blob_bytes[0] == kPefMagic[0] && blob_bytes[1] == kPefMagic[1] &&
149 blob_bytes[2] == kPefMagic[2] && blob_bytes[3] == kPefMagic[3]) {
150 // ========================================= //
151 // PEF executable has been detected.
152 // ========================================= //
153
154 fStartAddress = nullptr;
155
156 writer.Write("BootZ: PEF executable detected, won't load it.\r");
157 writer.Write("BootZ: note: PEF executables aren't supported for now.\r");
158 } else {
159 writer.Write("BootZ: Invalid Executable.\r");
160 }
161}

164Int32 BootThread::Start(HEL::BootInfoHeader* handover, Bool own_stack) {
165 if (!fStartAddress) {
166 return kEfiFail;
167 }
168
169 if (!handover) {
170 return kEfiFail;
171 }
172
173 fHandover = handover;
174
175 BootTextWriter writer;
176
177 writer.Write("BootZ: Starting: ").Write(fBlobName).Write("\r");
178 writer.Write("BootZ: Handover address: ").Write((UIntPtr) fHandover).Write("\r");
179
180 if (own_stack) {
181 writer.Write("BootZ: Using it's own stack.\r");
182 writer.Write("BootZ: Stack address: ").Write((UIntPtr) &fStack[kBootThreadSz - 1]).Write("\r");
183 writer.Write("BootZ: Stack size: ").Write(kBootThreadSz).Write("\r");
184
185 fHandover->f_StackTop = &fStack[kBootThreadSz - 1];
186 fHandover->f_StackSz = kBootThreadSz;
187
188 auto ret = rt_jump_to_address(fStartAddress, fHandover, &fStack[kBootThreadSz - 1]);
189
190 // we don't need the stack anymore.
191
192 delete[] fStack;
193 fStack = nullptr;
194
195 return ret;
196 } else {
197 writer.Write("BootZ: Using the bootloader's stack.\r");
198
199 return reinterpret_cast<HEL::HandoverProc>(fStartAddress)(fHandover);
200 }
201
202 return kEfiFail;
203}
