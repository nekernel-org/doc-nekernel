29EXTERN_C Void rtl_init_nic_rtl8139(UInt16 io_base) noexcept {
30 if (kTXEnabled) return;
31
32 kRTLIOBase = io_base;
33
34 MUST_PASS(io_base != 0xFFFF);
35
36 kRXBuffer = reinterpret_cast<UInt8*>(rtl_dma_alloc(sizeof(UInt8) * kRXBufferSize, 0));
37
38 MUST_PASS(kRXBuffer);
39
41
42 rt_out8(io_base + 0x37, 0x10);
43
44 UInt16 timeout = 0U;
45
46 while (rt_in8(io_base + 0x37) & 0x10) {
47 ++timeout;
48 if (timeout > 0x1000) break;
49 }
50
51 if (timeout <= 0x1000) {
52 ke_panic(RUNTIME_CHECK_BAD_BEHAVIOR, "RTL8139: Reset failed");
53 return;
54 }
55
56 rt_out32(io_base + 0x30, (UInt32) (UIntPtr) kRXBuffer);
57
58 rt_out8(io_base + 0x37, 0x0C);
59
60 rt_out32(io_base + 0x44, 0xF | (1 << 7));
61
62 rt_out16(io_base + 0x3C, 0x0005);
63
64 kTXEnabled = YES;
65}

72EXTERN_C void rtl_rtl8139_interrupt_handler() {
73 if (kRTLIOBase == 0xFFFF) return;
74
75 UInt16 status = rt_in16(kRTLIOBase + 0x3E);
76 rt_out16(kRTLIOBase + 0x3E, status);
77
78 if (status & 0x01) {
79 // While we receive data.
80 while ((rt_in8(kRTLIOBase + 0x37) & 0x01) == 0) {
81 // We grab an offset from the RX buffer.
82 UInt32 offset = kRXOffset % kRXBufferSize;
83
84 // If the offset is too high, we reset it.
85 if (offset >= (kRXBufferSize - 16)) {
86 kRXOffset = 0UL;
87 offset = 0UL;
88 }
89
90 volatile UInt8* packet = kRXBuffer + offset + 4;
91 UInt16 len = *(UInt16*) (kRXBuffer + offset + 2);
92
93 kRXUpperLayer[(offset + 4)] = *packet;
94 kRXOffset += (len + 4);
95
96 rt_out16(kRTLIOBase + 0x38, (UInt16) (kRXOffset - 16));
97 }
98 }
99
100 if (!(status & 0x04)) {
101 err_global_get() = kErrorNoNetwork;
102 }
103}
