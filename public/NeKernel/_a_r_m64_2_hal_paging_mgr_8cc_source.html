16EXTERN_C UIntPtr mm_get_page_addr(VoidPtr virtual_address) {
17 if (!virtual_address) return 0;
18
19 UInt64 ttbr0_val = 0;
20
21 asm volatile("mrs %0, ttbr0_el1" : "=r"(ttbr0_val));
22 volatile UInt64* l1_table = reinterpret_cast<volatile UInt64*>(ttbr0_val);
23
24 UInt64 l1_idx = (reinterpret_cast<UIntPtr>(virtual_address) >> 39) & 0x1FF;
25 UInt64 l2_idx = (reinterpret_cast<UIntPtr>(virtual_address) >> 30) & 0x1FF;
26 UInt64 l3_idx = (reinterpret_cast<UIntPtr>(virtual_address) >> 21) & 0x1FF;
27
28 if (!l1_table[l1_idx]) return 0;
29
30 volatile UInt64* l2_table = reinterpret_cast<volatile UInt64*>(l1_table[l1_idx] & ~0xFFFUL);
31
32 if (!l2_table[l2_idx]) return 0;
33
34 volatile UInt64* l3_table = reinterpret_cast<volatile UInt64*>(l2_table[l2_idx] & ~0xFFFUL);
35
36 if (!l3_table[l3_idx]) return 0;
37
38 return (l3_table[l3_idx] & ~0xFFFUL);
39}

46EXTERN_C Int32 mm_map_page(VoidPtr virtual_address, VoidPtr physical_address, UInt32 flags,
47 UInt32 level) {
48 if (!virtual_address || !flags || !physical_address) return kErrorInvalidData;
49
50 UInt64 ttbr0_val = 0;
51
52 asm volatile("mrs %0, ttbr0_el1" : "=r"(ttbr0_val));
53 volatile UInt64* l1_table = reinterpret_cast<volatile UInt64*>(ttbr0_val);
54
55 UInt64 l1_idx = (reinterpret_cast<UIntPtr>(virtual_address) >> 39) & 0x1FF;
56 UInt64 l2_idx = (reinterpret_cast<UIntPtr>(virtual_address) >> 30) & 0x1FF;
57 UInt64 l3_idx = (reinterpret_cast<UIntPtr>(virtual_address) >> 21) & 0x1FF;
58
59 if (!l1_table[l1_idx]) return kErrorInvalidData;
60
61 volatile UInt64* l2_table = reinterpret_cast<volatile UInt64*>(l1_table[l1_idx] & ~0xFFFUL);
62
63 if (!l2_table[l2_idx]) return kErrorInvalidData;
64
65 volatile UInt64* l3_table = reinterpret_cast<volatile UInt64*>(l2_table[l2_idx] & ~0xFFFUL);
66
67 l3_table[l3_idx] = (reinterpret_cast<UInt64>(physical_address) & ~0xFFFUL) | flags;
68
69 switch (level) {
70 case 2: {
71 l3_table[l3_idx] = (reinterpret_cast<UInt64>(physical_address) & ~0xFFFUL) | flags;
72 return kErrorSuccess;
73 }
74 case 1: {
75 l1_table[l1_idx] = (reinterpret_cast<UInt64>(physical_address) & ~0xFFFUL) | flags;
76 return kErrorSuccess;
77 }
78 case 0: {
79 l1_table[l1_idx] = (reinterpret_cast<UInt64>(physical_address) & ~0xFFFUL) | flags;
80 return kErrorSuccess;
81 }
82 }
83
84 return kErrorInvalidData;
85}
