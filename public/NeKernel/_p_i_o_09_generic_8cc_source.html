34static Boolean drv_pio_std_wait_io(UInt16 IO) {
35 for (int i = 0; i < 400; i++) rt_in8(IO + ATA_REG_STATUS);
36
37ATAWaitForIO_Retry:
38 auto stat_rdy = rt_in8(IO + ATA_REG_STATUS);
39
40 if ((stat_rdy & ATA_SR_BSY)) goto ATAWaitForIO_Retry;
41
42ATAWaitForIO_Retry2:
43 stat_rdy = rt_in8(IO + ATA_REG_STATUS);
44
45 if (stat_rdy & ATA_SR_ERR) return false;
46
47 if (!(stat_rdy & ATA_SR_DRDY)) goto ATAWaitForIO_Retry2;
48
49 return true;
50}

52STATIC Void drv_pio_std_select(UInt16 Bus) {
53 if (Bus == ATA_PRIMARY_IO)
54 rt_out8(Bus + ATA_REG_HDDEVSEL, ATA_PRIMARY_SEL);
55 else
56 rt_out8(Bus + ATA_REG_HDDEVSEL, ATA_SECONDARY_SEL);

59Boolean drv_pio_std_init(UInt16 Bus, UInt8 Drive, UInt16& OutBus, UInt8& OutMaster) {
60 UInt16 IO = Bus;
61
62 NE_UNUSED(Drive);
63
64 drv_pio_std_select(IO);
65
66 // Bus init, NEIN bit.
67 rt_out8(IO + ATA_REG_NEIN, 1);
68
69 // identify until it's good.
70ATAInit_Retry:
71 auto stat_rdy = rt_in8(IO + ATA_REG_STATUS);
72
73 if (stat_rdy & ATA_SR_ERR) {
74 return false;
75 }
76
77 if ((stat_rdy & ATA_SR_BSY)) goto ATAInit_Retry;
78
79 OutBus = (Bus == ATA_PRIMARY_IO) ? ATA_PRIMARY_IO : ATA_SECONDARY_IO;
80 OutMaster = (Bus == ATA_PRIMARY_IO) ? ATA_MASTER : ATA_SLAVE;
81
82 drv_pio_std_select(IO);
83
84 rt_out8(OutBus + ATA_REG_COMMAND, ATA_CMD_IDENTIFY);
85
86 while (!(rt_in8(IO + ATA_REG_STATUS) & ATA_SR_DRQ))
87 ;
88
91
92 for (SizeT i = 0ul; i < kATADataLen; ++i) {
93 kATAIdentifyData[i] = rt_in16(OutBus + ATA_REG_DATA);
94 }
95
96 for (Int32 i = 0; i < 20; i++) {
97 kATADiskModel[i * 2] = (kATAIdentifyData[27 + i] >> 8) & 0xFF;
98 kATADiskModel[i * 2 + 1] = kATAIdentifyData[27 + i] & 0xFF;
99 }
100
101 kATADiskModel[40] = '\0';
102
103 (Void)(kout << "Drive Model: " << kATADiskModel << kendl);
104
105 return true;
106}

108Void drv_pio_std_read(UInt64 Lba, UInt16 IO, UInt8 Master, Char* Buf, SizeT SectorSz, SizeT Size) {
109 Lba /= SectorSz;
110
111 UInt8 Command = ((!Master) ? 0xE0 : 0xF0);
112
113 drv_pio_std_wait_io(IO);
114 drv_pio_std_select(IO);
115
116 rt_out8(IO + ATA_REG_HDDEVSEL, (Command) | (((Lba) >> 24) & 0x0F));
117
118 rt_out8(IO + ATA_REG_SEC_COUNT0, ((Size + SectorSz) / SectorSz));
119
120 rt_out8(IO + ATA_REG_LBA0, (Lba) &0xFF);
121 rt_out8(IO + ATA_REG_LBA1, (Lba) >> 8);
122 rt_out8(IO + ATA_REG_LBA2, (Lba) >> 16);
123 rt_out8(IO + ATA_REG_LBA3, (Lba) >> 24);
124
125 rt_out8(IO + ATA_REG_COMMAND, ATA_CMD_READ_PIO);
126
127 while (!(rt_in8(IO + ATA_REG_STATUS) & ATA_SR_DRQ))
128 ;
129
130 for (SizeT IndexOff = 0; IndexOff < Size; IndexOff += 2) {
131 drv_pio_std_wait_io(IO);
132
133 auto in = rt_in16(IO + ATA_REG_DATA);
134
135 Buf[IndexOff] = in & 0xFF;
136 Buf[IndexOff + 1] = (in >> 8) & 0xFF;
137 }
138}

140Void drv_pio_std_write(UInt64 Lba, UInt16 IO, UInt8 Master, Char* Buf, SizeT SectorSz, SizeT Size) {
141 Lba /= SectorSz;
142
143 UInt8 Command = ((!Master) ? 0xE0 : 0xF0);
144
145 drv_pio_std_wait_io(IO);
146 drv_pio_std_select(IO);
147
148 rt_out8(IO + ATA_REG_HDDEVSEL, (Command) | (((Lba) >> 24) & 0x0F));
149
150 rt_out8(IO + ATA_REG_SEC_COUNT0, ((Size + SectorSz) / SectorSz));
151
152 rt_out8(IO + ATA_REG_LBA0, (Lba) &0xFF);
153 rt_out8(IO + ATA_REG_LBA1, (Lba) >> 8);
154 rt_out8(IO + ATA_REG_LBA2, (Lba) >> 16);
155 rt_out8(IO + ATA_REG_LBA3, (Lba) >> 24);
156
157 rt_out8(IO + ATA_REG_COMMAND, ATA_CMD_WRITE_PIO);
158
159 while (!(rt_in8(IO + ATA_REG_STATUS) & ATA_SR_DRQ))
160 ;
161
162 for (SizeT IndexOff = 0; IndexOff < Size; IndexOff += 2) {
163 drv_pio_std_wait_io(IO);
164
165 UInt8 low = (UInt8) Buf[IndexOff];
166 UInt8 high = (IndexOff + 1 < Size) ? (UInt8) Buf[IndexOff + 1] : 0;
167 UInt16 packed = (high << 8) | low;
168
169 rt_out16(IO + ATA_REG_DATA, packed);
170 }
171}
