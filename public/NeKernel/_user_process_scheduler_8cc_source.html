88template <typename T>
89STATIC T* sched_try_go_upper_ptr_tree(T* tree) {
90 if (!tree) {
91 return nullptr;
92 }
93
94 tree = tree->Parent;
95
96 if (tree) {
97 auto tree_tmp = tree->Next;
98
99 if (!tree_tmp) {
100 return tree;
101 }
102
103 return tree_tmp;
104 }
105
106 return tree;
107}

113ErrorOr<VoidPtr> USER_PROCESS::New(SizeT sz, SizeT pad_amount) {
114 if (this->UsedMemory > kSchedMaxMemoryLimit) return ErrorOr<VoidPtr>(-kErrorHeapOutOfMemory);
115
116#ifdef __NE_VIRTUAL_MEMORY_SUPPORT__
117 auto vm_register = kKernelVM;
118
119 hal_write_cr3(this->VMRegister);
120
121 auto ptr = mm_alloc_ptr(sz, Yes, Yes, pad_amount);
122
123 hal_write_cr3(vm_register);
124#else
125 auto ptr = mm_alloc_ptr(sz, Yes, Yes, pad_amount);
126#endif
127
128 if (!this->HeapTree) {
129 this->HeapTree = new PROCESS_HEAP_TREE<VoidPtr>();
130
131 if (!this->HeapTree) {
132 this->Crash();
133 return ErrorOr<VoidPtr>(-kErrorHeapOutOfMemory);
134 }
135
136 this->HeapTree->EntryPad = pad_amount;
137 this->HeapTree->EntrySize = sz;
138
139 this->HeapTree->Entry = ptr;
140
141 this->HeapTree->Color = kBlackTreeKind;
142
143 this->HeapTree->Prev = nullptr;
144 this->HeapTree->Next = nullptr;
145 this->HeapTree->Parent = nullptr;
146 this->HeapTree->Child = nullptr;
147 } else {
148 PROCESS_HEAP_TREE<VoidPtr>* entry = this->HeapTree;
149 PROCESS_HEAP_TREE<VoidPtr>* prev_entry = entry;
150
151 BOOL is_parent = NO;
152
153 while (entry) {
154 if (entry->EntrySize < 1) break;
155
156 prev_entry = entry;
157
158 if (entry->Child && entry->Child->EntrySize > 0 && entry->Child->EntrySize == sz) {
159 entry = entry->Child;
160 is_parent = YES;
161 } else if (entry->Next && entry->Next->EntrySize > 0 && entry->Next->EntrySize == sz) {
162 is_parent = NO;
163 entry = entry->Next;
164 } else {
165 entry = sched_try_go_upper_ptr_tree(entry);
166 if (entry && entry->Color == kBlackTreeKind) break;
167 }
168 }
169
170 auto new_entry = new PROCESS_HEAP_TREE<VoidPtr>();
171
172 if (!new_entry) {
173 this->Crash();
174 return ErrorOr<VoidPtr>(-kErrorHeapOutOfMemory);
175 }
176
177 new_entry->Entry = ptr;
178 new_entry->EntrySize = sz;
179 new_entry->EntryPad = pad_amount;
180 new_entry->Parent = entry;
181 new_entry->Child = nullptr;
182 new_entry->Next = nullptr;
183 new_entry->Prev = nullptr;
184
185 new_entry->Color = kBlackTreeKind;
186 prev_entry->Color = kRedTreeKind;
187
188 if (is_parent) {
189 prev_entry->Child = new_entry;
190 new_entry->Parent = prev_entry;
191 } else {
192 prev_entry->Next = new_entry;
193 new_entry->Prev = prev_entry;
194 }
195 }
196
197 this->UsedMemory += sz;
198
199 return ErrorOr<VoidPtr>(ptr);
200}

237template <typename T>
238STATIC Void sched_free_ptr_tree(T* tree) {
239 // Deleting memory lists. Make sure to free all of them.
240 while (tree) {
241 if (tree->Entry) {
242 MUST_PASS(mm_free_ptr(tree->Entry));
243 }
244
245 auto next = tree->Next;
246
247 if (next->Child) sched_free_ptr_tree(next->Child);
248
249 tree->Child = nullptr;
250
251 mm_free_ptr(tree);
252
253 tree = nullptr;
254 tree = next;
255 }
256}

265Void USER_PROCESS::Exit(const Int32& exit_code) {
266 this->Status = exit_code > 0 ? ProcessStatusKind::kKilled : ProcessStatusKind::kFrozen;
267 this->LastExitCode = exit_code;
268 this->UTime = 0;
269
270#ifdef __NE_VIRTUAL_MEMORY_SUPPORT__
271 auto pd = kKernelVM;
272 hal_write_cr3(this->VMRegister);
273#endif
274
275 sched_free_ptr_tree(this->HeapTree);
276 this->HeapTree = nullptr;
277
278 sched_free_ptr_tree(this->FileTree);
279 this->FileTree = nullptr;
280
281#ifdef __NE_VIRTUAL_MEMORY_SUPPORT__
282 hal_write_cr3(pd);
283#endif
284
285#ifdef __NE_VIRTUAL_MEMORY_SUPPORT__
287 if (this->VMRegister) HAL::mm_free_bitmap(this->VMRegister);
288#endif
289
291 if (this->Image.fCode && mm_is_valid_ptr(this->Image.fCode)) mm_free_ptr(this->Image.fCode);
292
294 if (this->Image.fBlob && mm_is_valid_ptr(this->Image.fBlob)) mm_free_ptr(this->Image.fBlob);
295
297 if (this->StackFrame && mm_is_valid_ptr(this->StackFrame))
298 mm_free_ptr((VoidPtr) this->StackFrame);
299
301 this->Image.fBlob = nullptr;
302 this->Image.fCode = nullptr;
303 this->StackFrame = nullptr;
304
305 if (this->Kind == kExecutableDylibKind) {
306 Bool success = false;
307
308 rtl_fini_dylib_pef(*this, reinterpret_cast<IPEFDylibObject*>(this->DylibDelegate), &success);
309
310 if (!success) {
311 ke_panic(RUNTIME_CHECK_PROCESS);
312 }
313
314 this->DylibDelegate = nullptr;
315 }
316
317 this->ProcessId = 0UL;
318 this->Status = ProcessStatusKind::kFinished;
319
320 --this->ParentTeam->mProcessCur;
321}

327Bool USER_PROCESS::InitDylib() {
328 // React according to the process's kind.
329 switch (this->Kind) {
330 case USER_PROCESS::kExecutableDylibKind: {
331 this->DylibDelegate = rtl_init_dylib_pef(*this);
332
333 if (!this->DylibDelegate) {
334 this->Crash();
335 return NO;
336 }
337
338 return YES;
339 }
340 case USER_PROCESS::kExecutableKind: {
341 return NO;
342 }
343 default: {
344 break;
345 }
346 }
347
348 (Void)(kout << "Unknown process kind: " << hex_number(this->Kind) << kendl);
349 this->Crash();
350
351 return NO;
352}

360ProcessID UserProcessScheduler::Spawn(const Char* name, VoidPtr code, VoidPtr image) {
361 if (!name || !code) {
362 return -kErrorProcessFault;
363 }
364
365 if (*name == 0) {
366 return -kErrorProcessFault;
367 }
368
369 ProcessID pid = this->mTeam.mProcessCur;
370
371 if (pid > kSchedProcessLimitPerTeam) {
372 return -kErrorProcessFault;
373 }
374
375 ++this->mTeam.mProcessCur;
376
377 USER_PROCESS& process = this->mTeam.mProcessList[pid];
378
379 process.Image.fCode = code;
380 process.Image.fBlob = image;
381
382 SizeT len = rt_string_len(name);
383
384 if (len > kSchedNameLen) {
385 return -kErrorProcessFault;
386 }
387
388 rt_copy_memory_safe(reinterpret_cast<VoidPtr>(const_cast<Char*>(name)), process.Name, len,
389 kSchedNameLen);
390
391#ifdef __NE_VIRTUAL_MEMORY_SUPPORT__
392 process.VMRegister = kKernelVM;
393#else
394 process.VMRegister = 0UL;
395#endif // ifdef __NE_VIRTUAL_MEMORY_SUPPORT__
396
397 process.StackFrame = new HAL::StackFrame();
398
399 if (!process.StackFrame) {
400 process.Crash();
401 return -kErrorProcessFault;
402 }
403
404 rt_set_memory(process.StackFrame, 0, sizeof(HAL::StackFrame));
405
406 process.StackFrame->IP = reinterpret_cast<UIntPtr>(code);
407 process.StackFrame->SP = reinterpret_cast<UIntPtr>(&process.StackReserve[0] + process.StackSize);
408
409#ifdef __NE_VIRTUAL_MEMORY_SUPPORT__
410 HAL::mm_map_page((VoidPtr) process.StackFrame->IP,
411 (VoidPtr) HAL::mm_get_page_addr((VoidPtr) process.StackFrame->IP),
412 HAL::kMMFlagsUser | HAL::kMMFlagsPresent);
413 HAL::mm_map_page((VoidPtr) process.StackFrame->SP,
414 (VoidPtr) HAL::mm_get_page_addr((VoidPtr) process.StackFrame->SP),
415 HAL::kMMFlagsUser | HAL::kMMFlagsPresent);
416#endif // ifdef __NE_VIRTUAL_MEMORY_SUPPORT__
417
418 process.StackSize = kSchedMaxStackSz;
419
420 rt_set_memory(process.StackReserve, 0, process.StackSize);
421
422 process.ParentTeam = &mTeam;
423
424 process.ProcessId = pid;
425 process.Status = ProcessStatusKind::kRunning;
426 process.PTime = 0;
427 process.RTime = 0;
428
429 if (!process.FileTree) {
430 process.FileTree = new PROCESS_FILE_TREE<VoidPtr>();
431
432 if (!process.FileTree) {
433 process.Crash();
434 return ErrorOr<VoidPtr>(-kErrorHeapOutOfMemory);
435 }
436
438 }
439
440 (Void)(kout << "PID: " << number(process.ProcessId) << kendl);
441 (Void)(kout << "Name: " << process.Name << kendl);
442
443 return pid;
444}

500SizeT UserProcessScheduler::Run() noexcept {
501 if (mTeam.mProcessCur < 1) {
502 return 0UL;
503 }
504
505 SizeT process_index = 0UL;
507
508 for (; process_index < mTeam.AsArray().Capacity(); ++process_index) {
509 auto& process = mTeam.AsArray()[process_index];
510
512 if (UserProcessHelper::CanBeScheduled(process)) {
513 kout << process.Name << " will be run...\r";
514
516 if (process.UTime < process.PTime) {
517 ++process.UTime;
518 }
519
520 this->TheCurrentProcess() = process;
521
522 if (UserProcessHelper::Switch(process.StackFrame, process.ProcessId)) {
523 process.PTime = static_cast<Int32>(process.Affinity);
524
525 // We add a bigger cooldown according to the RTime and affinity here.
526 if (process.PTime < process.RTime && AffinityKind::kRealTime != process.Affinity) {
527 if (process.RTime < (Int32) AffinityKind::kVeryHigh)
528 process.RTime += (Int32) AffinityKind::kLowUsage;
529 else if (process.RTime < (Int32) AffinityKind::kHigh)
530 process.RTime += (Int32) AffinityKind::kStandard;
531 else if (process.RTime < (Int32) AffinityKind::kStandard)
532 process.RTime += (Int32) AffinityKind::kHigh;
533
534 process.PTime -= process.RTime;
535 process.RTime = 0UL;
536 }
537 }
538 } else {
539 ++process.RTime;
540 --process.PTime;
541 }
542 }
543
544 return process_index;
545}

587Bool UserProcessHelper::CanBeScheduled(const USER_PROCESS& process) {
588 if (process.Status != ProcessStatusKind::kRunning) return No;
589 if (process.StackSize > kSchedMaxStackSz) return No;
590 if (!process.Name[0]) return No;
591
592 // real time processes shouldn't wait that much.
593 if (process.Affinity == AffinityKind::kRealTime) return Yes;
594
595 if (process.Signal.SignalID == SIGTRAP) {
596 return No;
597 }
598
599 return process.PTime < 1;
600}

620Bool UserProcessHelper::Switch(HAL::StackFramePtr frame_ptr, PID new_pid) {
621 (Void)(kout << "IP: " << hex_number(frame_ptr->IP) << kendl);
622
623 for (SizeT index = 0UL; index < HardwareThreadScheduler::The().Capacity(); ++index) {
624 if (!HardwareThreadScheduler::The()[index].Leak()) continue;
625
626 if (HardwareThreadScheduler::The()[index].Leak()->Kind() == kAPInvalid ||
627 HardwareThreadScheduler::The()[index].Leak()->Kind() == kAPBoot)
628 continue;
629
630 (Void)(kout << "AP_" << hex_number(index) << kendl);
631
632 if (HardwareThreadScheduler::The()[index].Leak()->IsBusy()) {
633 (Void)(kout << "AP_" << hex_number(index));
634 kout << " is busy\r";
635
636 continue;
637 }
638
639 (Void)(kout << "AP_" << hex_number(index));
640 kout << " is now trying to run a new task!\r";
641
645
646 HardwareThreadScheduler::The()[index].Leak()->Busy(YES);
647
648 Bool ret = HardwareThreadScheduler::The()[index].Leak()->Switch(frame_ptr);
649
653
654 if (!ret) continue;
655
656 UserProcessHelper::TheCurrentPID().Leak().Leak() = new_pid;
657
658 HardwareThreadScheduler::The()[index].Leak()->fPTime =
659 UserProcessScheduler::The().TheCurrentTeam().AsArray()[new_pid].PTime;
660
661 (Void)(kout << "AP_" << hex_number(index));
662 kout << " is now running a new task!\r";
663
664 return YES;
665 }
666
667 kout << "Couldn't find a suitable core for the current process!\r";
668
669 return NO;
670}
