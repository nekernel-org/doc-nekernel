39 auto IsBitMap(VoidPtr page_ptr) -> Bool {
40 if (!page_ptr) return No;
41
42 UIntPtr* ptr_bit_set = reinterpret_cast<UIntPtr*>(page_ptr);
43
44 if (!ptr_bit_set[kBitMapMagIdx] || ptr_bit_set[kBitMapMagIdx] != kBitMapMagic) return No;
45
46 return Yes;

49 auto FreeBitMap(VoidPtr page_ptr) -> Bool {
50 if (this->IsBitMap(page_ptr) == No) return No;
51
52 UIntPtr* ptr_bit_set = reinterpret_cast<UIntPtr*>(page_ptr);
53
54 kBitMapCursor += ptr_bit_set[kBitMapSizeIdx];
55
56 ptr_bit_set[kBitMapMagIdx] = kBitMapMagic;
57 ptr_bit_set[kBitMapUsedIdx] = No;
58
59 this->GetBitMapStatus(ptr_bit_set);
60
61 return Yes;
62 }

82 auto FindBitMap(VoidPtr base_ptr, SizeT size, Bool wr, Bool user, SizeT pad) -> VoidPtr {
83 if (!size) return nullptr;
84
85 if (kBitMapCursor > kKernelBitMpSize) {
86 err_global_get() = kErrorOutOfBitMapMemory;
87
88 (Void)(kout << "Bitmap limit reached, can't allocate more bitmaps." << kendl);
89 return nullptr;
90 }
91
92 VoidPtr base = reinterpret_cast<VoidPtr>((UIntPtr) base_ptr);
93
94 MUST_PASS(base);
95
96 STATIC SizeT biggest = 0UL;
97
98 while (YES) {
99 UIntPtr* ptr_bit_set = reinterpret_cast<UIntPtr*>(base);
100
101 if (ptr_bit_set[kBitMapMagIdx] == kBitMapMagic &&
102 ptr_bit_set[kBitMapSizeIdx] == (size + pad)) {
103 if (ptr_bit_set[kBitMapUsedIdx] == No) {
104 ptr_bit_set[kBitMapSizeIdx] = size + pad;
105 ptr_bit_set[kBitMapUsedIdx] = Yes;
106
107 this->GetBitMapStatus(ptr_bit_set);
108
109 UInt32 flags = this->MakeMMFlags(wr, user);
110 mm_map_page(ptr_bit_set, ptr_bit_set, flags);
111
112 if (biggest < (size + pad)) biggest = size + pad;
113
114 kBitMapCursor += size + pad;
115
116 return (VoidPtr) ptr_bit_set;
117 }
118 } else if (ptr_bit_set[kBitMapMagIdx] != kBitMapMagic) {
119 ptr_bit_set[kBitMapMagIdx] = kBitMapMagic;
120 ptr_bit_set[kBitMapSizeIdx] = (size + pad);
121 ptr_bit_set[kBitMapUsedIdx] = Yes;
122
123 this->GetBitMapStatus(ptr_bit_set);
124
125 UInt32 flags = this->MakeMMFlags(wr, user);
126 mm_map_page(ptr_bit_set, ptr_bit_set, flags);
127
128 if (biggest < (size + pad)) biggest = (size + pad);
129
130 kBitMapCursor += size + pad;
131
132 return (VoidPtr) ptr_bit_set;
133 }
134
135 UIntPtr raw_base = reinterpret_cast<UIntPtr>(base);
136 UIntPtr offset = (ptr_bit_set[kBitMapMagIdx] != kBitMapMagic)
137 ? (size + pad)
138 : ptr_bit_set[kBitMapSizeIdx];
139
140 base = reinterpret_cast<VoidPtr>(raw_base + offset);
141
142 if (base == nullptr) return nullptr;
143 }
144
145 return nullptr;
146 }

149 auto GetBitMapStatus(UIntPtr* ptr_bit_set) -> Void {
150 if (!this->IsBitMap(ptr_bit_set)) {
151 (Void)(kout << "Not a BitMap: " << hex_number((UIntPtr) ptr_bit_set) << kendl);
152 return;
153 }
154
155 (Void)(kout << "Magic: " << hex_number(ptr_bit_set[kBitMapMagIdx]) << kendl);
156 (Void)(kout << "Is Allocated? " << (ptr_bit_set[kBitMapUsedIdx] ? "YES" : "NO") << kendl);
157 (Void)(kout << "Size of BitMap (B): " << number(ptr_bit_set[kBitMapSizeIdx]) << kendl);
158 (Void)(kout << "Size of BitMap (KIB): " << number(KIB(ptr_bit_set[kBitMapSizeIdx]))
159 << kendl);
160 (Void)(kout << "Size of BitMap (MIB): " << number(MIB(ptr_bit_set[kBitMapSizeIdx]))
161 << kendl);
162 (Void)(kout << "Size of BitMap (GIB): " << number(GIB(ptr_bit_set[kBitMapSizeIdx]))
163 << kendl);
164 (Void)(kout << "Size of BitMap (TIB): " << number(TIB(ptr_bit_set[kBitMapSizeIdx]))
165 << kendl);
166 (Void)(kout << "BitMap Address: " << hex_number((UIntPtr) ptr_bit_set) << kendl);

182 auto mm_alloc_bitmap(Boolean wr, Boolean user, SizeT size, Bool is_page, SizeT pad) -> VoidPtr {
183 VoidPtr ptr_new = nullptr;
184 Detail::IBitMapProxy bitmp;
185
186 if (is_page) return nullptr;
187
188 ptr_new = bitmp.FindBitMap(kKernelBitMpStart, size, wr, user, pad);
189
190 if (!ptr_new) {
191 ke_panic(RUNTIME_CHECK_VIRTUAL_OUT_OF_MEM, "Out of memory bitmap");
192 return nullptr;
193 }
194
195 return ptr_new;
