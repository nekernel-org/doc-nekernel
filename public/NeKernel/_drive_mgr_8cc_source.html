35Void io_drv_input(DriveTrait::DrivePacket& pckt) {
36#ifdef __AHCI__
37 drv_std_read(pckt.fPacketLba, (Char*) pckt.fPacketContent, kAHCISectorSize, pckt.fPacketSize);
38
39 if (err_global_get() != kErrorSuccess) pckt.fPacketGood = NO;
40#elif defined(__ATA_PIO__) || defined(__ATA_DMA__)
41 drv_std_read(pckt.fPacketLba, kATAIO, kATAMaster, (Char*) pckt.fPacketContent, kATASectorSize,
42 pckt.fPacketSize);

44}

49Void io_drv_output(DriveTrait::DrivePacket& pckt) {
50 if (pckt.fPacketReadOnly) {
51 pckt.fPacketGood = NO;
52 return;
53 }
54
55 // nothing starts before 512 anyways, even an EPM partition.
56 if (!pckt.fPacketReadOnly && pckt.fPacketLba == 0) {
57 pckt.fPacketGood = NO;
58 return;
59 }
60
61#ifdef __AHCI__
62 drv_std_write(pckt.fPacketLba, (Char*) pckt.fPacketContent, kAHCISectorSize, pckt.fPacketSize);
63
64 if (err_global_get() != kErrorSuccess) pckt.fPacketGood = NO;
65#elif defined(__ATA_PIO__) || defined(__ATA_DMA__)
66 drv_std_write(pckt.fPacketLba, kATAIO, kATAMaster, (Char*) pckt.fPacketContent, kATASectorSize,
67 pckt.fPacketSize);
68#endif
69}

74Void io_drv_init(DriveTrait::DrivePacket& pckt) {
75 NE_UNUSED(pckt);
76
77#if defined(__ATA_PIO__) || defined(__ATA_DMA__)
78 kATAMaster = 0;
79 kATAIO = 0;
80
81 kATAMaster = YES;
82 kATAIO = ATA_PRIMARY_IO;
83
84 if (drv_std_init(kATAIO, kATAMaster, kATAIO, kATAMaster)) {
85 pckt.fPacketGood = YES;
86 return;
87 }
88
89 kATAMaster = NO;
90 kATAIO = ATA_SECONDARY_IO;
91
92 if (!drv_std_init(kATAIO, kATAMaster, kATAIO, kATAMaster)) {
93 pckt.fPacketGood = YES;
94 return;
95 }
96
97 pckt.fPacketGood = YES;
98#elif defined(__AHCI__)
99 kAHCIPortsImplemented = 0;
100
101 if (drv_std_init(kAHCIPortsImplemented)) {
102 pckt.fPacketGood = YES;
103 }
104#endif // if defined(__ATA_PIO__) || defined (__ATA_DMA__)
105}

156 Void io_detect_drive(DriveTrait& trait) {
157 trait.fInit(trait.fPacket);
158
159 EPM_PART_BLOCK block_struct;
160
161 trait.fPacket.fPacketLba = kEPMBootBlockLba;
162 trait.fPacket.fPacketSize = sizeof(EPM_PART_BLOCK);
163 trait.fPacket.fPacketContent = &block_struct;
164
165 rt_copy_memory((VoidPtr) "fs/detect-packet", trait.fPacket.fPacketMime,
166 rt_string_len("fs/detect-packet"));
167
168 trait.fInput(trait.fPacket);
169
170 if (rt_string_cmp(block_struct.Magic, kEPMMagic, kEPMMagicLength) == 0) {
171 trait.fPacket.fPacketReadOnly = NO;
172 trait.fKind = kMassStorageDrive | kEPMDrive;
173
174 kout << "DriveMgr: Disk is EPM formatted.\r";
175
176 trait.fSectorSz = block_struct.SectorSz;
177 trait.fLbaEnd = block_struct.LbaEnd;
178 trait.fLbaStart = block_struct.LbaStart;
179 } else {
180 GPT_PARTITION_TABLE gpt_struct;
181
182 trait.fPacket.fPacketLba = kEPMBootBlockLba;
183 trait.fPacket.fPacketSize = sizeof(GPT_PARTITION_TABLE);
184 trait.fPacket.fPacketContent = &gpt_struct;
185
186 rt_copy_memory((VoidPtr) "fs/detect-packet", trait.fPacket.fPacketMime,
187 rt_string_len("fs/detect-packet"));
188
189 trait.fInput(trait.fPacket);
190
191 if (rt_string_cmp(gpt_struct.Signature, kMagicGPT, kMagicLenGPT) == 0) {
192 trait.fPacket.fPacketReadOnly = NO;
193 trait.fKind = kMassStorageDrive | kGPTDrive;
194
195 kout << "DriveMgr: Disk is GPT formatted.\r";
196
197 trait.fSectorSz = gpt_struct.SizeOfEntries;
198 trait.fLbaEnd = gpt_struct.LastGPTEntry;
199 trait.fLbaStart = gpt_struct.FirstGPTEntry;
200 } else {
201 kout << "DriveMgr: Disk is unformatted.\r";
202
203 trait.fPacket.fPacketReadOnly = YES;
204 trait.fKind = kMassStorageDrive | kUnformattedDrive | kReadOnlyDrive;
205
206 trait.fSectorSz = 512;
207 trait.fLbaEnd = drv_std_get_sector_count() - 1;
208 trait.fLbaStart = 0x400;
209 }
210 }
211
212 rt_copy_memory((VoidPtr) "*/*", trait.fPacket.fPacketMime, rt_string_len("*/*"));
213
214 trait.fPacket.fPacketLba = 0;
215 trait.fPacket.fPacketSize = 0UL;
216 trait.fPacket.fPacketContent = nullptr;
217 }
218} // namespace Detail

222DriveTrait io_construct_main_drive() noexcept {
223 DriveTrait trait;
224
225 constexpr auto kMainDrive = "/media/main/";
226
227 rt_copy_memory((VoidPtr) kMainDrive, trait.fName, rt_string_len(kMainDrive));
228
229 MUST_PASS(trait.fName[0] != 0);
230
231 trait.fVerify = io_drv_unimplemented;
232 trait.fOutput = io_drv_output;
233 trait.fInput = io_drv_input;
234 trait.fInit = io_drv_init;
235 trait.fProtocol = io_drv_kind;
236
237 kout << "DriveMgr: Detecting partition scheme of: " << trait.fName << ".\r";
238
239 Detail::io_detect_drive(trait);
240
241 return trait;
242}
