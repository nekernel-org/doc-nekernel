146Void mp_init_cores(VoidPtr vendor_ptr) noexcept {
147 if (!vendor_ptr) return;
148
149 if (!kHandoverHeader->f_HardwareTables.f_MultiProcessingEnabled) {
150 kSMPAware = NO;
151 return;
152 }
153
154 PowerFactoryInterface hw_and_pow_int{vendor_ptr};
155
156 auto pwr = hw_and_pow_int.Find(APIC_MAG);
157
158 if (pwr.HasError()) {
159 kSMPAware = NO;
160 return;
161 }
162
163 kRawMADT = pwr.Leak().Leak();
164 kSMPBlock = reinterpret_cast<HAL_APIC_MADT*>(kRawMADT);
165 kSMPAware = NO;
166
167 if (kSMPBlock) {
168 kSMPInterrupt = 0;
169 kSMPCount = 0;
170
171 UInt32 lo = 0U, hi = 0U;
172
173 hal_get_msr(APIC_BASE_MSR, &lo, &hi);
174
175 UInt64 apic_base = ((UInt64) hi << 32) | lo;
176
177 apic_base |= APIC_BASE_MSR_ENABLE; // Enable APIC.
178
179 lo = apic_base & 0xFFFFFFFF;
180 hi = apic_base >> 32;
181
182 hal_set_msr(APIC_BASE_MSR, lo, hi);
183
184 kApicBaseAddress = apic_base & 0xFFFFF000;
185
186 LAPICDmaWrapper controller{(VoidPtr) kApicBaseAddress};
187
188 controller.Write(LAPIC_REG_ENABLE, 0);
189 controller.Write(LAPIC_REG_SPURIOUS, 0x1FF); // Enable bit, spurious interrupt vector register.
190 controller.Write(LAPIC_REG_TIMER_DIV, 0b0011);
191 controller.Write(LAPIC_REG_TIMER_LVT, 0x20 | (1 << 17));
192 controller.Write(LAPIC_REG_TIMER_INITCNT, 1000000);
193
194 volatile UInt8* entry_ptr = reinterpret_cast<volatile UInt8*>(kSMPBlock->List);
195 volatile UInt8* end_ptr = ((UInt8*) kSMPBlock) + kSMPBlock->Length;
196
197 while (entry_ptr < end_ptr) {
198 UInt8 type = *entry_ptr;
199 UInt8 length = *(entry_ptr + 1);
200
201 // Avoid infinite loop on bad APIC tables.
202 if (length < 2) break;
203
204 if (type == 0) {
205 volatile LAPIC* entry_struct = (volatile LAPIC*) entry_ptr;
206
207 if (entry_struct->Flags & 0x1) {
208 kAPICLocales[kSMPCount] = entry_struct->ProcessorID;
209 kHWThread[kSMPCount].mThreadID = kAPICLocales[kSMPCount];
210
211 ++kSMPCount;
212
213 kout << "Kind: LAPIC: ON\r";
214
215 // 0x7c00, as recommended by the Intel SDM.
216 hal_send_ipi_msg(kApicBaseAddress, entry_struct->ProcessorID, 0x7c);
217 } else {
218 kout << "Kind: LAPIC: OFF\r";
219 }
220 } else {
221 kout << "Kind: UNKNOWN: OFF\r";
222 }
223
224 entry_ptr += length;
225 }
226
227 kSMPAware = kSMPCount > 1;
228 }
229}
