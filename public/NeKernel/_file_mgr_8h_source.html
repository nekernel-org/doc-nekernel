232class FileStream final {
233 public:
234 explicit FileStream(const Encoding* path, const Encoding* restrict_type);
235 ~FileStream();

238 FileStream& operator=(const FileStream&);

241 public:
242 ErrorOr<Int64> Write(SizeT offset, const VoidPtr data, SizeT len) noexcept {
243 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
244 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
245 this->fFileRestrict != kFileMgrRestrictWrite &&
246 this->fFileRestrict != kFileMgrRestrictWriteBinary)
247 return ErrorOr<Int64>(kErrorInvalidData);
248
249 if (data == nullptr) return ErrorOr<Int64>(kErrorInvalidData);
250
251 auto man = FSClass::GetMounted();
252
253 if (man) {
254 man->Write(offset, fFile, data, len);
255 return ErrorOr<Int64>(0);
256 }
257
258 return ErrorOr<Int64>(kErrorInvalidData);
259 }
260
261 ErrorOr<Int64> Write(const Char* name, const VoidPtr data, SizeT len) noexcept {
262 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
263 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
264 this->fFileRestrict != kFileMgrRestrictWrite &&
265 this->fFileRestrict != kFileMgrRestrictWriteBinary)
266 return ErrorOr<Int64>(kErrorInvalidData);
267
268 if (data == nullptr) return ErrorOr<Int64>(kErrorInvalidData);
269
270 auto man = FSClass::GetMounted();
271
272 if (man) {
273 man->Write(name, fFile, data, 0, len);
274 return ErrorOr<Int64>(0);
275 }
276
277 return ErrorOr<Int64>(kErrorInvalidData);
278 }
279
280 VoidPtr Read(const Char* name, SizeT sz) noexcept {
281 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
282 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
283 this->fFileRestrict != kFileMgrRestrictRead &&
284 this->fFileRestrict != kFileMgrRestrictReadBinary)
285 return nullptr;
286
287 auto man = FSClass::GetMounted();
288
289 if (man) {
290 VoidPtr ret = man->Read(name, fFile, kFileReadAll, sz);
291 return ret;
292 }
293
294 return nullptr;
295 }
296
297 VoidPtr Read(SizeT offset, SizeT sz) {
298 if (this->fFileRestrict != kFileMgrRestrictReadWrite &&
299 this->fFileRestrict != kFileMgrRestrictReadWriteBinary &&
300 this->fFileRestrict != kFileMgrRestrictRead &&
301 this->fFileRestrict != kFileMgrRestrictReadBinary)
302 return nullptr;
303
304 auto man = FSClass::GetMounted();
305
306 if (man) {
307 man->Seek(fFile, offset);
308 auto ret = man->Read(fFile, kFileReadChunk, sz);
309
310 return ret;
311 }
312
313 return nullptr;
314 }

355template <typename Encoding, typename Class>
356inline FileStream<Encoding, Class>::FileStream(const Encoding* path, const Encoding* restrict_type)
357 : fFile(Class::GetMounted()->Open(path, restrict_type)) {
358 SizeT kRestrictCount = kRestrictMax;
359 const FILEMGR_RESTRICT kRestrictList[] = {{
360 .fRestrict = kRestrictR,
361 .fMappedTo = kFileMgrRestrictRead,
362 },
363 {
364 .fRestrict = kRestrictRB,
365 .fMappedTo = kFileMgrRestrictReadBinary,
366 },
367 {
368 .fRestrict = kRestrictWRB,
369 .fMappedTo = kFileMgrRestrictReadWriteBinary,
370 },
371 {
372 .fRestrict = kRestrictW,
373 .fMappedTo = kFileMgrRestrictWrite,
374 },
375 {
376 .fRestrict = kRestrictWR,
377 .fMappedTo = kFileMgrRestrictReadWrite,
378 }};
379
380 for (SizeT index = 0; index < kRestrictCount; ++index) {
381 if (rt_string_cmp(restrict_type, kRestrictList[index].fRestrict,
382 rt_string_len(kRestrictList[index].fRestrict)) == 0) {
383 fFileRestrict = kRestrictList[index].fMappedTo;
384 break;
385 }
386 }
387
388 kout << "FileMgr: Open file at: " << path << ".\r";
389}
