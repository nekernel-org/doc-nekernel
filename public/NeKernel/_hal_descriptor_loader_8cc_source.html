26Void IDTLoader::Load(Register64& idt) {
27#ifndef __NE_MODULAR_KERNEL_COMPONENTS__
28 rt_cli();
29
30 volatile UIntPtr** ptr_ivt = (volatile UIntPtr**) idt.Base;
31
32 for (SizeT idt_indx = 0; idt_indx < kKernelIdtSize; ++idt_indx) {
33 Detail::kInterruptVectorTable[idt_indx].Selector = kIDTSelector;
34 Detail::kInterruptVectorTable[idt_indx].Ist = 0;
35 Detail::kInterruptVectorTable[idt_indx].TypeAttributes =
36 kKernelInterruptId ? kUserInterruptGate : kInterruptGate;
37 Detail::kInterruptVectorTable[idt_indx].OffsetLow = ((UIntPtr) ptr_ivt[idt_indx] & 0xFFFF);
38 Detail::kInterruptVectorTable[idt_indx].OffsetMid =
39 (((UIntPtr) ptr_ivt[idt_indx] >> 16) & 0xFFFF);
40 Detail::kInterruptVectorTable[idt_indx].OffsetHigh =
41 (((UIntPtr) ptr_ivt[idt_indx] >> 32) & 0xFFFFFFFF);
42
43 Detail::kInterruptVectorTable[idt_indx].Zero = 0;
44 }
45
46 idt.Base = (UIntPtr) &Detail::kInterruptVectorTable[0];
47 idt.Limit = sizeof(::Kernel::Detail::AMD64::InterruptDescriptorAMD64) * (kKernelIdtSize);
48
49 hal_load_idt(idt);
50 rt_sti();
51#endif // __NE_MODULAR_KERNEL_COMPONENTS__
52
53 return;
54}
