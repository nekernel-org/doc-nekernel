133EXTERN_C Kernel::Void hal_system_call_enter(Kernel::UIntPtr rcx_hash,
134 Kernel::UIntPtr rdx_syscall_arg) {
135 hal_idt_send_eoi(50);
136
137 if (!Kernel::kCurrentUser) return;
138
139 for (SizeT i = 0UL; i < kMaxDispatchCallCount; ++i) {
140 if (kSysCalls[i].fHooked && rcx_hash == kSysCalls[i].fHash) {
141 if (kSysCalls[i].fProc) {
142 (kSysCalls[i].fProc)((Kernel::VoidPtr) rdx_syscall_arg);

152 Kernel::UIntPtr arg, Kernel::SizeT sz) {
153 hal_idt_send_eoi(51);
154
155 if (!Kernel::kRootUser) return;
156 if (Kernel::kCurrentUser != Kernel::kRootUser) return;
157 if (!Kernel::kCurrentUser->IsSuperUser()) return;
158
159 for (SizeT i = 0UL; i < kMaxDispatchCallCount; ++i) {
160 if (kKernCalls[i].fHooked && rcx_hash == kKernCalls[rcx_hash].fHash) {
