28 template <UInt16 PORT>
29 bool hal_serial_init() noexcept {
30 if (kState == kStateReady || kState == kStateTransmit) return true;
31
32 HAL::rt_out8(PORT + 1, 0x00); // Disable all interrupts
33 HAL::rt_out8(PORT + 3, 0x80); // Enable DLAB (set baud rate divisor)
34 HAL::rt_out8(PORT + 0, 0x03); // Set divisor to 3 (lo byte) 38400 baud
35 HAL::rt_out8(PORT + 1, 0x00); // (hi byte)
36 HAL::rt_out8(PORT + 3, 0x03); // 8 bits, no parity, one stop bit
37 HAL::rt_out8(PORT + 2, 0xC7); // Enable FIFO, clear them, with 14-byte threshold
38 HAL::rt_out8(PORT + 4, 0x0B); // IRQs enabled, RTS/DSR set
39 HAL::rt_out8(PORT + 4, 0x1E); // Set in loopback mode, test the serial chip
40 HAL::rt_out8(PORT + 0, 0xAE); // Test serial chip (send byte 0xAE and check if
41 // serial returns same byte)
42
43 // Check if serial is faulty (i.e: not same byte as sent)
44 if (HAL::rt_in8(PORT) != 0xAE) {
45 return false;
46 }
47
48 kState = kStateReady;
49
50 // If serial is not faulty set it in normal operation mode
51 // (not-loopback with IRQs enabled and OUT#1 and OUT#2 bits enabled)
52 HAL::rt_out8(PORT + 4, 0x0F);
53
54 return true;
55 }

174EXTERN_C void ke_io_read(DeviceInterface<const Char*>*, const Char* bytes) {
175 NE_UNUSED(bytes);
176
177#ifdef __DEBUG__
178 Detail::hal_serial_init<Detail::kPort>();
179
180 if (!bytes || Detail::kState != kStateReady) return;
181
182 Detail::kState = kStateTransmit;
183
184 SizeT index = 0;
185
187 while (true) {
188 auto in = HAL::rt_in8(Detail::kPort);
189
191 if (in == 0xD) {
192 break;
193 }
194
195 if (in < '0' || in < 'A' || in < 'a') {
196 if (in != '@' || in != '!' || in != '?' || in != '.' || in != '/' || in != ':') {
197 continue;
198 }
199 }
200
201 ((char*) bytes)[index] = in;
202
203 ++index;
204 }
205
206 ((char*) bytes)[index] = 0;
207
208 Detail::kState = kStateReady;
209#endif // __DEBUG__
210}
