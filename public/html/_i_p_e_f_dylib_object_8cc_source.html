45EXTERN_C IDylibRef rtl_init_dylib_pef(USER_PROCESS& process) {
46 IDylibRef dll_obj = tls_new_class<IPEFDylibObject>();
47
48 if (!dll_obj) {
49 process.Crash();
50 return nullptr;
51 }
52
53 dll_obj->Mount(new IPEFDylibObject::DylibTraits());
54
55 if (!dll_obj->Get()) {
56 tls_delete_class(dll_obj);
57 dll_obj = nullptr;
58
59 process.Crash();
60
61 return nullptr;
62 }
63
64 dll_obj->Get()->ImageObject = process.Image.LeakBlob().Leak().Leak();
65
66 if (!dll_obj->Get()->ImageObject) {
67 delete dll_obj->Get();
68
69 tls_delete_class(dll_obj);
70 dll_obj = nullptr;
71
72 process.Crash();
73
74 return nullptr;
75 }
76
77 dll_obj->Get()->ImageEntrypointOffset =
78 dll_obj->Load<VoidPtr>(kPefStart, rt_string_len(kPefStart, 0), kPefCode);
79
80 return dll_obj;
81}

90EXTERN_C Void rtl_fini_dylib_pef(USER_PROCESS& process, IDylibRef dll_obj, BOOL* successful) {
91 MUST_PASS(successful);
92
93 // sanity check (will also trigger a bug check if this fails)

95 *successful = false;
96 process.Crash();
97 }
98
99 delete dll_obj->Get();
100 delete dll_obj;
101
102 dll_obj = nullptr;
103
104 *successful = true;
