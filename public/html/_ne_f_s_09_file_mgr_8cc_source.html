165Void NeFileSystemMgr::Write(_Input const Char* name, _Input NodePtr node, _Input VoidPtr data,
166 _Input Int32 flags, _Input SizeT size) {
167 if (!is_valid_nefs_catalog(node)) {
168 kout << "NeFS: Write(fork) called with invalid node pointer\n";
169 return;
170 }
171 if (!name || *name == 0) {
172 kout << "NeFS: Write(fork) called with null or empty fork name\n";
173 return;
174 }
175 if (!data) {
176 kout << "NeFS: Write(fork) called with null data pointer\n";
177 return;
178 }
179 if (!size || size > kNeFSForkSize) {
180 (Void)(kout << "NeFS: Write called with invalid size: " << hex_number(size));
181 kout << "\n";
182 return;
183 }
184 NE_UNUSED(flags);
185 auto cat = reinterpret_cast<NEFS_CATALOG_STRUCT*>(node);
186 if (cat->Kind == kNeFSCatalogKindFile) {
187 mParser->WriteCatalog(cat->Name, (flags & kFileFlagRsrc ? true : false), data, size, name);
188 }
189}

250static inline bool is_valid_nefs_catalog(NodePtr node) {
251 if (!node) return false;
252 auto cat = reinterpret_cast<NEFS_CATALOG_STRUCT*>(node);
253 switch (cat->Kind) {
254 case kNeFSCatalogKindFile:
255 case kNeFSCatalogKindDir:
256 case kNeFSCatalogKindAlias:
257 case kNeFSCatalogKindPage:
258 break;
259 default:
260 return false;
261 }
262 bool null_found = false;
263 for (int i = 0; i < kNeFSCatalogNameLen; ++i) {
264 if (cat->Name[i] == 0) {
265 null_found = true;
266 break;
267 }
268 }
269 if (!null_found) return false;
270 return true;
271}
