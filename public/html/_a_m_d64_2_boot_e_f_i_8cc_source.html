53EFI_EXTERN_C EFI_API Int32 BootloaderMain(EfiHandlePtr image_handle, EfiSystemTable* sys_table) {
54 fw_init_efi(sys_table);
55
56 ST->ConOut->ClearScreen(sys_table->ConOut);
57 ST->ConOut->SetAttribute(sys_table->ConOut, kEFIYellow);
58
59 ST->BootServices->SetWatchdogTimer(0, 0, 0, nullptr);
60 ST->ConOut->EnableCursor(ST->ConOut, false);
61
62 HEL::BootInfoHeader* handover_hdr = new HEL::BootInfoHeader();
63
64 UInt32 map_key = 0;
65 UInt32 size_struct_ptr = sizeof(EfiMemoryDescriptor);
66 EfiMemoryDescriptor* struct_ptr = nullptr;
67 UInt32 sz_desc = sizeof(EfiMemoryDescriptor);
68 UInt32 rev_desc = 0;
69
70 Boot::BootTextWriter writer;
71
72 if (!boot_init_fb()) {
73 writer.Write("BootZ: Invalid Framebuffer, can't boot to NeKernel.\r");
74 Boot::Stop();
75 }
76
77 for (SizeT index_vt = 0; index_vt < sys_table->NumberOfTableEntries; ++index_vt) {
78 Char* vendor_table =
79 reinterpret_cast<Char*>(sys_table->ConfigurationTable[index_vt].VendorTable);
80
81 // ACPI's 'RSD PTR', which contains the ACPI SDT (MADT, FACP...)
82 if (vendor_table[0] == 'R' && vendor_table[1] == 'S' && vendor_table[2] == 'D' &&
83 vendor_table[3] == ' ' && vendor_table[4] == 'P' && vendor_table[5] == 'T' &&
84 vendor_table[6] == 'R' && vendor_table[7] == ' ') {
85 handover_hdr->f_HardwareTables.f_VendorPtr = (VoidPtr) vendor_table;
86 break;
87 }
88 }
89
90 // ------------------------------------------ //
91 // draw background color.
92 // ------------------------------------------ //
93
94 handover_hdr->f_GOP.f_The = kGop->Mode->FrameBufferBase;
95 handover_hdr->f_GOP.f_Width = kGop->Mode->Info->VerticalResolution;
96 handover_hdr->f_GOP.f_Height = kGop->Mode->Info->HorizontalResolution;
97 handover_hdr->f_GOP.f_PixelPerLine = kGop->Mode->Info->PixelsPerScanLine;
98 handover_hdr->f_GOP.f_PixelFormat = kGop->Mode->Info->PixelFormat;
99 handover_hdr->f_GOP.f_Size = kGop->Mode->FrameBufferSize;
100
101 // ------------------------------------------- //
102 // Grab MP services, extended to runtime. //
103 // ------------------------------------------- //
104
105 EFI_GUID guid_mp = EFI_GUID(EFI_MP_SERVICES_PROTOCOL_GUID);
106 EfiMpServicesProtocol* mp = nullptr;
107
108 BS->LocateProtocol(&guid_mp, nullptr, reinterpret_cast<VoidPtr*>(&mp));
109
110 handover_hdr->f_HardwareTables.f_MpPtr = reinterpret_cast<VoidPtr>(mp);
111
112 kHandoverHeader = handover_hdr;
113
114 FB::fb_clear_video();
115
116 UInt32 cnt_enabled = 0;
117 UInt32 cnt_disabled = 0;
118
119 if (mp) {
120 mp->GetNumberOfProcessors(mp, &cnt_disabled, &cnt_enabled);
121 handover_hdr->f_HardwareTables.f_MultiProcessingEnabled = cnt_enabled > 1;
122 } else {
123 handover_hdr->f_HardwareTables.f_MultiProcessingEnabled = NO;
124 }
125
126 // Fill handover header now.
127
128 handover_hdr->f_BitMapStart = nullptr; /* Start of bitmap. */
129 handover_hdr->f_BitMapSize = kHandoverBitMapSz; /* Size of bitmap in bytes. */
130
131 kHandoverHeader->f_BitMapStart = nullptr; /* Start of bitmap. */
132 kHandoverHeader->f_BitMapSize = kHandoverBitMapSz; /* Size of bitmap in bytes. */
133
134 UInt16 trials = 5;
135
136 while (BS->AllocatePool(EfiLoaderData, kHandoverHeader->f_BitMapSize,
137 &kHandoverHeader->f_BitMapStart) != kEfiOk) {
138 --trials;
139
140 if (!trials) {
141 writer.Write("BootZ: Unable to allocate sufficient memory, trying again with 2GB...\r");
142
143 trials = 3;
144
145 kHandoverHeader->f_BitMapSize = kHandoverBitMapSz / 2; /* Size of bitmap in bytes. */
146
147 while (BS->AllocatePool(EfiLoaderData, kHandoverHeader->f_BitMapSize,
148 &kHandoverHeader->f_BitMapStart) != kEfiOk) {
149 --trials;
150
151 if (!trials) {
152 writer.Write("BootZ: Unable to allocate sufficient memory, aborting...\r");
153 Boot::Stop();
154 }
155 }
156 }
157 }
158
159 handover_hdr->f_FirmwareCustomTables[Kernel::HEL::kHandoverTableBS] = (VoidPtr) BS;
160 handover_hdr->f_FirmwareCustomTables[Kernel::HEL::kHandoverTableST] = (VoidPtr) ST;
161
162 // ------------------------------------------ //
163 // If we succeed in reading the blob, then execute it.
164 // ------------------------------------------ //
165
166 Boot::BootFileReader reader_syschk(L"chk.efi", image_handle);
167 reader_syschk.ReadAll(0);
168
169 Boot::BootThread* syschk_thread = nullptr;
170
171 if (reader_syschk.Blob()) {
172 syschk_thread = new Boot::BootThread(reader_syschk.Blob());
173 syschk_thread->SetName("SysChk");
174
175 syschk_thread->Start(handover_hdr, NO);
176 }
177
178 BS->GetMemoryMap(&size_struct_ptr, struct_ptr, &map_key, &sz_desc, &rev_desc);
179
180 handover_hdr->f_FirmwareVendorLen = Boot::BStrLen(sys_table->FirmwareVendor);
181
182 handover_hdr->f_Magic = kHandoverMagic;
183 handover_hdr->f_Version = kHandoverVersion;
184
185 handover_hdr->f_HardwareTables.f_ImageKey = map_key;
186 handover_hdr->f_HardwareTables.f_ImageHandle = image_handle;
187
188 // Provide fimware vendor name.
189
190 Boot::BCopyMem(handover_hdr->f_FirmwareVendorName, sys_table->FirmwareVendor,
191 handover_hdr->f_FirmwareVendorLen);
192
193 handover_hdr->f_FirmwareVendorLen = Boot::BStrLen(sys_table->FirmwareVendor);
194 // Assign to global 'kHandoverHeader'.
195
196 WideChar kernel_path[256U] = L"ne_kernel";
197 UInt32 kernel_path_sz = StrLen("ne_kernel");
198
199 UInt32 sz_ver = sizeof(UInt64);
200 UInt64 ver = KERNEL_VERSION_BCD;
201
202 ST->RuntimeServices->GetVariable(L"/props/kern_ver", kEfiGlobalNamespaceVarGUID, nullptr, &sz_ver,
203 &ver);
204
205 if (ver < KERNEL_VERSION_BCD) {
206 ver = KERNEL_VERSION_BCD;
207
208 ST->RuntimeServices->SetVariable(L"/props/kern_ver", kEfiGlobalNamespaceVarGUID, nullptr,
209 &sz_ver, &ver);
210
211 writer.Write("BootZ: Version has been updated: ").Write(ver).Write("\r");
212
213 if (ST->RuntimeServices->GetVariable(L"/props/kernel_path", kEfiGlobalNamespaceVarGUID, nullptr,
214 &kernel_path_sz, kernel_path) != kEfiOk) {
217 UInt32 attr = 0x00000001 | 0x00000002 | 0x00000004;
218
219 ST->RuntimeServices->SetVariable(L"/props/kernel_path", kEfiGlobalNamespaceVarGUID, &attr,
220 &kernel_path_sz, kernel_path);
221 }
222 } else {
223 writer.Write("BootZ: Version: ").Write(ver).Write("\r");
224 }
225
226 // boot to kernel, if not bootnet this.
227
228 Boot::BootFileReader reader_kernel(kernel_path, image_handle);
229 reader_kernel.ReadAll(0);
230
231 // ------------------------------------------ //
232 // If we succeed in reading the blob, then execute it.
233 // ------------------------------------------ //
234
235 if (reader_kernel.Blob()) {
236 handover_hdr->f_PageStart = boot_read_cr3();
237
238 auto kernel_thread = Boot::BootThread(reader_kernel.Blob());
239
240 kernel_thread.SetName("NeKernel");
241
242 handover_hdr->f_KernelImage = reader_kernel.Blob();
243 handover_hdr->f_KernelSz = reader_kernel.Size();
244
245 kernel_thread.Start(handover_hdr, YES);
246 }
247
248 Boot::BootFileReader reader_netboot(L"net.efi", image_handle);
249 reader_netboot.ReadAll(0);
250
251 if (!reader_netboot.Blob()) return kEfiFail;
252
253 auto netboot_thread = Boot::BootThread(reader_netboot.Blob());
254 netboot_thread.SetName("BootNet");
255
256 return netboot_thread.Start(handover_hdr, NO);
257}
