16ErrorOr<voidPtr> ACPIFactoryInterface::Find(const Char* signature) {
17 MUST_PASS(this->fRsdp);
18
19 if (!signature) return ErrorOr<voidPtr>{-kErrorInvalidData};
20 if (*signature == 0) return ErrorOr<voidPtr>{-kErrorInvalidData};
21
22 RSDP* rsp_ptr = reinterpret_cast<RSDP*>(this->fRsdp);
23
24 if (rsp_ptr->Revision < kMinACPIVer) return ErrorOr<voidPtr>{-kErrorInvalidData};
25
26 RSDT* xsdt = reinterpret_cast<RSDT*>(rsp_ptr->RsdtAddress);
27
28 Int64 num = (xsdt->Length - sizeof(SDT)) / sizeof(Int64);
29
30 /***
31 crucial to avoid underflows.
32 */
33 if (num < 1) {
35 ke_panic(RUNTIME_CHECK_ACPI);
36 return ErrorOr<voidPtr>{-1};
37 }
38
39 this->fEntries = num;
40
41 (Void)(kout << "ACPI: Number of entries: " << number(this->fEntries) << kendl);
42 (Void)(kout << "ACPI: Revision: " << number(xsdt->Revision) << kendl);
43 (Void)(kout << "ACPI: Signature: " << xsdt->Signature << kendl);
44 (Void)(kout << "ACPI: Address of XSDT: " << hex_number((UIntPtr) xsdt) << kendl);
45
46 const short cAcpiSignatureLength = 4;
47
48 for (Size index = 0; index < this->fEntries; ++index) {
49 SDT* sdt = reinterpret_cast<SDT*>(xsdt->AddressArr[index]);
50
51 (Void)(kout << "ACPI: Checksum: " << number(sdt->Checksum) << kendl);
52 (Void)(kout << "ACPI: Revision: " << number(sdt->Revision) << kendl);
53
54 for (short signature_index = 0; signature_index < cAcpiSignatureLength; ++signature_index) {
55 if (sdt->Signature[signature_index] != signature[signature_index]) break;
56
57 if (signature_index == (cAcpiSignatureLength - 1)) {
58 (Void)(kout << "ACPI: SDT Signature: " << sdt->Signature << kendl);
59 (Void)(kout << "ACPI: SDT OEM ID: " << sdt->OemId << kendl);
60 return ErrorOr<voidPtr>(reinterpret_cast<voidPtr>(xsdt->AddressArr[index]));
61 }
62 }
63 }
64
65 return ErrorOr<voidPtr>{-1};
66}

68/***

73bool ACPIFactoryInterface::Checksum(const Char* checksum, SSizeT len) {
74 if (len == 0) return 1;
75
76 char chr = 0;
77
78 for (int index = 0; index < len; ++index) {
79 chr += checksum[index];
80 }
81
82 return chr == 0;
83}
