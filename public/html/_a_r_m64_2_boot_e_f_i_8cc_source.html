41STATIC Bool boot_init_fb() noexcept {
42 kGopGuid = EFI_GUID(EFI_GRAPHICS_OUTPUT_PROTOCOL_GUID);
43 kGop = nullptr;
44
45 if (BS->LocateProtocol(&kGopGuid, nullptr, (VoidPtr*) &kGop) != kEfiOk) return No;
46
47 kGopStride = 4;
48
49 for (SizeT i = 0; i < kGop->Mode->MaxMode; ++i) {
50 EfiGraphicsOutputProtocolModeInformation* infoPtr = nullptr;
51 UInt32 sz = 0U;
52
53 kGop->QueryMode(kGop, i, &sz, &infoPtr);
54
55 if (infoPtr->HorizontalResolution == kExpectedWidth &&
56 infoPtr->VerticalResolution == kExpectedHeight) {
57 kGop->SetMode(kGop, i);
58 return Yes;
59 }
60 }
61
62 return No;
63}

71EFI_EXTERN_C EFI_API Int32 BootloaderMain(EfiHandlePtr image_handle, EfiSystemTable* sys_table) {
72 fw_init_efi(sys_table);
73
74 kHandoverHeader = new HEL::BootInfoHeader();
75
76#ifdef ZBA_USE_FB
77 if (!boot_init_fb()) return kEfiFail;
78
79 for (SizeT index_vt = 0; index_vt < sys_table->NumberOfTableEntries; ++index_vt) {
80 Char* vendor_table =
81 reinterpret_cast<Char*>(sys_table->ConfigurationTable[index_vt].VendorTable);
82
83 // ACPI's 'RSD PTR', which contains the ACPI SDT (MADT, FACP...)
84 if (vendor_table[0] == 'R' && vendor_table[1] == 'S' && vendor_table[2] == 'D' &&
85 vendor_table[3] == ' ' && vendor_table[4] == 'P' && vendor_table[5] == 'T' &&
86 vendor_table[6] == 'R' && vendor_table[7] == ' ') {
87 kHandoverHeader->f_HardwareTables.f_VendorPtr = (VoidPtr) vendor_table;
88 break;
89 }
90 }
91
92 // ------------------------------------------ //
93 // draw background color.
94 // ------------------------------------------ //
95
96 kHandoverHeader->f_GOP.f_The = kGop->Mode->FrameBufferBase;
97 kHandoverHeader->f_GOP.f_Width = kGop->Mode->Info->VerticalResolution;
98 kHandoverHeader->f_GOP.f_Height = kGop->Mode->Info->HorizontalResolution;
99 kHandoverHeader->f_GOP.f_PixelPerLine = kGop->Mode->Info->PixelsPerScanLine;
100 kHandoverHeader->f_GOP.f_PixelFormat = kGop->Mode->Info->PixelFormat;
101 kHandoverHeader->f_GOP.f_Size = kGop->Mode->FrameBufferSize;
102#endif // ZBA_USE_FB
103
104 // ------------------------------------------- //
105 // Grab MP services, extended to runtime. //
106 // ------------------------------------------- //
107
108 EFI_GUID guid_mp = EFI_GUID(EFI_MP_SERVICES_PROTOCOL_GUID);
109 EfiMpServicesProtocol* mp = nullptr;
110
111 BS->LocateProtocol(&guid_mp, nullptr, reinterpret_cast<VoidPtr*>(&mp));
112 kHandoverHeader->f_HardwareTables.f_MpPtr = reinterpret_cast<VoidPtr>(mp);
113
114 UInt32 cnt_enabled = 0;
115 UInt32 cnt_disabled = 0;
116
117 if (mp) {
118 mp->GetNumberOfProcessors(mp, &cnt_disabled, &cnt_enabled);
119 kHandoverHeader->f_HardwareTables.f_MultiProcessingEnabled = cnt_enabled > 1;
120 } else {
121 kHandoverHeader->f_HardwareTables.f_MultiProcessingEnabled = NO;
122 }
123
124 //-------------------------------------------------------------//
125 // Allocate heap.
126 //-------------------------------------------------------------//
127
128 Boot::BootTextWriter writer;
129
130 kHandoverHeader->f_BitMapStart = nullptr; /* Start of bitmap. */
131 kHandoverHeader->f_BitMapSize = kHandoverBitMapSz; /* Size of bitmap in bytes. */
132
133 UInt16 trials = 5;
134
135 while (BS->AllocatePool(EfiLoaderData, kHandoverHeader->f_BitMapSize,
136 &kHandoverHeader->f_BitMapStart) != kEfiOk) {
137 --trials;
138
139 if (!trials) {
140 writer.Write("BootZ: Unable to allocate sufficient memory, trying again with 2GB...\r");
141
142 trials = 3;
143
144 kHandoverHeader->f_BitMapSize = kHandoverBitMapSz / 2; /* Size of bitmap in bytes. */
145
146 while (BS->AllocatePool(EfiLoaderData, kHandoverHeader->f_BitMapSize,
147 &kHandoverHeader->f_BitMapStart) != kEfiOk) {
148 --trials;
149
150 if (!trials) {
151 writer.Write("BootZ: Unable to allocate sufficient memory, aborting...\r");
152 Boot::Stop();
153 }
154 }
155 }
156 }
157
158 // ------------------------------------------ //
159 // null these fields, to avoid being reused later.
160 // ------------------------------------------ //
161
162 kHandoverHeader->f_FirmwareCustomTables[0] = nullptr;
163 kHandoverHeader->f_FirmwareCustomTables[1] = nullptr;
164
165 kHandoverHeader->f_FirmwareVendorLen = Boot::BStrLen(sys_table->FirmwareVendor);
166
167 kHandoverHeader->f_Magic = kHandoverMagic;
168 kHandoverHeader->f_Version = kHandoverVersion;
169
170 // Provide fimware vendor name.
171
172 Boot::BCopyMem(kHandoverHeader->f_FirmwareVendorName, sys_table->FirmwareVendor,
173 kHandoverHeader->f_FirmwareVendorLen);
174
175 kHandoverHeader->f_FirmwareVendorLen = Boot::BStrLen(sys_table->FirmwareVendor);
176
177 Boot::BootFileReader reader_kernel(L"ne_kernel", image_handle);
178
179 reader_kernel.ReadAll(0);
180
181 // ------------------------------------------ //
182 // If we succeed in reading the blob, then execute it.
183 // ------------------------------------------ //
184
185 if (reader_kernel.Blob()) {
186 auto kernel_thread = Boot::BootThread(reader_kernel.Blob());
187 kernel_thread.SetName("NeKernel");
188
189 kHandoverHeader->f_KernelImage = reader_kernel.Blob();
190 kHandoverHeader->f_KernelSz = reader_kernel.Size();
191
192 kernel_thread.Start(kHandoverHeader, YES);
193 }
194
195 CANT_REACH();
196}
