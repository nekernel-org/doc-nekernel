55static inline bool is_valid_size(SizeT size, SizeT max_size) {
56 return size > 0 && size <= max_size;
57}

59static inline bool is_valid_lba(Lba lba, DriveTrait& drive) {
60 NEFS_ROOT_PARTITION_BLOCK part_block;
61 drive.fPacket.fPacketLba = kNeFSRootCatalogStartAddress;
62 drive.fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
63 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&part_block);
64 drive.fInput(drive.fPacket);
65
66 if (!drive.fPacket.fPacketGood) {
67 return false;
68 }
69
70 // Compute the maximum LBA (DiskSize / sector size)
71 SizeT sectorSize = drive.fSectorSz;
72 Lba maxLba = part_block.DiskSize / sectorSize;
73
74 return (lba >= part_block.StartCatalog) && (lba < maxLba);
75}

78/***********************************************************************************/
83/***********************************************************************************/
84_Output BOOL NeFileSystemParser::CreateFork(_Input NEFS_FORK_STRUCT& the_fork) {
85 if (the_fork.ForkName[0] == 0 || the_fork.CatalogName[0] == 0 || the_fork.DataSize == 0) {
86 return NO;
87 }
88
89 auto catalog = this->GetCatalog(the_fork.CatalogName);
90 if (!catalog) return NO;
91
92 Lba lba = catalog->DataFork;
93 if (lba < kNeFSCatalogStartAddress) {
94 delete catalog;
95 return NO;
96 }
97
98 auto& drv = kMountpoint.A();
99 Lba lba_prev = lba;
100 NEFS_FORK_STRUCT prev_fork{};
101 NEFS_FORK_STRUCT cur_fork{};
102
103 while (true) {
104 drv.fPacket.fPacketLba = lba;
105 drv.fPacket.fPacketSize = sizeof(NEFS_FORK_STRUCT);
106 drv.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&cur_fork);
107 drv.fInput(drv.fPacket);
108 if (!drv.fPacket.fPacketGood) {
109 delete catalog;
110 return NO;
111 }
112
113 if (cur_fork.Flags & kNeFSFlagCreated) {
114 if (KStringBuilder::Equals(cur_fork.ForkName, the_fork.ForkName) &&
115 KStringBuilder::Equals(cur_fork.CatalogName, the_fork.CatalogName)) {
116 break;
117 }
118 lba_prev = lba;
119 lba = cur_fork.NextSibling;
120 prev_fork = cur_fork;
121 } else {
122 if (lba >= kNeFSCatalogStartAddress) {
123 prev_fork.NextSibling = lba;
124 drv.fPacket.fPacketLba = lba_prev;
125 drv.fPacket.fPacketSize = sizeof(NEFS_FORK_STRUCT);
126 drv.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&prev_fork);
127 drv.fOutput(drv.fPacket);
128 }
129 break;
130 }
131 }
132
133 SizeT sectorCount = (the_fork.DataSize + kNeFSSectorSz - 1) / kNeFSSectorSz;
134 the_fork.DataOffset = lba + 1;
135 the_fork.PreviousSibling = lba_prev;
136 the_fork.NextSibling = lba + 1 + sectorCount;
137 the_fork.Flags |= kNeFSFlagCreated;
138
139 drv.fPacket.fPacketLba = lba;
140 drv.fPacket.fPacketSize = sizeof(NEFS_FORK_STRUCT);
141 drv.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&the_fork);
142 drv.fOutput(drv.fPacket);
143
144 fs_ifs_write(&kMountpoint, drv, MountpointInterface::kDriveIndexA);
145
146 delete catalog;
147 return YES;
148}

155/***********************************************************************************/
156_Output NEFS_FORK_STRUCT* NeFileSystemParser::FindFork(_Input NEFS_CATALOG_STRUCT* catalog,
157 _Input const Char* name,
158 _Input Boolean is_data) {
159 if (!catalog || !name) return nullptr;
160
161 auto& drive = kMountpoint.A();
162 Lba lba = is_data ? catalog->DataFork : catalog->ResourceFork;
163 NEFS_FORK_STRUCT local_buf{};
164
165 while (lba >= kNeFSCatalogStartAddress) {
166 drive.fPacket.fPacketLba = lba;
167 drive.fPacket.fPacketSize = sizeof(NEFS_FORK_STRUCT);
168 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&local_buf);
169
170 rt_copy_memory_safe((VoidPtr) "fs/nefs-packet", drive.fPacket.fPacketMime,
171 rt_string_len("fs/nefs-packet"), sizeof(drive.fPacket.fPacketMime));
172
173 if (auto res = fs_ifs_read(&kMountpoint, drive, this->mDriveIndex); res) {
174 switch (res) {
175 case 1:
176 err_global_get() = kErrorDiskReadOnly;
177 break;
178 case 2:
179 err_global_get() = kErrorDiskIsFull;
180 break;
181 case 3:
182 err_global_get() = kErrorNoSuchDisk;
183 break;
184 default:
185 break;
186 }
187 return nullptr;
188 }
189
190 if (KStringBuilder::Equals(local_buf.ForkName, name) &&
191 KStringBuilder::Equals(local_buf.CatalogName, catalog->Name)) {
192 auto result = new NEFS_FORK_STRUCT();
193 rt_copy_memory_safe(&local_buf, result, sizeof(NEFS_FORK_STRUCT), sizeof(NEFS_FORK_STRUCT));
194 return result;
195 }
196
197 lba = local_buf.NextSibling;
198 }
199
200 return nullptr;
201}

220_Output NEFS_CATALOG_STRUCT* NeFileSystemParser::CreateCatalog(_Input const Char* name,
221 _Input const Int32& flags,
222 _Input const Int32& kind) {
223 kout << "CreateCatalog(*...*)\r";
224
225 if (!name) return nullptr;
226 SizeT nameLen = rt_string_len(name);
227 if (nameLen == 0) return nullptr;
228
229 Lba out_lba = 0UL;
230
231 kout << "Checking for path separator...\r";
232
234 if (kind == kNeFSCatalogKindDir && name[nameLen - 1] != NeFileSystemHelper::Separator())
235 return nullptr;
236
238 if (kind != kNeFSCatalogKindDir && name[nameLen - 1] == NeFileSystemHelper::Separator())
239 return nullptr;
240
241 NEFS_CATALOG_STRUCT* catalog_copy = this->FindCatalog(name, out_lba);
242
243 if (catalog_copy) {
244 kout << "Catalog already exists: " << name << ".\r";
245 err_global_get() = kErrorFileExists;
246 delete catalog_copy;
247 catalog_copy = nullptr;
248 return nullptr;
249 }
250
251 Char* parent_name = (Char*) mm_alloc_ptr(nameLen + 1, Yes, No);
252 rt_copy_memory_safe(const_cast<Char*>(name), parent_name, nameLen + 1, nameLen + 1);
253
254 if (nameLen < 2) {
255 mm_free_ptr(parent_name);
256 err_global_get() = kErrorFileNotFound;
257 return nullptr;
258 }
259
260 SizeT index_reverse_copy = nameLen - 1;
261 if (parent_name[index_reverse_copy] == NeFileSystemHelper::Separator()) {
262 parent_name[index_reverse_copy] = 0;
263 --index_reverse_copy;
264 }
265 while (index_reverse_copy > 0 &&
266 parent_name[index_reverse_copy] != NeFileSystemHelper::Separator()) {
267 parent_name[index_reverse_copy] = 0;
268 --index_reverse_copy;
269 }
270 if (index_reverse_copy == 0 && parent_name[0] != NeFileSystemHelper::Separator()) {
271 mm_free_ptr(parent_name);
272 err_global_get() = kErrorFileNotFound;
273 return nullptr;
274 }
275
276 NEFS_CATALOG_STRUCT* catalog = this->FindCatalog(parent_name, out_lba);
277 mm_free_ptr(parent_name);
278
279 auto& drive = kMountpoint.A();
280 if (catalog && catalog->Kind == kNeFSCatalogKindFile) {
281 kout << "Parent is a file.\r";
282 delete catalog;
283 return nullptr;
284 } else if (!catalog) {
285 Char part_block[sizeof(NEFS_ROOT_PARTITION_BLOCK)] = {0};
286 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(part_block);
287 drive.fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
288 drive.fPacket.fPacketLba = kNeFSRootCatalogStartAddress;
289 drive.fInput(drive.fPacket);
290
291 NEFS_ROOT_PARTITION_BLOCK* blk_nefs = (NEFS_ROOT_PARTITION_BLOCK*) part_block;
292 out_lba = blk_nefs->StartCatalog;
293 }
294
295 if (drive.fPacket.fPacketReadOnly) {
296 delete catalog;
297 return nullptr;
298 }
299
300 NEFS_CATALOG_STRUCT* child_catalog = new NEFS_CATALOG_STRUCT();
301 child_catalog->Checksum = 0;
302 child_catalog->ResourceForkSize = 0UL;
303 child_catalog->DataForkSize = 0UL;
304 child_catalog->CatalogFlags = kNeFSStatusUnlocked;
305 child_catalog->NextSibling = out_lba;
306 child_catalog->PrevSibling = out_lba;
307 child_catalog->Kind = kind;
308 child_catalog->Flags |= kNeFSFlagCreated;
309 child_catalog->CatalogFlags = flags;
310
311 SizeT i = nameLen;
312 --i;
313 if (kind == kNeFSCatalogKindDir) --i;
314 while (name[i] != '/') --i;
315 rt_copy_memory_safe((VoidPtr) (name + i), (VoidPtr) child_catalog->Name, rt_string_len(name) - i,
316 kNeFSCatalogNameLen);
317
318 NEFS_CATALOG_STRUCT temporary_catalog{};
319 Lba start_free = out_lba;
320
321 rt_copy_memory_safe((VoidPtr) "fs/nefs-packet", drive.fPacket.fPacketMime,
322 rt_string_len("fs/nefs-packet"), sizeof(drive.fPacket.fPacketMime));
323
324 Char buf_part_block[sizeof(NEFS_ROOT_PARTITION_BLOCK)] = {0};
325 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(buf_part_block);
326 drive.fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
327 drive.fPacket.fPacketLba = kNeFSRootCatalogStartAddress;
328 drive.fInput(drive.fPacket);
329
330 NEFS_ROOT_PARTITION_BLOCK* part_block = (NEFS_ROOT_PARTITION_BLOCK*) buf_part_block;
331 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&temporary_catalog);
332 drive.fPacket.fPacketSize = sizeof(NEFS_CATALOG_STRUCT);
333 drive.fPacket.fPacketLba = start_free;
334 drive.fInput(drive.fPacket);
335
336 if (part_block->FreeCatalog < 1) {
337 delete child_catalog;
338 delete catalog;
339 return nullptr;
340 }
341
342 kout << "Start finding catalog to allocate or empty space...\r";
343
344 SizeT catalogSectors = (sizeof(NEFS_CATALOG_STRUCT) + drive.fSectorSz - 1) / drive.fSectorSz;
345 while (start_free < part_block->StartCatalog + (part_block->CatalogCount * catalogSectors)) {
346 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&temporary_catalog);
347 drive.fPacket.fPacketSize = sizeof(NEFS_CATALOG_STRUCT);
348 drive.fPacket.fPacketLba = start_free;
349 drive.fInput(drive.fPacket);
350
351 if ((temporary_catalog.Flags & kNeFSFlagCreated) == 0) {
352 child_catalog->NextSibling = start_free + catalogSectors;
353
354 NEFS_CATALOG_STRUCT placeholder{};
355 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&placeholder);
356 drive.fPacket.fPacketSize = sizeof(NEFS_CATALOG_STRUCT);
357 drive.fPacket.fPacketLba = start_free;
358 drive.fOutput(drive.fPacket);
359
360 child_catalog->DataFork = part_block->DiskSize - start_free;
361 child_catalog->ResourceFork = child_catalog->DataFork;
362
363 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(child_catalog);
364 drive.fPacket.fPacketSize = sizeof(NEFS_CATALOG_STRUCT);
365 drive.fPacket.fPacketLba = start_free;
366 drive.fOutput(drive.fPacket);
367
368 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(buf_part_block);
369 drive.fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
370 drive.fPacket.fPacketLba = kNeFSRootCatalogStartAddress;
371 drive.fInput(drive.fPacket);
372
373 part_block->FreeSectors -= catalogSectors;
374 part_block->CatalogCount += 1;
375 part_block->FreeCatalog -= 1;
376
377 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(part_block);
378 drive.fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
379 drive.fPacket.fPacketLba = kNeFSRootCatalogStartAddress;
380 drive.fOutput(drive.fPacket);
381
382 delete catalog;
383 NEFS_CATALOG_STRUCT* found_catalog = new NEFS_CATALOG_STRUCT();
384 rt_copy_memory_safe(&temporary_catalog, found_catalog, sizeof(NEFS_CATALOG_STRUCT),
385 sizeof(NEFS_CATALOG_STRUCT));
386
387 delete child_catalog;
388 return found_catalog;
389 } else if ((temporary_catalog.Flags & kNeFSFlagCreated) &&
390 KStringBuilder::Equals(temporary_catalog.Name, name)) {
391 rt_copy_memory_safe(&temporary_catalog, child_catalog, sizeof(NEFS_CATALOG_STRUCT),
392 sizeof(NEFS_CATALOG_STRUCT));
393 delete catalog;
394 return child_catalog;
395 }
396
397 start_free += catalogSectors;
398 }
399
400 delete child_catalog;
401 delete catalog;
402 return nullptr;
403}

410bool NeFileSystemParser::Format(_Input _Output DriveTrait* drive, _Input const Int32 flags,
411 const Char* part_name) {
412 if (!part_name || *part_name == 0) return false;
413 NE_UNUSED(flags);
414
415 // verify disk.
416 drive->fVerify(drive->fPacket);
417 rt_copy_memory_safe((VoidPtr) "fs/nefs-packet", drive->fPacket.fPacketMime,
418 rt_string_len("fs/nefs-packet"), sizeof(drive->fPacket.fPacketMime));
419 if (!drive->fPacket.fPacketGood) {
420 err_global_get() = kErrorDiskIsCorrupted;
421 return false;
422 }
423
424 Char fs_buf[sizeof(NEFS_ROOT_PARTITION_BLOCK)] = {0};
425 Lba start = drive->fLbaStart;
426
427 drive->fPacket.fPacketContent = reinterpret_cast<VoidPtr>(fs_buf);
428 drive->fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
429 drive->fPacket.fPacketLba = start;
430 drive->fInput(drive->fPacket);
431
432 NEFS_ROOT_PARTITION_BLOCK* part_block = (NEFS_ROOT_PARTITION_BLOCK*) fs_buf;
433 if (rt_string_cmp(kNeFSIdent, part_block->Ident, kNeFSIdentLen) == 0) return true;
434
435 const auto kNeFSUntitledHD = part_name;
436 rt_copy_memory_safe((VoidPtr) kNeFSIdent, (VoidPtr) part_block->Ident, kNeFSIdentLen,
437 sizeof(part_block->Ident));
438 rt_copy_memory_safe((VoidPtr) kNeFSUntitledHD, (VoidPtr) part_block->PartitionName,
439 rt_string_len(kNeFSUntitledHD), sizeof(part_block->PartitionName));
440
441 SizeT sectorCount = drv_std_get_sector_count();
442 SizeT sectorSize = drive->fSectorSz;
443 SizeT totalBytes = sectorCount * sectorSize;
444 SizeT catalogEntries = totalBytes / sizeof(NEFS_CATALOG_STRUCT);
445 SizeT catalogSectors = (sizeof(NEFS_CATALOG_STRUCT) + sectorSize - 1) / sectorSize;
446
447 part_block->Version = kNeFSVersionInteger;
448 part_block->Kind = kNeFSPartitionTypeStandard;
449 part_block->StartCatalog = start + catalogSectors;
450 part_block->Flags = 0UL;
451 part_block->CatalogCount = catalogEntries;
452 part_block->FreeCatalog = catalogEntries - 1;
453 part_block->SectorCount = sectorCount;
454 part_block->DiskSize = totalBytes;
455 part_block->SectorSize = sectorSize;
456 part_block->FreeSectors = sectorCount - catalogSectors;
457
458 drive->fPacket.fPacketContent = reinterpret_cast<VoidPtr>(fs_buf);
459 drive->fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
460 drive->fPacket.fPacketLba = start;
461 drive->fOutput(drive->fPacket);
462
463 (Void)(kout << "Drive kind: " << drive->fProtocol() << kendl);
464 (Void)(kout << "Partition name: " << part_block->PartitionName << kendl);
465 (Void)(kout << "Start catalog: " << hex_number(part_block->StartCatalog) << kendl);
466 (Void)(kout << "Number of catalogs: " << hex_number(part_block->CatalogCount) << kendl);
467 (Void)(kout << "Free catalog: " << hex_number(part_block->FreeCatalog) << kendl);
468 (Void)(kout << "Free sectors: " << hex_number(part_block->FreeSectors) << kendl);
469 (Void)(kout << "Sector size: " << hex_number(part_block->SectorSize) << kendl);
470
471 return true;
472}

474/***********************************************************************************/
479/***********************************************************************************/
480bool NeFileSystemParser::WriteCatalog(_Input const Char* catalog_name, Bool is_rsrc_fork,
481 _Input VoidPtr data, _Input SizeT size_of_data,
482 _Input const Char* fork_name) {
483 if (size_of_data < 1) return NO;
484
485 auto catalog = this->GetCatalog(catalog_name);
486 if (!catalog) {
487 kout << "NeFS: WriteCatalog failed to find catalog: " << catalog_name << "\n";
488 return false;
489 }
490
491 SizeT maxSize = is_rsrc_fork ? catalog->ResourceForkSize : catalog->DataForkSize;
492
493 if (!is_valid_size(size_of_data, maxSize)) {
494 (Void)(kout << "NeFS: WriteCatalog called with invalid size: " << hex_number(size_of_data));
495 kout << "\n";
496
497 delete catalog;
498 return false;
499 }
500
501 Lba startFork = is_rsrc_fork ? catalog->ResourceFork : catalog->DataFork;
502 auto& drive = kMountpoint.A();
503
504 if (!is_valid_lba(startFork, drive)) {
505 (Void)(kout << "NeFS: WriteCatalog called with invalid LBA: " << hex_number(startFork));
506 kout << "\n";
507
508 delete catalog;
509 return false;
510 }
511
512 NEFS_ROOT_PARTITION_BLOCK part_block;
513 drive.fPacket.fPacketLba = kNeFSRootCatalogStartAddress;
514 drive.fPacket.fPacketSize = sizeof(part_block);
515 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&part_block);
516 drive.fInput(drive.fPacket);
517
518 auto buf = new UInt8[size_of_data];
519 rt_set_memory(buf, 0, size_of_data);
520 rt_copy_memory_safe(data, buf, size_of_data, size_of_data);
521 rt_copy_memory_safe((VoidPtr) "fs/nefs-packet", drive.fPacket.fPacketMime,
522 rt_string_len("fs/nefs-packet"), sizeof(drive.fPacket.fPacketMime));
523
524 NEFS_FORK_STRUCT* fork_data_input = new NEFS_FORK_STRUCT();
525 NEFS_FORK_STRUCT prev_fork{};
526
527 while (startFork >= part_block.StartCatalog && drive.fPacket.fPacketGood) {
528 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(fork_data_input);
529 drive.fPacket.fPacketSize = sizeof(NEFS_FORK_STRUCT);
530 drive.fPacket.fPacketLba = startFork;
531 drive.fInput(drive.fPacket);
532
533 if (!drive.fPacket.fPacketGood) {
534 break;
535 }
536
537 if ((fork_data_input->Flags & kNeFSFlagCreated) &&
538 KStringBuilder::Equals(fork_data_input->ForkName, fork_name) &&
539 KStringBuilder::Equals(fork_data_input->CatalogName, catalog_name) &&
540 fork_data_input->DataSize == size_of_data) {
541 SizeT bytes_left = size_of_data;
542 SizeT offset = 0;
543 Lba base_lba = fork_data_input->DataOffset;
544
545 while (bytes_left > 0) {
546 SizeT chunk = (bytes_left > kNeFSSectorSz) ? kNeFSSectorSz : bytes_left;
547 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(buf + offset);
548 drive.fPacket.fPacketSize = chunk;
549 drive.fPacket.fPacketLba = base_lba + (offset / kNeFSSectorSz);
550 drive.fOutput(drive.fPacket);
551 offset += chunk;
552 bytes_left -= chunk;
553 }
554
555 delete fork_data_input;
556 delete[] buf;
557 delete catalog;
558 return true;
559 }
560
561 prev_fork = *fork_data_input;
562 startFork = fork_data_input->NextSibling;
563 }
564
565 delete fork_data_input;
566 delete[] buf;
567 delete catalog;
568 return false;
569}

571/***********************************************************************************/
575/***********************************************************************************/
576_Output NEFS_CATALOG_STRUCT* NeFileSystemParser::FindCatalog(_Input const Char* catalog_name,
577 Lba& out_lba, Bool search_hidden,
578 Bool local_search) {
579 if (!catalog_name || *catalog_name == 0) return nullptr;
580
581 NEFS_ROOT_PARTITION_BLOCK part{};
582 auto& drive = kMountpoint.A();
583
584 rt_copy_memory_safe((VoidPtr) "fs/nefs-packet", drive.fPacket.fPacketMime,
585 rt_string_len("fs/nefs-packet"), sizeof(drive.fPacket.fPacketMime));
586 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&part);
587 drive.fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
588 drive.fPacket.fPacketLba = kNeFSRootCatalogStartAddress;
589 drive.fInput(drive.fPacket);
590
591 auto start_catalog_lba = kNeFSCatalogStartAddress;
592
593 // Helper lambda to scan from a given LBA
594 auto scan_from = [&](Lba lba_start, Bool allow_hidden) -> NEFS_CATALOG_STRUCT* {
595 Lba cursor = lba_start;
596 NEFS_CATALOG_STRUCT tmp{};
597 while (cursor >= part.StartCatalog && drive.fPacket.fPacketGood) {
598 drive.fPacket.fPacketLba = cursor;
599 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&tmp);
600 drive.fPacket.fPacketSize = sizeof(NEFS_CATALOG_STRUCT);
601 drive.fInput(drive.fPacket);
602
603 if (KStringBuilder::Equals(
604 tmp.Name, catalog_name + (rt_string_len(catalog_name) - rt_string_len(tmp.Name)))) {
605 if (tmp.Status == kNeFSStatusLocked && !allow_hidden) {
606 err_global_get() = kErrorFileLocked;
607 return nullptr;
608 }
609 if (!(tmp.Flags & kNeFSFlagCreated)) {
610 err_global_get() = kErrorFileNotFound;
611 return nullptr;
612 }
613 NEFS_CATALOG_STRUCT* catalog_ptr = new NEFS_CATALOG_STRUCT();
614 rt_copy_memory_safe(&tmp, catalog_ptr, sizeof(NEFS_CATALOG_STRUCT),
615 sizeof(NEFS_CATALOG_STRUCT));
616 out_lba = cursor;
617 return catalog_ptr;
618 }
619 cursor = tmp.NextSibling;
620 }
621 return nullptr;
622 };
623
624 if (!KStringBuilder::Equals(catalog_name, NeFileSystemHelper::Root()) && local_search) {
625 Char parent_name[kNeFSCatalogNameLen] = {0};
626 SizeT nameLen = rt_string_len(catalog_name);
627 rt_copy_memory_safe(const_cast<Char*>(catalog_name), parent_name, nameLen + 1,
628 kNeFSCatalogNameLen);
629
630 SizeT indexReverseCopy = nameLen - 1;
631 if (parent_name[indexReverseCopy] == NeFileSystemHelper::Separator()) {
632 parent_name[indexReverseCopy] = 0;
633 --indexReverseCopy;
634 }
635 while (indexReverseCopy > 0 &&
636 parent_name[indexReverseCopy] != NeFileSystemHelper::Separator()) {
637 parent_name[indexReverseCopy] = 0;
638 --indexReverseCopy;
639 }
640 if (indexReverseCopy == 0 && parent_name[0] != NeFileSystemHelper::Separator()) {
641 return nullptr;
642 }
643

684_Output Boolean NeFileSystemParser::RemoveCatalog(_Input const Char* catalog_name) {
685 if (!catalog_name || KStringBuilder::Equals(catalog_name, NeFileSystemHelper::Root())) {
686 err_global_get() = kErrorInternal;
687 return false;
688 }
689
690 Lba out_lba = 0;
691 auto catalog = this->FindCatalog(catalog_name, out_lba, YES, YES);
692 if (!catalog) return false;
693
694 auto& drive = kMountpoint.A();
695 NEFS_FORK_STRUCT fork_buf{};
696 Lba fork_lba = catalog->DataFork;
697 while (fork_lba >= kNeFSCatalogStartAddress) {
698 drive.fPacket.fPacketLba = fork_lba;
699 drive.fPacket.fPacketSize = sizeof(fork_buf);
700 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&fork_buf);
701 drive.fInput(drive.fPacket);
702
703 fork_buf.Flags &= (~kNeFSFlagCreated);
704 fork_buf.Flags |= kNeFSFlagDeleted;
705
706 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&fork_buf);
707 drive.fPacket.fPacketSize = sizeof(fork_buf);
708 drive.fPacket.fPacketLba = fork_lba;
709 drive.fOutput(drive.fPacket);
710
711 fork_lba = fork_buf.NextSibling;
712 }
713
714 fork_lba = catalog->ResourceFork;
715 while (fork_lba >= kNeFSCatalogStartAddress) {
716 drive.fPacket.fPacketLba = fork_lba;
717 drive.fPacket.fPacketSize = sizeof(fork_buf);
718 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&fork_buf);
719 drive.fInput(drive.fPacket);
720
721 fork_buf.Flags &= (~kNeFSFlagCreated);
722 fork_buf.Flags |= kNeFSFlagDeleted;
723
724 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(&fork_buf);
725 drive.fPacket.fPacketSize = sizeof(fork_buf);
726 drive.fPacket.fPacketLba = fork_lba;
727 drive.fOutput(drive.fPacket);
728
729 fork_lba = fork_buf.NextSibling;
730 }
731
732 if (out_lba >= kNeFSCatalogStartAddress || (catalog->Flags & kNeFSFlagCreated)) {
733 catalog->Flags &= (~kNeFSFlagCreated);
734 catalog->Flags |= kNeFSFlagDeleted;
735
736 rt_copy_memory_safe((VoidPtr) "fs/nefs-packet", drive.fPacket.fPacketMime,
737 rt_string_len("fs/nefs-packet"), sizeof(drive.fPacket.fPacketMime));
738 drive.fPacket.fPacketLba = out_lba;
739 drive.fPacket.fPacketSize = sizeof(NEFS_CATALOG_STRUCT);
740 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(catalog);
741 drive.fOutput(drive.fPacket);
742
743 Char partitionBlockBuf[sizeof(NEFS_ROOT_PARTITION_BLOCK)] = {0};
744 drive.fPacket.fPacketLba = kNeFSRootCatalogStartAddress;
745 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(partitionBlockBuf);
746 drive.fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
747 drive.fInput(drive.fPacket);
748
749 NEFS_ROOT_PARTITION_BLOCK* part_block = (NEFS_ROOT_PARTITION_BLOCK*) partitionBlockBuf;
750 if (part_block->CatalogCount > 0) --part_block->CatalogCount;
751 ++part_block->FreeSectors;
752
753 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(part_block);
754 drive.fPacket.fPacketSize = sizeof(NEFS_ROOT_PARTITION_BLOCK);
755 drive.fPacket.fPacketLba = kNeFSRootCatalogStartAddress;
756 drive.fOutput(drive.fPacket);
757
758 delete catalog;
759 catalog = nullptr;
760 return true;
761 }
762
763 delete catalog;
764 catalog = nullptr;
765 return false;
766}

772/***********************************************************************************/
777/***********************************************************************************/
778VoidPtr NeFileSystemParser::ReadCatalog(_Input _Output NEFS_CATALOG_STRUCT* catalog,
779 _Input Bool is_rsrc_fork, _Input SizeT dataSz,
780 _Input const Char* forkName) {
781 if (!catalog) {
782 err_global_get() = kErrorInvalidData;
783 return nullptr;
784 }
785 // Validate size against fork size
786 SizeT maxSize = is_rsrc_fork ? catalog->ResourceForkSize : catalog->DataForkSize;
787 if (!is_valid_size(dataSz, maxSize)) {
788 kout << "NeFS: ReadCatalog called with invalid size: ";
789 hex_number(dataSz);
790 kout << "\n";
791 return nullptr;
792 }
793
794 Lba dataForkLba = is_rsrc_fork ? catalog->ResourceFork : catalog->DataFork;
795 auto& drive = kMountpoint.A();
796 if (!is_valid_lba(dataForkLba, drive)) {
797 kout << "NeFS: ReadCatalog called with invalid LBA: ";
798 hex_number(dataForkLba);
799 kout << "\n";
800 return nullptr;
801 }
802
803 auto* fs_buf = new NEFS_FORK_STRUCT();
804 rt_copy_memory_safe((VoidPtr) "fs/nefs-packet", drive.fPacket.fPacketMime,
805 rt_string_len("fs/nefs-packet"), sizeof(drive.fPacket.fPacketMime));
806
807 NEFS_FORK_STRUCT* fs_fork_data = nullptr;
808 while (dataForkLba >= kNeFSCatalogStartAddress) {
809 drive.fPacket.fPacketLba = dataForkLba;
810 drive.fPacket.fPacketSize = sizeof(NEFS_FORK_STRUCT);
811 drive.fPacket.fPacketContent = reinterpret_cast<VoidPtr>(fs_buf);
812 drive.fInput(drive.fPacket);
813
814 fs_fork_data = fs_buf;
815 (Void)(kout << "ForkName: " << fs_fork_data->ForkName << kendl);
816 (Void)(kout << "CatalogName: " << fs_fork_data->CatalogName << kendl);
817
818 if (KStringBuilder::Equals(forkName, fs_fork_data->ForkName) &&
819 KStringBuilder::Equals(catalog->Name, fs_fork_data->CatalogName)) {
820 break;
821 }
822 dataForkLba = fs_fork_data->NextSibling;
823 }
824
825 if (dataForkLba < kNeFSCatalogStartAddress) {
826 delete fs_buf;
827 return nullptr;
828 }
829 return fs_fork_data;
830}
