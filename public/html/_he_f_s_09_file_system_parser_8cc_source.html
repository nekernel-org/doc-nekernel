82 STATIC UInt64 hefsi_to_big_endian_64(UInt64 val) {
83 return ((val >> 56) & 0x00000000000000FFULL) | ((val >> 40) & 0x000000000000FF00ULL) |
84 ((val >> 24) & 0x0000000000FF0000ULL) | ((val >> 8) & 0x00000000FF000000ULL) |
85 ((val << 8) & 0x000000FF00000000ULL) | ((val << 24) & 0x0000FF0000000000ULL) |
86 ((val << 40) & 0x00FF000000000000ULL) | ((val << 56) & 0xFF00000000000000ULL);
87 }
91 STATIC UInt64 hefsi_hash_64(const Utf8Char* path) {
92 if (!path || *path == 0) return 0;
93
94 const UInt64 FNV_OFFSET_BASIS = 0xcbf29ce484222325ULL;
95 const UInt64 FNV_PRIME = 0x100000001b3ULL;
96
97 UInt64 hash = FNV_OFFSET_BASIS;
98
99 while (*path) {
100 hash ^= (Utf8Char) (*path++);
101 hash *= FNV_PRIME;
102 }
103
104 return hefsi_to_big_endian_64(hash);
105 }

112 STATIC ATTRIBUTE(unused) Void hefsi_traverse_tree(HEFS_INDEX_NODE_DIRECTORY* dir, DriveTrait* mnt,
113 const Lba& ind_start, Lba& start) {
114 if (!mnt || !dir) return;
115
116 BOOL check_is_good = NO;
117 HEFS_INDEX_NODE_DIRECTORY* dir_tmp = new HEFS_INDEX_NODE_DIRECTORY();
118
119 while (YES) {
120 mnt->fPacket.fPacketLba = start;
121 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
122 mnt->fPacket.fPacketContent = dir_tmp;
123 mnt->fInput(mnt->fPacket);
124
125 if (!mnt->fPacket.fPacketGood) break;
126
127 if (dir_tmp->fNext != 0) {
128 if (check_is_good) break;
129
130 start = dir_tmp->fNext;
131
132 check_is_good = YES;
133 continue;
134 } else if (dir_tmp->fPrev != 0) {
135 if (check_is_good) break;
136
137 start = dir_tmp->fPrev;
138 check_is_good = YES;
139 continue;
140 } else {
141 if (dir_tmp->fParent != 0) {
142 if (check_is_good) break;
143
144 start = dir_tmp->fParent;
145 check_is_good = YES;
146 continue;
147 } else if (dir_tmp->fPrev != 0) {
148 if (check_is_good) break;
149
150 start = dir_tmp->fPrev;
151 check_is_good = YES;
152 continue;
153 } else {
154 if (start == 0) {
155 start = ind_start;
156 continue;
157 }
158
159 start += sizeof(HEFS_INDEX_NODE_DIRECTORY);
160 break;
161 }
162 }
163 }
164
165 delete dir_tmp;
166
167 start += sizeof(HEFS_INDEX_NODE_DIRECTORY);
168 if (start == 0) start = ind_start;
169 }

175 STATIC ATTRIBUTE(unused) _Output Void hefsi_rotate_tree(Lba& start, DriveTrait* mnt) {
176 if (!start || !mnt) return;
177
178 HEFS_INDEX_NODE_DIRECTORY* cur =
179 (HEFS_INDEX_NODE_DIRECTORY*) RTL_ALLOCA(sizeof(HEFS_INDEX_NODE_DIRECTORY));
180
181 mnt->fPacket.fPacketLba = start;
182 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
183 mnt->fPacket.fPacketContent = cur;
184
185 mnt->fInput(mnt->fPacket);
186
187 if (cur->fHashPath == 0) return;
188
189 HEFS_INDEX_NODE_DIRECTORY* sibling =
190 (HEFS_INDEX_NODE_DIRECTORY*) RTL_ALLOCA(sizeof(HEFS_INDEX_NODE_DIRECTORY));
191
192 mnt->fPacket.fPacketLba = cur->fPrev;
193 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
194 mnt->fPacket.fPacketContent = sibling;
195
196 mnt->fInput(mnt->fPacket);
197
198 if (sibling->fHashPath == 0) return;
199
200 auto child_sibling = sibling->fChild;
201 auto child_cur = cur->fChild;
202
203 cur->fChild = child_sibling;
204 sibling->fChild = child_cur;
205
206 sibling->fChecksum = ke_calculate_crc32((Char*) sibling, sizeof(HEFS_INDEX_NODE_DIRECTORY));
207 cur->fChecksum = ke_calculate_crc32((Char*) cur, sizeof(HEFS_INDEX_NODE_DIRECTORY));
208
209 mnt->fPacket.fPacketLba = cur->fParent;
210 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
211 mnt->fPacket.fPacketContent = sibling;
212
213 mnt->fOutput(mnt->fPacket);
214
215 mnt->fPacket.fPacketLba = start;
216 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
217 mnt->fPacket.fPacketContent = cur;
218
219 mnt->fOutput(mnt->fPacket);
220
221 HEFS_INDEX_NODE_DIRECTORY* sibling_child =
222 (HEFS_INDEX_NODE_DIRECTORY*) RTL_ALLOCA(sizeof(HEFS_INDEX_NODE_DIRECTORY));
223
224 mnt->fPacket.fPacketLba = child_sibling;
225 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
226 mnt->fPacket.fPacketContent = sibling_child;
227
228 mnt->fInput(mnt->fPacket);
229
230 sibling_child->fParent = cur->fParent;
231
232 sibling_child->fChecksum =
233 ke_calculate_crc32((Char*) sibling, sizeof(HEFS_INDEX_NODE_DIRECTORY));
234
235 mnt->fOutput(mnt->fPacket);
236
237 HEFS_INDEX_NODE_DIRECTORY* cur_child =
238 (HEFS_INDEX_NODE_DIRECTORY*) RTL_ALLOCA(sizeof(HEFS_INDEX_NODE_DIRECTORY));
239
240 mnt->fPacket.fPacketLba = child_cur;
241 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
242 mnt->fPacket.fPacketContent = cur_child;
243
244 mnt->fInput(mnt->fPacket);
245
246 cur_child->fParent = start;
247
248 cur_child->fChecksum = ke_calculate_crc32((Char*) sibling, sizeof(HEFS_INDEX_NODE_DIRECTORY));
249
250 mnt->fOutput(mnt->fPacket);
251
252 kout << "RB-Tree has been rotated.\r";
253 }

262 STATIC _Output BOOL hefsi_update_ind_status(HEFS_BOOT_NODE* boot, DriveTrait* mnt,
263 const Utf8Char* dir_name, UInt16 flags,
264 const BOOL delete_or_create) {
265 if (mnt) {
266 HEFS_INDEX_NODE_DIRECTORY* tmpdir =
267 (HEFS_INDEX_NODE_DIRECTORY*) mm_alloc_ptr(sizeof(HEFS_INDEX_NODE_DIRECTORY), Yes, No);
268
269 auto start = boot->fStartIND;
270 auto prev_location = start;
271 auto parent_location = 0UL;
272
273 MUST_PASS(boot->fStartIND > mnt->fLbaStart);
274
275 while (YES) {
276 auto prev_start = start;
277
278 if (start)
279 mnt->fPacket.fPacketLba = start;
280 else
281 mnt->fPacket.fPacketLba = prev_location + sizeof(HEFS_INDEX_NODE_DIRECTORY);
282
283 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
284 mnt->fPacket.fPacketContent = tmpdir;
285
286 mnt->fInput(mnt->fPacket);
287
288 BOOL expr = NO;
289
290 if (!delete_or_create) {
291 expr = (!tmpdir->fCreated && tmpdir->fDeleted) || tmpdir->fHashPath == 0;
292 } else {
293 expr =
294 tmpdir->fCreated && !tmpdir->fDeleted && hefsi_hash_64(dir_name) == tmpdir->fHashPath;
295 }
296
297 if (expr) {
298 HEFS_INDEX_NODE_DIRECTORY* dirent =
299 (HEFS_INDEX_NODE_DIRECTORY*) mm_alloc_ptr(sizeof(HEFS_INDEX_NODE_DIRECTORY), Yes, No);
300
301 rt_set_memory(dirent, 0, sizeof(HEFS_INDEX_NODE_DIRECTORY));
302
303 dirent->fHashPath = delete_or_create ? 0UL : hefsi_hash_64(dir_name);
304 dirent->fAccessed = 0UL;
305 dirent->fCreated = delete_or_create ? 0UL : 1UL;
306 dirent->fDeleted = delete_or_create ? 1UL : 0UL;
307 dirent->fModified = 0UL;
308 dirent->fEntryCount = 0UL;
309
310 dirent->fReserved = 0;
311 dirent->fFlags = flags;
312 dirent->fChecksum = 0;
313
314 dirent->fEntryCount = 0;
315
316 if (parent_location) {
317 HEFS_INDEX_NODE_DIRECTORY* tmpend =
318 (HEFS_INDEX_NODE_DIRECTORY*) RTL_ALLOCA(sizeof(HEFS_INDEX_NODE_DIRECTORY));
319
320 mnt->fPacket.fPacketLba = parent_location;
321 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
322 mnt->fPacket.fPacketContent = tmpend;
323
324 mnt->fInput(mnt->fPacket);
325
326 if (tmpend->fChecksum !=
327 ke_calculate_crc32((Char*) tmpend, sizeof(HEFS_INDEX_NODE_DIRECTORY)))
328 ke_panic(RUNTIME_CHECK_FILESYSTEM, "Bad CRC32 value, aborting.");
329
330 if (delete_or_create)
331 --tmpend->fEntryCount;
332 else
333 ++tmpend->fEntryCount;
334
335 tmpend->fChecksum =
336 ke_calculate_crc32((Char*) tmpend, sizeof(HEFS_INDEX_NODE_DIRECTORY));
337
338 mnt->fOutput(mnt->fPacket);
339
340 auto child_first = tmpend->fChild;
341
342 while (YES) {
343 mnt->fPacket.fPacketLba = child_first;
344 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
345 mnt->fPacket.fPacketContent = tmpend;
346
347 mnt->fInput(mnt->fPacket);
348
349 if ((!tmpend->fCreated && tmpend->fDeleted) || tmpend->fHashPath == 0) {
350 start = child_first;
351 break;
352 }
353
354 hefsi_traverse_tree(tmpend, mnt, boot->fStartIND, child_first);
355 }
356 }
357
358 dirent->fNext = tmpdir->fNext;
359 dirent->fPrev = tmpdir->fPrev;
360 dirent->fParent = tmpdir->fParent;
361 dirent->fChild = tmpdir->fChild;
362 dirent->fColor = tmpdir->fColor;
363
364 if (dirent->fColor == 0) {
365 dirent->fColor = dirent->fNext ? kHeFSRed : kHeFSBlack;
366 }
367
368 if (dirent->fPrev == 0) {
369 dirent->fPrev = boot->fStartIND;
370 }
371
372 if (dirent->fParent == 0) {
373 dirent->fParent = boot->fStartIND;
374 }
375
376 if (tmpdir->fChild == 0) {
377 auto child = dirent->fNext + sizeof(HEFS_INDEX_NODE_DIRECTORY);
378
379 HEFS_INDEX_NODE_DIRECTORY* tmpend =
380 (HEFS_INDEX_NODE_DIRECTORY*) RTL_ALLOCA(sizeof(HEFS_INDEX_NODE_DIRECTORY));
381
382 while (YES) {
383 mnt->fPacket.fPacketLba = child;
384 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
385 mnt->fPacket.fPacketContent = tmpend;
386
387 mnt->fInput(mnt->fPacket);
388
389 if ((!tmpend->fCreated && tmpend->fDeleted) || tmpdir->fHashPath == 0) {
390 break;
391 }
392
393 child += sizeof(HEFS_INDEX_NODE_DIRECTORY);
394 if (child > boot->fEndIND) break;
395 }
396
397 dirent->fColor = kHeFSRed;
398 dirent->fChild = child;
399
400 if (child > boot->fEndIND) dirent->fChild = boot->fStartIND;
401 }
402
403 for (SizeT index = 0UL; index < kHeFSSliceCount; ++index) {
404 dirent->fINSlices[index] = 0UL;
405 }
406
407 dirent->fChecksum = ke_calculate_crc32((Char*) dirent, sizeof(HEFS_INDEX_NODE_DIRECTORY));
408
409 mnt->fPacket.fPacketLba = prev_start;
410 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
411 mnt->fPacket.fPacketContent = dirent;
412
413 mnt->fOutput(mnt->fPacket);
414
415 err_global_get() = kErrorSuccess;
416
417 mm_free_ptr(dirent);
418 mm_free_ptr(tmpdir);
419
420 if (!delete_or_create)
421 ++boot->fINDCount;
422 else
423 --boot->fINDCount;
424
425 boot->fChecksum = ke_calculate_crc32((Char*) boot, sizeof(HEFS_BOOT_NODE));
426
427 mnt->fPacket.fPacketLba = mnt->fLbaStart;
428 mnt->fPacket.fPacketSize = sizeof(HEFS_BOOT_NODE);
429 mnt->fPacket.fPacketContent = boot;
430
431 mnt->fOutput(mnt->fPacket);
432
433 return YES;
434 }
435
436 prev_location = start;
437
438 hefsi_traverse_tree(tmpdir, mnt, boot->fStartIND, start);
439 if (start > boot->fEndIND || start == 0) break;
440 }
441
442 err_global_get() = kErrorDisk;
443 mm_free_ptr(tmpdir);
444
445 return NO;
446 }
447
448 err_global_get() = kErrorDiskIsFull;
449 return NO;
450 }

459 STATIC ATTRIBUTE(unused) _Output HEFS_INDEX_NODE* hefsi_fetch_in(HEFS_BOOT_NODE* boot,
460 DriveTrait* mnt,
461 const Utf8Char* dir_name,
462 const Utf8Char* file_name,
463 UInt8 kind) {
464 if (mnt) {
465 if (boot->fStartIND > boot->fEndIND) return nullptr;
466 if (boot->fStartIN > boot->fEndIN) return nullptr;
467
468 auto start = boot->fStartIND;
469 HEFS_INDEX_NODE* node = new HEFS_INDEX_NODE();
470
471 HEFS_INDEX_NODE_DIRECTORY* dir =
472 (HEFS_INDEX_NODE_DIRECTORY*) mm_alloc_ptr(sizeof(HEFS_INDEX_NODE_DIRECTORY), Yes, No);
473
474 while (YES) {
475 if (err_global_get() == kErrorDiskIsCorrupted) {
476 delete dir;
477 dir = nullptr;
478
479 delete node;
480 node = nullptr;
481
482 return nullptr;
483 }
484
485 mnt->fPacket.fPacketLba = start;
486 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
487 mnt->fPacket.fPacketContent = dir;
488
489 mnt->fInput(mnt->fPacket);
490
491 (Void)(kout << hex_number(hefsi_hash_64(dir_name)) << kendl);
492 (Void)(kout << hex_number(dir->fHashPath) << kendl);
493
494 if (dir->fHashPath == 0) break;
495
496 if (hefsi_hash_64(dir_name) == dir->fHashPath) {
497 for (SizeT inode_index = 0UL; inode_index < kHeFSSliceCount; ++inode_index) {
498 mnt->fPacket.fPacketLba = dir->fINSlices[inode_index];
499 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE);
500 mnt->fPacket.fPacketContent = node;
501
502 mnt->fInput(mnt->fPacket);
503
504 (Void)(kout << hex_number(hefsi_hash_64(file_name)) << kendl);
505 (Void)(kout << hex_number(node->fHashPath) << kendl);
506
507 if (hefsi_hash_64(file_name) == node->fHashPath && node->fKind == kind) {
508 delete dir;
509 dir = nullptr;
510
511 return node;
512 }
513 }
514 }
515
516 hefsi_traverse_tree(dir, mnt, boot->fStartIND, start);
517 if (start == boot->fStartIND || start == boot->fStartIND) break;
518 }
519
520 delete node;
521 node = nullptr;
522
523 delete dir;
524 dir = nullptr;
525 }
526
527 kout << "Error: Failed to find IN.\r";
528
529 err_global_get() = kErrorFileNotFound;
530
531 return nullptr;
532 }

534 STATIC ATTRIBUTE(unused) _Output BOOL
535 hefsi_update_in_status(HEFS_BOOT_NODE* boot, DriveTrait* mnt, const Utf8Char* dir_name,
536 HEFS_INDEX_NODE* node, BOOL delete_or_create) {
537 if (!boot || !mnt) return NO;
538
539 auto start = boot->fStartIND;
540
541 if (start > boot->fEndIND) return NO;
542 if (boot->fStartIN > boot->fEndIN) return NO;
543 ;
544 if (boot->fStartBlock > boot->fEndBlock) return NO;
545
546 if (mnt) {
547 HEFS_INDEX_NODE_DIRECTORY* dir =
548 (HEFS_INDEX_NODE_DIRECTORY*) mm_alloc_ptr(sizeof(HEFS_INDEX_NODE_DIRECTORY), Yes, No);
549
550 auto hash_file = node->fHashPath;
551
552 while (YES) {
553 mnt->fPacket.fPacketLba = start;
554 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
555 mnt->fPacket.fPacketContent = dir;
556
557 mnt->fInput(mnt->fPacket);
558
559 (Void)(kout << hex_number(hefsi_hash_64(dir_name)) << kendl);
560 (Void)(kout << hex_number(dir->fHashPath) << kendl);
561
562 if (hefsi_hash_64(dir_name) == dir->fHashPath) {
563 for (SizeT inode_index = 0UL; inode_index < kHeFSSliceCount; ++inode_index) {
564 if (dir->fINSlices[inode_index] == 0 && !delete_or_create) {
565 dir->fINSlices[inode_index] = boot->fStartIN;
566
567 ++dir->fEntryCount;
568
569 dir->fChecksum = ke_calculate_crc32((Char*) dir, sizeof(HEFS_INDEX_NODE_DIRECTORY));
570
571 mnt->fPacket.fPacketLba = start;
572 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
573 mnt->fPacket.fPacketContent = dir;
574
575 mnt->fOutput(mnt->fPacket);
576
577 auto lba = dir->fINSlices[inode_index];
578
579 node->fOffsetSliceLow = (UInt32) (boot->fStartBlock);
580 node->fOffsetSliceHigh = (UInt32) (boot->fStartBlock >> 32);
581
582 node->fChecksum = ke_calculate_crc32((Char*) node, sizeof(HEFS_INDEX_NODE));
583
584 mnt->fPacket.fPacketLba = lba;
585 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE);
586 mnt->fPacket.fPacketContent = node;
587
588 mnt->fOutput(mnt->fPacket);
589
590 boot->fStartIN += sizeof(HEFS_INDEX_NODE);
591 boot->fStartBlock += kHeFSBlockLen;
592
593 boot->fChecksum = ke_calculate_crc32((Char*) boot, sizeof(HEFS_BOOT_NODE));
594
595 mnt->fPacket.fPacketLba = mnt->fLbaStart;
596 mnt->fPacket.fPacketSize = sizeof(HEFS_BOOT_NODE);
597 mnt->fPacket.fPacketContent = boot;
598
599 mnt->fOutput(mnt->fPacket);
600
601 mm_free_ptr(dir);
602
603 return YES;
604 } else if (dir->fINSlices[inode_index] != 0 && delete_or_create) {
605 auto lba = dir->fINSlices[inode_index];
606
607 HEFS_INDEX_NODE tmp_node{};
608
609 mnt->fPacket.fPacketLba = lba;
610 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE);
611 mnt->fPacket.fPacketContent = &tmp_node;
612
613 mnt->fInput(mnt->fPacket);
614
615 if (tmp_node.fHashPath != hash_file) {
616 continue;
617 }
618
619 node->fOffsetSliceLow = 0;
620 node->fOffsetSliceHigh = 0;
621
622 boot->fStartIN -= sizeof(HEFS_INDEX_NODE);
623 boot->fStartBlock -= kHeFSBlockLen;
624
625 boot->fChecksum = ke_calculate_crc32((Char*) boot, sizeof(HEFS_BOOT_NODE));
626
627 mnt->fPacket.fPacketLba = mnt->fLbaStart;
628 mnt->fPacket.fPacketSize = sizeof(HEFS_BOOT_NODE);
629 mnt->fPacket.fPacketContent = boot;
630
631 mnt->fOutput(mnt->fPacket);
632
633 mnt->fPacket.fPacketLba = lba;
634 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE);
635 mnt->fPacket.fPacketContent = node;
636
637 mnt->fOutput(mnt->fPacket);
638
639 dir->fINSlices[inode_index] = 0;
640
641 if (dir->fEntryCount) --dir->fEntryCount;
642
643 dir->fChecksum = ke_calculate_crc32((Char*) dir, sizeof(HEFS_INDEX_NODE_DIRECTORY));
644
645 mnt->fPacket.fPacketLba = start;
646 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
647 mnt->fPacket.fPacketContent = dir;
648
649 mnt->fOutput(mnt->fPacket);
650
651 mm_free_ptr(dir);
652
653 return YES;
654 }
655 }
656 }
657
658 hefsi_traverse_tree(dir, mnt, boot->fStartIND, start);
659 if (start > boot->fEndIND || start == 0) break;
660 }
661
662 mm_free_ptr(dir);
663 err_global_get() = kErrorFileNotFound;
664 return NO;
665 }
666
667 err_global_get() = kErrorDiskIsFull;
668 return NO;
669 }

675 STATIC ATTRIBUTE(unused) _Output BOOL hefsi_balance_ind(HEFS_BOOT_NODE* boot, DriveTrait* mnt) {
676 if (mnt) {
677 HEFS_INDEX_NODE_DIRECTORY* dir =
678 (HEFS_INDEX_NODE_DIRECTORY*) RTL_ALLOCA(sizeof(HEFS_INDEX_NODE_DIRECTORY));
679
680 auto start = boot->fStartIND;
681
682 while (YES) {
683 if (start == 0UL || start > boot->fEndIND) break;
684
685 mnt->fPacket.fPacketLba = start;
686 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
687 mnt->fPacket.fPacketContent = dir;
688
689 mnt->fInput(mnt->fPacket);
690
691 if (!mnt->fPacket.fPacketGood) {
692 err_global_get() = kErrorDiskIsCorrupted;
693
694 return NO;
695 }
696
697 if (start == boot->fStartIND) {
698 dir->fColor = kHeFSBlack;
699
700 mnt->fPacket.fPacketLba = start;
701 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
702 mnt->fPacket.fPacketContent = dir;
703
704 mnt->fOutput(mnt->fPacket);
705 }
706
707 if (dir->fColor == kHeFSBlack && dir->fChild != 0UL) {
708 dir->fColor = kHeFSRed;
709 hefsi_rotate_tree(start, mnt);
710 } else if (dir->fColor == kHeFSBlack && dir->fChild == 0UL) {
711 dir->fColor = kHeFSBlack;
712
713 mnt->fPacket.fPacketLba = start;
714 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);
715 mnt->fPacket.fPacketContent = dir;
716
717 mnt->fOutput(mnt->fPacket);
718 }
719
720 if (ke_calculate_crc32((Char*) dir, sizeof(HEFS_INDEX_NODE_DIRECTORY)) != dir->fChecksum) {

724 mnt->fPacket.fPacketSize = sizeof(HEFS_INDEX_NODE_DIRECTORY);

726
727 mnt->fOutput(mnt->fPacket);
728 }
729
730 hefsi_traverse_tree(dir, mnt, boot->fStartIND, start);
731 }
732
733 err_global_get() = kErrorSuccess;
734 return YES;
735 }

751_Output Bool HeFileSystemParser::Format(_Input _Output DriveTrait* mnt, _Input const Int32 flags,
752 _Input const Utf8Char* vol_name) {
753 // Verify Disk.
754 mnt->fVerify(mnt->fPacket);
755
756 // if disk isn't good, then error out.
757 if (false == mnt->fPacket.fPacketGood) {
758 err_global_get() = kErrorDiskIsCorrupted;
759 return false;
760 }
761
762 if (drv_std_get_size() < kHeFSMinimumDiskSize) {
763 (Void)(kout << "OpenHeFS recommends at least 128 GiB of free space." << kendl);
764 (Void)(
765 kout
766 << "The OS will still try to format a HeFS disk here anyway, don't expect perfect geometry."
767 << kendl);
768 }
769
770 HEFS_BOOT_NODE* boot = (HEFS_BOOT_NODE*) RTL_ALLOCA(sizeof(HEFS_BOOT_NODE));
771
772 mnt->fPacket.fPacketLba = mnt->fLbaStart;
773 mnt->fPacket.fPacketSize = sizeof(HEFS_BOOT_NODE);
774 mnt->fPacket.fPacketContent = boot;
775
776 mnt->fInput(mnt->fPacket);
777
778 if (!mnt->fPacket.fPacketGood) {
779 err_global_get() = kErrorDiskIsCorrupted;
780
781 return NO;
782 }
783
784 // Check if the disk is already formatted.
785
786 if (KStringBuilder::Equals(boot->fMagic, kHeFSMagic) && boot->fVersion == kHeFSVersion) {
787 if (ke_calculate_crc32((Char*) boot, sizeof(HEFS_BOOT_NODE)) != boot->fChecksum &&
788 boot->fChecksum > 0) {
789 err_global_get() = kErrorDiskIsCorrupted;
790 return NO;
791 }
792
793 err_global_get() = kErrorSuccess;
794 return YES;
795 }
796
797 if (ke_calculate_crc32((Char*) boot, sizeof(HEFS_BOOT_NODE)) != boot->fChecksum &&
798 boot->fChecksum > 0) {
799 err_global_get() = kErrorDiskIsCorrupted;
800 return NO;
801 }
802
803 rt_copy_memory_safe((VoidPtr) "fs/hefs-packet", mnt->fPacket.fPacketMime,
804 rt_string_len("fs/hefs-packet"), sizeof(mnt->fPacket.fPacketMime));
805
806 urt_copy_memory((VoidPtr) vol_name, boot->fVolName, urt_string_len(vol_name) + 1);
807 rt_copy_memory_safe((VoidPtr) kHeFSMagic, boot->fMagic, kHeFSMagicLen - 1, sizeof(boot->fMagic));
808
809 if (mnt->fLbaStart > mnt->fLbaEnd) {
810 err_global_get() = kErrorDiskIsCorrupted;
811
812 return NO;
813 }
814
815 boot->fBadSectors = 0;
816
817 boot->fSectorCount = drv_std_get_sector_count();
818 boot->fSectorSize = mnt->fSectorSz;
819
820 MUST_PASS(boot->fSectorSize);
821
823 const SizeT max_lba = (drv_std_get_size()) / boot->fSectorSize;
824
825 const SizeT dir_max = max_lba / 300; // 5% for directory inodes
826 const SizeT inode_max = max_lba / 400; // 5% for inodes
827
828 boot->fStartIND = mnt->fLbaStart + kHeFSINDStartOffset;
829 boot->fEndIND = boot->fStartIND + dir_max;
830
831 boot->fStartIN = boot->fEndIND;
832 boot->fEndIN = boot->fStartIN + inode_max;
833
834 boot->fStartBlock = boot->fEndIN;
835 boot->fEndBlock = drv_std_get_size();
836
837 boot->fINDCount = 0;
838
839 boot->fDiskSize = drv_std_get_size();
840 boot->fDiskStatus = kHeFSStatusUnlocked;
841
842 boot->fDiskFlags = flags;
843
844 if (mnt->fKind & kMassStorageDrive) {
845 boot->fDiskKind = kHeFSMassStorageDevice;
846 } else if (mnt->fKind & kHeFSOpticalDrive) {
847 boot->fDiskKind = kHeFSOpticalDrive;
848 } else {
849 boot->fDiskKind = kHeFSUnknown;
850 }
851
852 boot->fVersion = kHeFSVersion;
853
854 boot->fVID = kHeFSInvalidVID;
855
856 boot->fChecksum = ke_calculate_crc32((Char*) boot, sizeof(HEFS_BOOT_NODE));
857
858 mnt->fPacket.fPacketLba = mnt->fLbaStart;
859 mnt->fPacket.fPacketSize = sizeof(HEFS_BOOT_NODE);
860 mnt->fPacket.fPacketContent = boot;
861
862 mnt->fOutput(mnt->fPacket);
863
864 (Void)(kout << "Protocol: " << mnt->fProtocol() << kendl);
865 (Void)(kout8 << u8"Volume Name: " << boot->fVolName << kendl8);
866 (Void)(kout << "Start IND: " << hex_number(boot->fStartIND) << kendl);
867 (Void)(kout << "End IND: " << hex_number(boot->fEndIND) << kendl);
868 (Void)(kout << "Start IN: " << hex_number(boot->fStartIN) << kendl);
869 (Void)(kout << "End IN: " << hex_number(boot->fEndIN) << kendl);
870 (Void)(kout << "Start Block: " << hex_number(boot->fStartBlock) << kendl);
871 (Void)(kout << "End Block: " << hex_number(boot->fEndBlock) << kendl);
872 (Void)(kout << "Number of IND: " << hex_number(boot->fINDCount) << kendl);
873 (Void)(kout << "Sector Size: " << hex_number(boot->fSectorSize) << kendl);
874 (Void)(kout << "Drive Kind: " << Detail::hefs_drive_kind_to_string(boot->fDiskKind) << kendl);
875
876 if (!mnt->fPacket.fPacketGood) {
877 err_global_get() = kErrorDiskIsCorrupted;
878 return NO;
879 }
880
881 constexpr const SizeT kHeFSPreallocateCount = 0x6UL;
882
883 const Utf8Char* kFileMap[kHeFSPreallocateCount] = {u8"/", u8"/boot", u8"/system",
884 u8"/network", u8"/devices", u8"/media"};
885
886 for (SizeT i = 0; i < kHeFSPreallocateCount; ++i) {
887 this->CreateINodeDirectory(mnt, kHeFSEncodingFlagsUTF8, kFileMap[i]);
888 }
889
890 err_global_get() = kErrorSuccess;
891
892 return YES;
893}

977_Output Bool HeFileSystemParser::INodeManip(_Input DriveTrait* mnt, VoidPtr block, SizeT block_sz,
978 const Utf8Char* dir, const Utf8Char* name,
979 const UInt8 kind, const BOOL is_input) {
980 if (urt_string_len(dir) > kHeFSFileNameLen) {
981 err_global_get() = kErrorDisk;
982 return NO;
983 }
984
985 if (urt_string_len(name) > kHeFSFileNameLen) {
986 err_global_get() = kErrorDisk;
987 return NO;
988 }
989
990 HEFS_BOOT_NODE* boot = (HEFS_BOOT_NODE*) mm_alloc_ptr(sizeof(HEFS_BOOT_NODE), Yes, No);
991
992 if (!boot) {
993 err_global_get() = kErrorInvalidData;
994 return NO;
995 }
996
997 rt_copy_memory_safe((VoidPtr) "fs/hefs-packet", mnt->fPacket.fPacketMime,
998 rt_string_len("fs/hefs-packet"), sizeof(mnt->fPacket.fPacketMime));
999
1000 mnt->fPacket.fPacketLba = mnt->fLbaStart;
1001 mnt->fPacket.fPacketSize = sizeof(HEFS_BOOT_NODE);
1002 mnt->fPacket.fPacketContent = boot;
1003
1004 mnt->fInput(mnt->fPacket);
1005
1006 if (!KStringBuilder::Equals(boot->fMagic, kHeFSMagic) || boot->fVersion != kHeFSVersion) {
1007 (Void)(kout << "Invalid Boot Node, HeFS partition is invalid." << kendl);
1008 mm_free_ptr((VoidPtr) boot);
1009 err_global_get() = kErrorDisk;
1010 return NO;
1011 }
1012
1013 auto start = Detail::hefsi_fetch_in(boot, mnt, dir, name, kind);
1014
1015 if (start) {
1016 (Void)(kout << hex_number(start->fHashPath) << kendl);
1017 (Void)(kout << hex_number(start->fOffsetSliceLow) << kendl);
1018
1019 if (start->fOffsetSliceLow && start->fHashPath) {
1020 mnt->fPacket.fPacketLba = ((UInt64) start->fOffsetSliceHigh << 32) | start->fOffsetSliceLow;
1021 mnt->fPacket.fPacketSize = block_sz;
1022 mnt->fPacket.fPacketContent = block;
1023
1024 if (is_input) {
1025 mnt->fInput(mnt->fPacket);
1026 } else {
1027 if (start->fFlags & kHeFSFlagsReadOnly) {
1028 mm_free_ptr((VoidPtr) boot);
1029 delete start;
1030
1031 kout << "Error: File is read-only\r";
1032
1033 return NO;
1034 }
1035
1036 mnt->fOutput(mnt->fPacket);
1037 }
1038 }
1039 }
1040
1041 mm_free_ptr((VoidPtr) boot);
1042 delete start;
1043 return YES;
1044}

1052_Output Bool HeFileSystemParser::INodeCtlManip(_Input DriveTrait* mnt, _Input const Int32 flags,
1053 const Utf8Char* dir, const Utf8Char* name,

1075

1077 mm_free_ptr((VoidPtr) node);
1078 err_global_get() = kErrorInvalidData;

1082
1083 rt_copy_memory_safe((VoidPtr) "fs/hefs-packet", mnt->fPacket.fPacketMime,

1102 }

1120 node->fDeleted = delete_or_create ? 1UL : NO;
1121 node->fModified = 0;
1122 node->fSize = 0;
1123 node->fKind = kind;
1124 node->fFlags = flags;
1125 node->fChecksum = 0;

1129

1131 mm_free_ptr((VoidPtr) node);
1132
1133 Detail::hefsi_balance_ind(boot, mnt);
1134
1135 err_global_get() = kErrorSuccess;
1136 return YES;
1137 }
1138
1139 mm_free_ptr((VoidPtr) node);
1140 err_global_get() = kErrorDirectoryNotFound;
1141
1142 return NO;
1143}
