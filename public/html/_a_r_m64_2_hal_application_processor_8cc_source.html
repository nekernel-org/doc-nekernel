48 STATIC Void mp_setup_gic_el0(Void) {
49 ke_dma_write<UInt32>(GICD_BASE, GICD_CTLR, YES);
50
51 UInt32 gicc_ctlr = ke_dma_read<UInt32>(GICC_BASE, GICC_CTLR);
52
53 const UInt8 kEnableSignalInt = 0x1;
54
55 gicc_ctlr |= kEnableSignalInt;
56 gicc_ctlr |= (kEnableSignalInt << 0x1);
57
58 ke_dma_write<UInt32>(GICC_BASE, GICC_CTLR, gicc_ctlr);
59
60 ke_dma_write<UInt32>(GICC_BASE, GICC_PMR, 0xFF);
61
62 UInt32 icfgr = ke_dma_read<UInt32>(GICD_BASE, GICD_ICFGR + (0x20 / 0x10) * 4);
63
64 icfgr |= (0x2 << ((32 % 16) * 2));
65
66 ke_dma_write<UInt32>(GICD_BASE, GICD_ICFGR + (0x20 / 0x10) * 4, icfgr);
67 ke_dma_write<UInt32>(GICD_BASE, GICD_ITARGETSR + (0x20 / 0x04) * 4, 0x2 << ((32 % 4) * 8));
68 ke_dma_write<UInt32>(GICD_BASE, GICD_IPRIORITYR + (0x20 / 0x04) * 4, 0xFF << ((32 % 4) * 8));
69 ke_dma_write<UInt32>(GICD_BASE, GICD_ISENABLER + 4, 0x01);
70 }

72 EXTERN_C BOOL mp_handle_gic_interrupt_el0(Void) {
73 UInt32 interrupt_id = ke_dma_read<UInt32>(GICC_BASE, GICC_IAR);
74
75 if ((interrupt_id & 0x3FF) < 1020) {
76 auto interrupt = interrupt_id & 0x3FF;
77
78 const UInt16 kInterruptScheduler = 0x20;
79
80 (Void)(kout << "SMP: AP: " << hex_number(interrupt) << kendl);
81
82 switch (interrupt) {
83 case kInterruptScheduler: {
84 ke_dma_write<UInt32>(GICC_BASE, GICC_EOIR, interrupt_id);
85 UserProcessHelper::StartScheduling();
86 break;
87 }
88 default: {
89 ke_dma_write<UInt32>(GICC_BASE, GICC_EOIR, interrupt_id);
90 break;
91 }
92 }
93
94 return YES;
95 }
96
97 return NO;
98 }

116EXTERN_C Bool mp_register_task(HAL::StackFramePtr stack_frame, ProcessID thrdid) {
117 MUST_PASS(Detail::kGICEnabled);
118
119 if (!stack_frame) return NO;
120 if (thrdid > kMaxAPInsideSched) return NO;
121
122 const auto process_index = thrdid;
123
124 kHWThread[process_index].mFramePtr = stack_frame;
125 kHWThread[process_index].mThreadID = thrdid;
126
127 return YES;
