18SInt32 _NeMain(SInt32 argc, Char** argv) {

28 MmStrLen("-diutil-output-size") == 0)) {

38 MmStrLen("-diutil-part-name") == 0)) {
39 if ((arg + 1) < argc) {
40 MmCopyMemory((VoidPtr) kDiskName, argv[arg + 1], kDIDiskNameLen);
41 }
42 }
43 }
44
45 // create disk image, by appending an EPM partition to it.
46
47 DI::DI_DISK_IMAGE img{};

53 MmCopyMemory((VoidPtr) img.disk_name, (VoidPtr) kDiskName, kDIDiskNameLen);
54 MmCopyMemory((VoidPtr) img.out_name, (VoidPtr) kOutDisk, kDIDiskNameLen);
55
56 // format disk image to explicit partition map.
57 return DI::DIFormatPartitionEPM(img);
