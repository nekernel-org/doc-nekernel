740typedef struct EfiIOToken {
741 //
742 // If Event is NULL, then blocking I/O is performed.
743 // If Event is not NULL and non-blocking I/O is supported, then non-blocking
744 // I/O is performed, and Event will be signaled when the read request is
745 // completed. The caller must be prepared to handle the case where the
746 // callback associated with Event occurs before the original asynchronous I/O
747 // request call returns.
748 //
749 UInt64 Event;
750
751 //
752 // Defines whether or not the signaled event encountered an error.
753 //
754 UInt64 Status;
755
756 //
757 // For OpenEx(): Not Used, ignored.
758 // For ReadEx(): On input, the size of the Buffer. On output, the amount of
759 // data returned in Buffer.
760 // In both cases, the size is measured in bytes.
761 // For WriteEx(): On input, the size of the Buffer. On output, the amount of
762 // data actually written.
763 // In both cases, the size is measured in bytes.
764 // For FlushEx(): Not used, ignored.
765 //
766 UInt32 BufferSize;
767
768 //
769 // For OpenEx(): Not Used, ignored.
770 // For ReadEx(): The buffer into which the data is read.
771 // For WriteEx(): The buffer of data to write.
772 // For FlushEx(): Not Used, ignored.
773 //
774 Void* Buffer;
775} EfiIOToken;
