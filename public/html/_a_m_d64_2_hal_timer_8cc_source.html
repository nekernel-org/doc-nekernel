48HardwareTimer::HardwareTimer(UInt64 ms) : fWaitFor(ms) {
49 auto power = PowerFactoryInterface(kHandoverHeader->f_HardwareTables.f_VendorPtr);
50
51 auto hpet = (Detail::HPET_BLOCK*) power.Find(kHPETSignature).Leak().Leak();
52 MUST_PASS(hpet);
53
54 fDigitalTimer = (UInt8*) hpet->address.Address;
55
56 if (hpet->page_protection) {
57 HAL::mm_map_page((VoidPtr) fDigitalTimer, (VoidPtr) fDigitalTimer,
58 HAL::kMMFlagsWr | HAL::kMMFlagsPCD | HAL::kMMFlagsPwt);
59 }
60
61 // if not enabled yet.
62 if (!(*((volatile UInt64*) ((UInt8*) fDigitalTimer + kHPETConfigRegValue)) & (1 << 0))) {
63 *((volatile UInt64*) ((UInt8*) fDigitalTimer + kHPETConfigRegValue)) =
64 *((volatile UInt64*) ((UInt8*) fDigitalTimer + kHPETConfigRegValue)) | (1 << 0) |
65 (1 << 3); // enable timer & one shot conf
66 }
67}

78BOOL HardwareTimer::Wait() noexcept {
79 if (fWaitFor < 1) return NO;
80 if (fWaitFor > 1'000'000) return NO; // max 1000s = 16 minutes
81
82 UInt64 hpet_cap = *((volatile UInt64*) (fDigitalTimer));
83 UInt64 femtoseconds_per_tick = (hpet_cap >> 32);
84
85 if (femtoseconds_per_tick == 0) return NO;
86
87 volatile UInt64* timer = (volatile UInt64*) (fDigitalTimer + kHPETCounterRegValue);
88
89 UInt64 now = *timer;
90
91 UInt64 fs_wait = fWaitFor * 1'000'000'000'000ULL;
92 UInt64 stop_at = now + (fs_wait / femtoseconds_per_tick);
93
94 while (*timer < (stop_at)) asm volatile("pause");
95
96 return YES;
97}
