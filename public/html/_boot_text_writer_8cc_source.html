134Boot::BootTextWriter& Boot::BootTextWriter::_Write(const UInt64& x) {
135 NE_UNUSED(x);
136
137#ifdef __DEBUG__
138 UInt64 y = (x > 0 ? x : -x) / 16;
139 UInt64 h = (x > 0 ? x : -x) % 16;
140
141 if (y) this->_Write(y);
142
143 /* fail if the hex number is not base-16 */
144 if (h > 16) {
145 this->WriteCharacter('?');
146 return *this;
147 }
148
149 if (y == ~0UL) y = -y;
150
151 const char kNumberList[] = "0123456789ABCDEF";
152
153 this->WriteCharacter(kNumberList[h]);
154#endif // ifdef __DEBUG__
