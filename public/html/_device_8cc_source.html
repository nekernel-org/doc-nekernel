16static Kernel::UInt NE_PCIReadRaw(Kernel::UInt bar, Kernel::UShort bus, Kernel::UShort dev,
17 Kernel::UShort fun) {
18 Kernel::UInt target = PCI_ENABLE_BIT | ((Kernel::UInt) bus << 16) | ((Kernel::UInt) dev << 11) |
19 ((Kernel::UInt) fun << 8) | (bar & 0xFC);
20
21 Kernel::HAL::rt_out32((Kernel::UShort) Kernel::PCI::PciConfigKind::ConfigAddress, target);
22
23 Kernel::HAL::rt_wait_400ns();
24
25 return Kernel::HAL::rt_in32((Kernel::UShort) Kernel::PCI::PciConfigKind::ConfigData);
26}

28static Kernel::Void NE_PCISetCfgTarget(Kernel::UInt bar, Kernel::UShort bus, Kernel::UShort dev,
29 Kernel::UShort fun) {
30 Kernel::UInt target = 0x80000000 | ((Kernel::UInt) bus << 16) | ((Kernel::UInt) dev << 11) |
31 ((Kernel::UInt) fun << 8) | (bar & 0xFC);
32
33 Kernel::HAL::rt_out32((Kernel::UShort) Kernel::PCI::PciConfigKind::ConfigAddress, target);
34
35 Kernel::HAL::rt_wait_400ns();
36}

44UInt Device::Read(UInt bar, Size sz) {
45 // Ensure aligned access by masking to 4-byte boundary
46 NE_PCISetCfgTarget(bar & 0xFC, fBus, fDevice, fFunction);
47
48 // Read 4 bytes and shift out the correct value
49 UInt data = HAL::rt_in32((UShort) PciConfigKind::ConfigData);
50
51 if (sz == 4) return data;
52 if (sz == 2) return (data >> ((bar & 2) * 8)) & 0xFFFF;
53 if (sz == 1) return (data >> ((bar & 3) * 8)) & 0xFF;
54
55 return (UShort) PciConfigKind::Invalid;
56}
57
58void Device::Write(UInt bar, UIntPtr data, Size sz) {
59 NE_PCISetCfgTarget(bar & 0xFC, fBus, fDevice, fFunction);
60
61 if (sz == 4) {
62 HAL::rt_out32((UShort) PciConfigKind::ConfigAddress, (UInt) data);
63 } else if (sz == 2) {
64 UInt temp = HAL::rt_in32((UShort) PciConfigKind::ConfigData);
65
66 temp &= ~(0xFFFF << ((bar & 2) * 8));
67 temp |= (data & 0xFFFF) << ((bar & 2) * 8);
68
69 HAL::rt_out32((UShort) PciConfigKind::ConfigAddress, temp);
70 } else if (sz == 1) {
71 UInt temp = HAL::rt_in32((UShort) PciConfigKind::ConfigData);
72
73 temp &= ~(0xFF << ((bar & 3) * 8));
74 temp |= (data & 0xFF) << ((bar & 3) * 8);
75
76 HAL::rt_out32((UShort) PciConfigKind::ConfigAddress, temp);
77 }
78}

108void Device::EnableMmio() {

121UIntPtr Device::Bar(UInt32 bar_in) {
122 UInt32 bar = NE_PCIReadRaw(bar_in, fBus, fDevice, fFunction);
123
124 if (bar & PCI_BAR_IO) return static_cast<UIntPtr>(bar & ~0x03);
125
126 if (bar & PCI_BAR_64) {
127 UInt32 high = NE_PCIReadRaw((bar_in + 4) & ~0x03, fBus, fDevice, fFunction);
128 return (static_cast<UIntPtr>(high) << 32) | (bar & ~0x0F);
129 }
130
131 return static_cast<UIntPtr>(bar & ~0x0F);
132}
