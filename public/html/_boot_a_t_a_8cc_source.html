33STATIC Boolean boot_ata_wait_io(UInt16 IO) {
34 for (int i = 0; i < 400; i++) rt_in8(IO + ATA_REG_STATUS);
35
36ATAWaitForIO_Retry:
37 auto status_rdy = rt_in8(IO + ATA_REG_STATUS);
38
39 if ((status_rdy & ATA_SR_BSY)) goto ATAWaitForIO_Retry;

41ATAWaitForIO_Retry2:

58Boolean boot_ata_init(UInt16 Bus, UInt8 Drive, UInt16& OutBus, UInt8& OutMaster) {
59 NE_UNUSED(Drive);
60
61 if (boot_ata_detected()) return true;
62
63 BootTextWriter writer;
64
65 UInt16 IO = Bus;
66
67 boot_ata_select(IO);
68
69 // Bus init, NEIN bit.
70 rt_out8(IO + ATA_REG_NEIN, 1);
71
72 // identify until it's good.
73ATAInit_Retry:
74 auto status_rdy = rt_in8(IO + ATA_REG_STATUS);
75
76 if (status_rdy & ATA_SR_ERR) {
77 writer.Write(L"VMBoot: ATA: Not an IDE based drive.\r");
78
79 return false;
80 }
81
82 if ((status_rdy & ATA_SR_BSY)) goto ATAInit_Retry;
83
84 boot_ata_select(IO);
85
86 rt_out8(IO + ATA_REG_COMMAND, ATA_CMD_IDENTIFY);
87
90
91 while (!(rt_in8(IO + ATA_REG_STATUS) & ATA_SR_DRQ))
92 ;
93
94 for (SizeT indexData = 0ul; indexData < kATADataLen; ++indexData) {
95 kATAData[indexData] = rt_in16(IO + ATA_REG_DATA);
96 }
97
98 OutBus = (Bus == ATA_PRIMARY_IO) ? BootDeviceATA::kPrimary : BootDeviceATA::kSecondary;
99
100 OutMaster = (Bus == ATA_PRIMARY_IO) ? ATA_MASTER : ATA_SLAVE;
101
102 return true;
103}

105Void boot_ata_read(UInt64 Lba, UInt16 IO, UInt8 Master, CharacterTypeASCII* Buf, SizeT SectorSz,
106 SizeT Size) {
107 Lba /= SectorSz;
108
109 UInt8 Command = ((!Master) ? 0xE0 : 0xF0);
110
111 boot_ata_wait_io(IO);
112 boot_ata_select(IO);
113
114 rt_out8(IO + ATA_REG_HDDEVSEL, (Command) | (((Lba) >> 24) & 0x0F));
115
116 rt_out8(IO + ATA_REG_SEC_COUNT0, ((Size + SectorSz) / SectorSz));
117
118 rt_out8(IO + ATA_REG_LBA0, (Lba) &0xFF);
119 rt_out8(IO + ATA_REG_LBA1, (Lba) >> 8);
120 rt_out8(IO + ATA_REG_LBA2, (Lba) >> 16);
121 rt_out8(IO + ATA_REG_LBA3, (Lba) >> 24);
122
123 rt_out8(IO + ATA_REG_COMMAND, ATA_CMD_READ_PIO);
124
125 while (!(rt_in8(IO + ATA_REG_STATUS) & ATA_SR_DRQ))
126 ;
127
128 for (SizeT IndexOff = 0; IndexOff < Size; IndexOff += 2) {
129 boot_ata_wait_io(IO);
130
131 auto in = rt_in16(IO + ATA_REG_DATA);
132
133 Buf[IndexOff] = in & 0xFF;
134 Buf[IndexOff + 1] = (in >> 8) & 0xFF;
135 boot_ata_wait_io(IO);
136 }
137}

139Void boot_ata_write(UInt64 Lba, UInt16 IO, UInt8 Master, CharacterTypeASCII* Buf, SizeT SectorSz,
140 SizeT Size) {
141 Lba /= SectorSz;
142
143 UInt8 Command = ((!Master) ? 0xE0 : 0xF0);
144
145 boot_ata_wait_io(IO);
146 boot_ata_select(IO);
147
148 rt_out8(IO + ATA_REG_HDDEVSEL, (Command) | (((Lba) >> 24) & 0x0F));
149
150 rt_out8(IO + ATA_REG_SEC_COUNT0, ((Size + (SectorSz)) / SectorSz));
151
152 rt_out8(IO + ATA_REG_LBA0, (Lba) &0xFF);
153 rt_out8(IO + ATA_REG_LBA1, (Lba) >> 8);
154 rt_out8(IO + ATA_REG_LBA2, (Lba) >> 16);
155 rt_out8(IO + ATA_REG_LBA3, (Lba) >> 24);
156
157 rt_out8(IO + ATA_REG_COMMAND, ATA_CMD_WRITE_PIO);
158
159 while (!(rt_in8(IO + ATA_REG_STATUS) & ATA_SR_DRQ))
160 ;
161
162 for (SizeT IndexOff = 0; IndexOff < Size; IndexOff += 2) {
163 boot_ata_wait_io(IO);
164
165 UInt8 low = (UInt8) Buf[IndexOff];
166 UInt8 high = (IndexOff + 1 < Size) ? (UInt8) Buf[IndexOff + 1] : 0;
167 UInt16 packed = (high << 8) | low;
168
169 rt_out16(IO + ATA_REG_DATA, packed);
170
171 boot_ata_wait_io(IO);
172 }
173
174 boot_ata_wait_io(IO);
175}

212BootDeviceATA& BootDeviceATA::Read(CharacterTypeASCII* Buf, SizeT SectorSz) {
