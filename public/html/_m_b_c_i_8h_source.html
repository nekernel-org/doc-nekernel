100inline BOOL busi_test_mmio(_Input struct IMBCIHost* host, _Input const UInt32 test) {
101 host->MMIOTest = test;
102 UInt16 timeout = 0UL;
103
104 while (host->MMIOTest == test) {
105 ++timeout;
106
107 if (timeout > 0x1000) return NO;
108 }
109
110 return host->MMIOTest == 0;
111}

116inline MBCIAuthKeyType mbci_read_auth_key(_Input struct IMBCIHost* host) {
117 auto const kChallengeMBCI = 0x1; // MBCI Challenge test
118
119 if (!busi_test_mmio(host, kChallengeMBCI)) return ~0;
120
121 return (host->Esb[kMBCIESBSz - 1] << 16) | (host->Esb[kMBCIESBSz - 2] << 8) |
122 (host->Esb[kMBCIESBSz - 3] & 0xFF);
123}
