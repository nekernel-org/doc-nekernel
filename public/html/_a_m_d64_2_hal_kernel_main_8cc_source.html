26EXTERN_C Int32 hal_init_platform(Kernel::HEL::BootInfoHeader* handover_hdr) {
27 using namespace Kernel;
28
29 if (handover_hdr->f_Magic != kHandoverMagic && handover_hdr->f_Version != kHandoverVersion) {
30 return kEfiFail;
31 }
32
33 HAL::rt_sti();
34
35 kHandoverHeader = handover_hdr;
36
37 FB::fb_clear_video();
38
39 fw_init_efi((EfiSystemTable*) handover_hdr->f_FirmwareCustomTables[1]);
40
41 Boot::ExitBootServices(handover_hdr->f_HardwareTables.f_ImageKey,
42 handover_hdr->f_HardwareTables.f_ImageHandle);
43
44 kKernelVM = kHandoverHeader->f_PageStart;
45
46 if (!kKernelVM) {
47 MUST_PASS(kKernelVM);
48 return kEfiFail;
49 }
50
51 hal_write_cr3(kKernelVM);
52
53 /************************************** */
54 /* INITIALIZE BIT MAP. */
55 /************************************** */
56
57 kKernelBitMpSize = kHandoverHeader->f_BitMapSize;
58 kKernelBitMpStart =
59 reinterpret_cast<VoidPtr>(reinterpret_cast<UIntPtr>(kHandoverHeader->f_BitMapStart));
60
61 /************************************** */
62 /* INITIALIZE GDT AND SEGMENTS. */
63 /************************************** */
64
65 STATIC CONST auto kGDTEntriesCount = 8;
66
67 STATIC HAL::Detail::NE_TSS kKernelTSS{};
68
69 kKernelTSS.fRsp0 = (UInt64) kHandoverHeader->f_StackTop;
70 kKernelTSS.fIopb = sizeof(HAL::Detail::NE_TSS);
71
72 /* The GDT, mostly descriptors for user and kernel segments. */
73 STATIC HAL::Detail::NE_GDT_ENTRY ALIGN(0x08) kGDTArray[kGDTEntriesCount] = {
74 {.fLimitLow = 0,
75 .fBaseLow = 0,
76 .fBaseMid = 0,
77 .fAccessByte = 0x00,
78 .fFlags = 0x00,
79 .fBaseHigh = 0}, // Null entry
80 {.fLimitLow = 0x0,
81 .fBaseLow = 0,
82 .fBaseMid = 0,
83 .fAccessByte = 0x9A,
84 .fFlags = 0xAF,
85 .fBaseHigh = 0}, // Kernel code
86 {.fLimitLow = 0x0,
87 .fBaseLow = 0,
88 .fBaseMid = 0,
89 .fAccessByte = 0x92,
90 .fFlags = 0xCF,
91 .fBaseHigh = 0}, // Kernel data
92 {}, // TSS data low
93 {}, // TSS data high
94 {.fLimitLow = 0x0,
95 .fBaseLow = 0,
96 .fBaseMid = 0,
97 .fAccessByte = 0xFA,
98 .fFlags = 0xAF,
99 .fBaseHigh = 0}, // User code
100 {.fLimitLow = 0x0,
101 .fBaseLow = 0,
102 .fBaseMid = 0,
103 .fAccessByte = 0xF2,
104 .fFlags = 0xCF,
105 .fBaseHigh = 0}, // User data
106 };
107
108 kGDTArray[3].fLimitLow = sizeof(HAL::Detail::NE_TSS) - 1;
109 kGDTArray[3].fBaseLow = ((UIntPtr) &kKernelTSS) & 0xFFFF;
110 kGDTArray[3].fBaseMid = (((UIntPtr) &kKernelTSS) >> 16) & 0xFF;
111 kGDTArray[3].fAccessByte = 0x89; // Present, type 9 = 64-bit available TSS
112 kGDTArray[3].fFlags = 0x20 | ((((UIntPtr) &kKernelTSS) >> 24) & 0x0F);
113 kGDTArray[3].fBaseHigh = (((UIntPtr) &kKernelTSS) >> 24) & 0xFF;
114
115 kGDTArray[4].fLimitLow = ((UIntPtr) &kKernelTSS >> 32) & 0xFFFF;
116 kGDTArray[4].fBaseLow = 0;
117 kGDTArray[4].fBaseMid = 0;
118 kGDTArray[4].fAccessByte = 0;
119 kGDTArray[4].fFlags = 0;
120 kGDTArray[4].fBaseHigh = 0;
121
122 // Load memory descriptors.
123 HAL::Register64 gdt_reg;
124
125 gdt_reg.Base = reinterpret_cast<UIntPtr>(kGDTArray);
126 gdt_reg.Limit = (sizeof(HAL::Detail::NE_GDT_ENTRY) * kGDTEntriesCount) - 1;
127
129 HAL::GDTLoader gdt_loader;
130 gdt_loader.Load(gdt_reg);
131
132 return kEfiFail;
133}
