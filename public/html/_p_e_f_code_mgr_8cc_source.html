64PEFLoader::PEFLoader(const Char* path) : fCachedBlob(nullptr), fFatBinary(false), fBad(false) {
65 fFile.New(const_cast<Char*>(path), kRestrictRB);
66 fPath = KStringBuilder::Construct(path).Leak();
67
68 constexpr auto kPefHeader = "PEF_CONTAINER";
69
71 fCachedBlob = fFile->Read(kPefHeader, 0UL);
72
73 PEFContainer* container = reinterpret_cast<PEFContainer*>(fCachedBlob);
74
75 if (container->Cpu == Detail::ldr_get_platform() && container->Magic[0] == kPefMagic[0] &&
76 container->Magic[1] == kPefMagic[1] && container->Magic[2] == kPefMagic[2] &&
77 container->Magic[3] == kPefMagic[3] && container->Magic[4] == kPefMagic[4] &&
78 container->Abi == kPefAbi) {
79 return;
80 } else if (container->Magic[4] == kPefMagic[0] && container->Magic[3] == kPefMagic[1] &&
81 container->Magic[2] == kPefMagic[2] && container->Magic[1] == kPefMagic[3] &&
82 container->Magic[0] == kPefMagic[4] && container->Abi == kPefAbi) {
84 this->fFatBinary = true;
85 return;
86 }
87
88 fBad = true;
89
90 if (fCachedBlob) mm_free_ptr(fCachedBlob);
91
92 kout << "PEFLoader: Warning: Executable format error!\r";
93
94 fCachedBlob = nullptr;
95}

111ErrorOr<VoidPtr> PEFLoader::FindSymbol(const Char* name, Int32 kind) {
112 if (!fCachedBlob || fBad || !name) return ErrorOr<VoidPtr>{kErrorInvalidData};
113
114 PEFContainer* container = reinterpret_cast<PEFContainer*>(fCachedBlob);
115
116 auto blob = fFile->Read(name, sizeof(PEFCommandHeader));
117
118 PEFCommandHeader* container_header = reinterpret_cast<PEFCommandHeader*>(blob);
119
120 constexpr auto kMangleCharacter = '$';
121 const Char* kContainerKinds[] = {".code64", ".data64", ".zero64", nullptr};
122
123 ErrorOr<BasicKString<>> error_or_symbol;
124
125 switch (kind) {
126 case kPefCode: {
127 error_or_symbol = KStringBuilder::Construct(kContainerKinds[0]); // code symbol.
128 break;
129 }
130 case kPefData: {
131 error_or_symbol = KStringBuilder::Construct(kContainerKinds[1]); // data symbol.
132 break;
133 }
134 case kPefZero: {
135 error_or_symbol = KStringBuilder::Construct(kContainerKinds[2]); // block starting symbol.
136 break;
137 }
138 default:
139 return ErrorOr<VoidPtr>{kErrorInvalidData};
140 ; // prevent that from the kernel's mode perspective, let that happen if it
141 // were a user process.
142 }
143
144 Char* unconst_symbol = const_cast<Char*>(name);
145
146 for (SizeT i = 0UL; i < rt_string_len(unconst_symbol, kPefNameLen); ++i) {
147 if (unconst_symbol[i] == ' ') {
148 unconst_symbol[i] = kMangleCharacter;
149 }
150 }
151
152 error_or_symbol.Leak().Leak() += name;
153
154 for (SizeT index = 0; index < container->Count; ++index) {
155 if (KStringBuilder::Equals(container_header->Name, error_or_symbol.Leak().Leak().CData())) {
156 if (container_header->Kind == kind) {
157 if (container_header->Cpu != Detail::ldr_get_platform()) {
158 if (!this->fFatBinary) {
159 mm_free_ptr(blob);
160 return ErrorOr<VoidPtr>{kErrorInvalidData};
161 }
162 }
163
164 Char* container_blob_value = new Char[container_header->VMSize];
165
166 rt_copy_memory_safe((VoidPtr) ((Char*) blob + sizeof(PEFCommandHeader)),
167 container_blob_value, container_header->VMSize,
168 container_header->VMSize);
169 mm_free_ptr(blob);
170
171 kout << "PEFLoader: Information: Loaded stub: " << container_header->Name << "!\r";
172
173 auto ret = HAL::mm_map_page((VoidPtr) container_header->VMAddress,
174 (VoidPtr) HAL::mm_get_page_addr(container_blob_value),
175 HAL::kMMFlagsPresent | HAL::kMMFlagsUser);
176
177 if (ret != kErrorSuccess) {
178 mm_free_ptr(container_blob_value);
179 return ErrorOr<VoidPtr>{kErrorInvalidData};
180 }
181
182 return ErrorOr<VoidPtr>{container_blob_value};
183 }
184 }
185 }
186
187 mm_free_ptr(blob);
188 return ErrorOr<VoidPtr>{kErrorInvalidData};
189}

234 ProcessID rtl_create_user_process(PEFLoader& exec, const Int32& process_kind) noexcept {
235 auto errOrStart = exec.FindStart();
236
237 if (errOrStart.Error() != kErrorSuccess) return kSchedInvalidPID;
238
239 auto symname = exec.FindSymbol(kPefNameSymbol, kPefData);
240
241 if (!symname) {
242 symname = ErrorOr<VoidPtr>{(VoidPtr) rt_alloc_string("USER_PROCESS")};
243 }
244
245 auto id =
246 UserProcessScheduler::The().Spawn(reinterpret_cast<const Char*>(symname.Leak().Leak()),
247 errOrStart.Leak().Leak(), exec.GetBlob().Leak().Leak());
248
249 mm_free_ptr(symname.Leak().Leak());
250
251 if (id != kSchedInvalidPID) {
252 auto stacksym = exec.FindSymbol(kPefStackSizeSymbol, kPefData);
253
254 if (!stacksym) {
255 stacksym = ErrorOr<VoidPtr>{(VoidPtr) new UIntPtr(kSchedMaxStackSz)};
256 }
257
258 if ((*(volatile UIntPtr*) stacksym.Leak().Leak()) > kSchedMaxStackSz) {
259 *(volatile UIntPtr*) stacksym.Leak().Leak() = kSchedMaxStackSz;
260 }
261
262 UserProcessScheduler::The().TheCurrentTeam().AsArray()[id].Kind = process_kind;
263 UserProcessScheduler::The().TheCurrentTeam().AsArray()[id].StackSize =
264 *(UIntPtr*) stacksym.Leak().Leak();
265
266 mm_free_ptr(stacksym.Leak().Leak());
267 }
268
269 return id;
270 }
