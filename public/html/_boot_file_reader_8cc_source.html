34Boot::BootFileReader::BootFileReader(const CharacterTypeUTF16* path, EfiHandlePtr ImageHandle) {
35 if (path != nullptr) {
36 SizeT index = 0UL;
37 for (; path[index] != L'\0'; ++index) {
38 mPath[index] = path[index];
39 }
40
41 mPath[index] = 0;
42 }
43
45
46 EFI_GUID guidEfp = EFI_GUID(EFI_SIMPLE_FILE_SYSTEM_PROTOCOL_GUID);
47
48 EfiSimpleFilesystemProtocol* efp = nullptr;
49
50 EfiLoadImageProtocol* img = nullptr;
51 EFI_GUID guidImg = EFI_GUID(EFI_LOADED_IMAGE_PROTOCOL_GUID);
52
53 if (BS->HandleProtocol(ImageHandle, &guidImg, (void**) &img) != kEfiOk) {
54 mWriter.Write(L"BootZ: Handle-Protocol: No-Such-Protocol").Write(L"\r");
55 this->mErrorCode = kNotSupported;
56 }
57
58 if (BS->HandleProtocol(img->DeviceHandle, &guidEfp, (void**) &efp) != kEfiOk) {
59 mWriter.Write(L"BootZ: Handle-Protocol: No-Such-Protocol").Write(L"\r");
60 this->mErrorCode = kNotSupported;
61 return;
62 }
63
65
66 if (efp->OpenVolume(efp, &mRootFs) != kEfiOk) {
67 mWriter.Write(L"BootZ: Fetch-Protocol: No-Such-Volume").Write(L"\r");
68 this->mErrorCode = kNotSupported;
69 return;
70 }
71
72 EfiFileProtocol* fileFs = nullptr;
73
74 if (mRootFs->Open(mRootFs, &fileFs, mPath, kEFIFileRead, kEFIReadOnly) != kEfiOk) {
75 mWriter.Write(L"BootZ: Fetch-Protocol: No-Such-Path: ").Write(mPath).Write(L"\r");
76 this->mErrorCode = kNotSupported;
77
78 fb_render_string("BootZ: PLEASE RECOVER YOUR MINKRNL INSTALL.", 40, 10, RGB(0xFF, 0xFF, 0xFF));
79
80 mRootFs->Close(mRootFs);
81
82 return;
83 }
84
85 mSizeFile = 0;
86 mFile = fileFs;
87 mErrorCode = kOperationOkay;
88}

114Void Boot::BootFileReader::ReadAll(SizeT readUntil, SizeT chunkToRead, UIntPtr out_address) {
115 UInt32 szInfo = sizeof(EfiFileInfo);
116
117 EfiFileInfo newPtrInfo{};
118
119 EFI_GUID kFileInfoGUID = EFI_FILE_INFO_GUID;
120
121 if (mFile->GetInfo(mFile, &kFileInfoGUID, &szInfo, &newPtrInfo) == kEfiOk) {
122 readUntil = newPtrInfo.FileSize;
123 mWriter.Write(L"BootZ: File size: ").Write(readUntil).Write("\r");
124 }
125
126 if (readUntil == 0) {
127 mErrorCode = kNotSupported;
128 return;
129 }
130
131 if (mBlob == nullptr) {
132 if (!out_address) {
133 if (auto err = BS->AllocatePool(EfiLoaderCode, readUntil, (VoidPtr*) &mBlob) != kEfiOk) {
134 mWriter.Write(L"*** error: ").Write(err).Write(L" ***\r");
135 Boot::ThrowError(L"OutOfMemory", L"Out of memory.");
136 }
137 } else {
138 mBlob = (VoidPtr) out_address;
139 }
140 }
141
142 mWriter.Write(L"*** Bytes to read: ").Write(readUntil).Write(L" ***\r");
143
144 UInt64 bufSize = chunkToRead;
145 UInt64 szCnt = 0UL;
146
147 while (szCnt < readUntil) {
148 auto res = mFile->Read(mFile, &bufSize, (VoidPtr) (&((Char*) mBlob)[szCnt]));
149
150 szCnt += bufSize;
151
152 if (res == kBufferTooSmall) {
153 bufSize = chunkToRead;
154 }
155 }
156
157 mSizeFile = szCnt;
158 mErrorCode = kOperationOkay;
159}
