101_Output VoidPtr mm_alloc_ptr(SizeT sz, Bool wr, Bool user, SizeT pad_amount) {
102 auto sz_fix = sz;
103
104 if (sz_fix == 0) return nullptr;
105
106 sz_fix += sizeof(Detail::MM_INFORMATION_BLOCK);
107
108 auto wrapper = kPageMgr.Request(wr, user, No, sz_fix, pad_amount);
109
110 Detail::MM_INFORMATION_BLOCK_PTR heap_info_ptr =
111 reinterpret_cast<Detail::MM_INFORMATION_BLOCK_PTR>(wrapper.VirtualAddress() +
112 sizeof(Detail::MM_INFORMATION_BLOCK));
113
114 heap_info_ptr->fSize = sz_fix;
115 heap_info_ptr->fMagic = kHeapMgrMagic;
116 heap_info_ptr->fCRC32 = 0U; // dont fill it for now.
117 heap_info_ptr->fOffset =
118 reinterpret_cast<UIntPtr>(heap_info_ptr) + sizeof(Detail::MM_INFORMATION_BLOCK);
119 heap_info_ptr->fPage = No;
120 heap_info_ptr->fWriteRead = wr;
121 heap_info_ptr->fUser = user;
122 heap_info_ptr->fPresent = Yes;
123 heap_info_ptr->fPad = pad_amount;
124
125 rt_set_memory(heap_info_ptr->fPadding, 0, kHeapMgrAlignSz);
126
127 auto result = reinterpret_cast<VoidPtr>(heap_info_ptr->fOffset);
128
129 (Void)(kout << "HeapMgr: Registered heap address: "
130 << hex_number(reinterpret_cast<UIntPtr>(heap_info_ptr)) << kendl);
131
132 return result;
133}

138_Output Int32 mm_make_page(VoidPtr heap_ptr) {
139 if (Detail::mm_check_ptr_address(heap_ptr) == No) return kErrorHeapNotPresent;
140
141 Detail::MM_INFORMATION_BLOCK_PTR heap_info_ptr =
142 reinterpret_cast<Detail::MM_INFORMATION_BLOCK_PTR>((UIntPtr) heap_ptr -
143 sizeof(Detail::MM_INFORMATION_BLOCK));
144
145 if (!heap_info_ptr) return kErrorHeapNotPresent;
146
147 heap_info_ptr->fPage = true;
148
149 (Void)(kout << "HeapMgr: Registered page from heap address: "
150 << hex_number(reinterpret_cast<UIntPtr>(heap_info_ptr)) << kendl);
151
152 return kErrorSuccess;
153}

187_Output Int32 mm_free_ptr(VoidPtr heap_ptr) {
188 if (Detail::mm_check_ptr_address(heap_ptr) == No) return kErrorHeapNotPresent;
189
190 Detail::MM_INFORMATION_BLOCK_PTR heap_info_ptr =
191 reinterpret_cast<Detail::MM_INFORMATION_BLOCK_PTR>((UIntPtr) (heap_ptr) -
192 sizeof(Detail::MM_INFORMATION_BLOCK));
193
194 if (heap_info_ptr && heap_info_ptr->fMagic == kHeapMgrMagic) {
195 if (!heap_info_ptr->fPresent) {
196 return kErrorHeapNotPresent;
197 }
198
199 heap_info_ptr->fSize = 0UL;
200 heap_info_ptr->fPresent = No;
201 heap_info_ptr->fOffset = 0;
202 heap_info_ptr->fCRC32 = 0;
203 heap_info_ptr->fWriteRead = No;
204 heap_info_ptr->fUser = No;
205 heap_info_ptr->fMagic = 0;
206 heap_info_ptr->fPad = 0;
207
208 (Void)(kout << "HeapMgr: Freed heap address: "
209 << hex_number(reinterpret_cast<UIntPtr>(heap_info_ptr)) << kendl);
210
211 PTEWrapper page_wrapper(
212 No, No, No,
213 reinterpret_cast<UIntPtr>(heap_info_ptr) - sizeof(Detail::MM_INFORMATION_BLOCK));
214
215 Ref<PTEWrapper> pte_address{page_wrapper};
216
217 kPageMgr.Free(pte_address);
218
219 return kErrorSuccess;
220 }
221
222 return kErrorInternal;
223}

228_Output Boolean mm_is_valid_ptr(VoidPtr heap_ptr) {
229 if (heap_ptr && HAL::mm_is_bitmap(heap_ptr)) {
230 Detail::MM_INFORMATION_BLOCK_PTR heap_info_ptr =
231 reinterpret_cast<Detail::MM_INFORMATION_BLOCK_PTR>((UIntPtr) (heap_ptr) -
232 sizeof(Detail::MM_INFORMATION_BLOCK));
233
234 return (heap_info_ptr && heap_info_ptr->fPresent && heap_info_ptr->fMagic == kHeapMgrMagic);
235 }
236
237 return No;
238}

243_Output Boolean mm_protect_ptr(VoidPtr heap_ptr) {
244 if (heap_ptr) {
245 Detail::MM_INFORMATION_BLOCK_PTR heap_info_ptr =
246 reinterpret_cast<Detail::MM_INFORMATION_BLOCK_PTR>((UIntPtr) heap_ptr -
247 sizeof(Detail::MM_INFORMATION_BLOCK));
248
250 if (heap_info_ptr && heap_info_ptr->fPresent && kHeapMgrMagic == heap_info_ptr->fMagic) {
251 heap_info_ptr->fCRC32 =
252 ke_calculate_crc32((Char*) heap_info_ptr->fOffset, heap_info_ptr->fSize);
253
254 return Yes;
255 }
256 }
257
258 return No;
