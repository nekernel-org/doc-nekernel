62struct JsonStreamReader final {
63 STATIC Json In(const Char* full_array) {
64 auto start_val = '{';
65 auto end_val = '}';
66 Boolean probe_value = false;
67
68 if (full_array[0] != start_val) {
69 if (full_array[0] != '[') return Json::kNull;
70
71 start_val = '[';
72 end_val = ']';
73
74 probe_value = true;
75 }
76
77 SizeT len = rt_string_len(full_array);
78
79 SizeT key_len = 0;
80 SizeT value_len = 0;
81
82 Json type(kJSONMaxLen, kJSONMaxLen);
83
84 for (SizeT i = 1; i < len; ++i) {
85 if (full_array[i] == '\r' || full_array[i] == '\n') continue;
86
87 if (probe_value) {
88 if (full_array[i] == end_val || full_array[i] == ',') {
89 probe_value = false;
90
91 ++value_len;
92 } else {
93 if (full_array[i] == '\'') {
94 type.AsValue().Data()[value_len] = 0;
95 break;
96 }
97
98 type.AsValue().Data()[value_len] = full_array[i];
99
100 ++value_len;
101 }
102 } else {
103 if (start_val == '[') continue;
104
105 if (full_array[i] == ':') {
106 type.AsKey().Data()[key_len] = 0;
107 ++key_len;
108
109 ++i;
110
111 while (full_array[i] == ' ' || full_array[i] == '\t') ++i;
112
113 probe_value = true;
114 } else {
115 type.AsKey().Data()[key_len] = full_array[i];
116
117 ++key_len;
118 }
119 }
120 }
121
122 type.AsValue().Data()[value_len] = 0;
123
124 return type;
125 }
126};
