40ATAWaitForIO_Retry:

45ATAWaitForIO_Retry2:

93Void drv_std_read(UInt64 Lba, UInt16 IO, UInt8 Master, Char* Buf, SizeT SectorSz, SizeT Size) {
94 Lba /= SectorSz;
95
96 if (Size > kib_cast(64)) return;
97
98 UInt8 Command = ((!Master) ? 0xE0 : 0xF0);
99
100 rt_copy_memory((VoidPtr) Buf, (VoidPtr) kReadAddr, Size);
101
102 drv_std_select(IO);
103
104 rt_out8(IO + ATA_REG_HDDEVSEL, (Command) | (((Lba) >> 24) & 0x0F));
105
106 rt_out8(IO + ATA_REG_SEC_COUNT0, ((Size + SectorSz - 1) / SectorSz));
107
108 rt_out8(IO + ATA_REG_LBA0, (Lba) &0xFF);
109 rt_out8(IO + ATA_REG_LBA1, (Lba) >> 8);
110 rt_out8(IO + ATA_REG_LBA2, (Lba) >> 16);
111 rt_out8(IO + ATA_REG_LBA3, (Lba) >> 24);
112
113 Kernel::Detail::PRDEntry* prd =
114 (Kernel::Detail::PRDEntry*) (kATADevice.Bar(0x20) + 4); // The PRDEntry is not correct.
115
116 prd->mAddress = (UInt32) (UIntPtr) kReadAddr;
117 prd->mByteCount = Size - 1;
118 prd->mFlags = 0x8000; // indicate the end of prd.
119
120 rt_out32(kATADevice.Bar(0x20) + 0x04, (UInt32) (UIntPtr) prd);
121
122 rt_out8(kATADevice.Bar(0x20) + ATA_REG_COMMAND, ATA_CMD_READ_DMA);
123
124 rt_out8(kATADevice.Bar(0x20) + 0x00, 0x09); // Start DMA engine
125
126 while (rt_in8(kATADevice.Bar(0x20) + ATA_REG_STATUS) & 0x01)
127 ;
128
129 rt_out8(kATADevice.Bar(0x20) + 0x00, 0x00); // Stop DMA engine
130
131 rt_copy_memory((VoidPtr) kReadAddr, (VoidPtr) Buf, Size);
132
133 delete prd;
134 prd = nullptr;
135}
